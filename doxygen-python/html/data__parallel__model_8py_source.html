22 def Parallelize_GPU(
23  model_helper_obj,
24  input_builder_fun,
25  forward_pass_builder_fun,
26  param_update_builder_fun,
27  devices=range(0, workspace.NumCudaDevices()),
28  rendezvous=None,
29  net_type='dag',
30  broadcast_computed_params=True,
31  optimize_gradient_memory=False,
32 ):
33  '''
34  Function to create a model that can run on many GPUs.
35  model_helper_obj: an object of ModelHelperBase, such as CNNModelHelper
36  input_builder_fun:
37  Function that adds the input operators
38  Note: Remember to instantiate reader outside of this
39  function so all GPUs share same reader object.
40  Signature: input_builder_fun(model)
41  forward_pass_builder_fun:
42  Function to add the operators to the model.
43  Must return list of loss-blob references that
44  are used to build the gradient. Loss scale parameter
45  is passed, as you should scale the loss of your model
46  by 1.0 / the total number of gpus.
47  Signature: forward_pass_builder_fun(model, loss_scale)
48  param_update_builder_fun:
49  Function that adds operators that are run after
50  gradient update, such as updating the weights and
51  weight decaying.
52  Signature: param_update_builder_fun(model)
53  devices: List of GPU ids, such as [0, 1, 2, 3],
54  rendezvous: used for rendezvous in distributed computation, if None
55  then only one node is used. To create rendezvous,
56  use <TBD>.
57  net_type: Network type
58 
59  '''
60  log.info("Parallelizing model for devices: {}".format(devices))
61  extra_workers = 8 if rendezvous is not None else 0 # best-guess
62  model_helper_obj.net.Proto().num_workers = len(devices) * 4 + extra_workers
63  model_helper_obj.net.Proto().type = net_type
64 
65  # Store some information in the model -- a bit ugly
66  model_helper_obj._devices = devices
67  model_helper_obj._rendezvous = rendezvous
68  model_helper_obj._grad_names = []
69 
70  assert isinstance(model_helper_obj, model_helper.ModelHelperBase)
71 
72  # Keep track of params that were in the model before: they are not
73  # data parallel, so we need to handle them separately
74  non_datapar_params = copy.copy(model_helper_obj.params)
75 
76  # Add input and model
77  log.info("Create input and model training operators")
78 
79  losses_by_gpu = {}
80  num_shards = 1 if rendezvous is None else rendezvous['num_shards']
81  loss_scale = 1.0 / (len(devices) * num_shards)
82 
83  for device in devices:
84  device_opt = core.DeviceOption(caffe2_pb2.CUDA, device)
85  with core.DeviceScope(device_opt):
86  with core.NameScope("gpu_{}".format(device)):
87  log.info("Model for GPU: {}".format(device))
88  input_builder_fun(model_helper_obj)
89  losses = forward_pass_builder_fun(model_helper_obj, loss_scale)
90  # Losses are not needed for test net
91  if param_update_builder_fun is not None:
92  assert isinstance(losses, list), \
93  'Model builder function must return list of loss blobs'
94  for loss in losses:
95  assert isinstance(loss, core.BlobReference), \
96  'Model builder func must return list of loss blobs'
97 
98  losses_by_gpu[device] = losses
99 
100  # Create parameter map
101  model_helper_obj._device_grouped_blobs =\
102  _GroupByDevice(devices, model_helper_obj.params, non_datapar_params)
103 
104  # computed params
105  computed_params_grouped =\
106  _GroupByDevice(devices, model_helper_obj.computed_params, [])
107  model_helper_obj._device_grouped_blobs.update(computed_params_grouped)
108 
109  model_helper_obj._param_names =\
110  model_helper_obj._device_grouped_blobs.keys()
111  model_helper_obj._computed_param_names = computed_params_grouped.keys()
112 
113  if (param_update_builder_fun is None):
114  log.info("Parameter update function not defined --> only forward")
115  _InferBlobDevice(model_helper_obj)
116  return
117 
118  log.info("Adding gradient operators")
119  _AddGradientOperators(devices, model_helper_obj, losses_by_gpu)
120 
121  # Group gradients by device and register to blob lookup
122  param_to_grad = model_helper_obj.param_to_grad
123  grads_ordered = [param_to_grad[p] for p in
124  model_helper_obj.params if p in param_to_grad]
125  non_datapar_grads = [param_to_grad[p] for p in non_datapar_params]
126 
127  gradients_grouped = _GroupByDevice(
128  devices,
129  grads_ordered,
130  non_datapar_grads
131  )
132  model_helper_obj._device_grouped_blobs.update(gradients_grouped)
133  model_helper_obj._grad_names = gradients_grouped.keys()
134 
135  _InferBlobDevice(model_helper_obj)
136 
137  log.info("Add gradient all-reduces for SyncSGD")
138  if broadcast_computed_params:
139  _BroadcastComputedParams(devices, model_helper_obj, rendezvous)
140 
141  _AllReduceGradients(
142  devices, model_helper_obj, rendezvous
143  )
144 
145  log.info("Post-iteration operators for updating params")
146  num_shards = 1 if rendezvous is None else rendezvous['num_shards']
147  # The following check is necessary for ring reduce to work
148  if rendezvous is not None:
149  assert num_shards > 1, \
150  "Please use more than one shard for distributed training"
151  for device in devices:
152  device_opt = core.DeviceOption(caffe2_pb2.CUDA, device)
153  with core.DeviceScope(device_opt):
154  with core.NameScope("gpu_{}".format(device)):
155  param_update_builder_fun(model_helper_obj)
156 
157  _InferBlobDevice(model_helper_obj)
158  _AnalyzeOperators(model_helper_obj)
159 
160  # Configure dagnet to run with only one worker on the first iteration,
161  # to prevent concurrency problems with allocs and nccl.
162  arg = model_helper_obj.Proto().arg.add()
163  arg.name = "first_iter_only_one_worker"
164  arg.i = 1
165 
166  # Add initial parameter syncs
167  log.info("Add initial parameter sync")
168  if (rendezvous is not None):
169  _AddDistributedParameterSync(
170  devices,
171  model_helper_obj,
172  model_helper_obj.param_init_net,
173  model_helper_obj.param_init_net,
174  rendezvous,
175  )
176 
177  _SyncParams(devices, model_helper_obj, model_helper_obj.param_init_net)
178 
179  if optimize_gradient_memory:
180  _OptimizeGradientMemory(model_helper_obj, losses_by_gpu, devices)

368 def _AllReduceGradientsDistributed(
369  devices,
370  model,
371  rendezvous,
372 ):
373  num_workers = model.net.Proto().num_workers
374  assert num_workers > 1, "Please specify more than 1 worker"
375  all_reduce_engine = rendezvous['engine']
376 
377  # Make list of gradients in reverse order
378  reverse_ordered_grads = _GetReverseOrderedGrads(model)
379 
380  master_device_opt = core.DeviceOption(caffe2_pb2.CUDA, devices[0])
381  reducing_device_opt = master_device_opt
382 
383  # We need to specify a partial order using control_input to
384  # ensure progress (since all machines need to do same all reduces
385  # in parallel)
386  num_controls = min(4, num_workers - 1)
387 
388  cyclical_controls = []
389  counter = 0
390  nccl_control_blob = None
391 
392  # Note: sorted order to ensure each host puts the operators in
393  # same order.
394  for grad_name in reverse_ordered_grads:
395  master_grad = model._device_grouped_blobs[grad_name][devices[0]]
396  grads_group = model._device_grouped_blobs[grad_name].values()
397 
398  assert master_grad in grads_group
399 
400  # Remark: NCCLReduce does not support in-place modifications
401  # so we need a temporary gradient blob
402  reduced_grad = str(master_grad) + "_red"
403 
404  control_input = None if len(cyclical_controls) < num_controls \
405  else cyclical_controls[counter % num_controls]
406 
407  def allreduce(grads):
408  with core.DeviceScope(reducing_device_opt):
409  comm_world = model.param_init_net.CreateCommonWorld(
410  rendezvous['kv_handler'],
411  "{}_cw".format(grad_name),
412  name="{}_cw_op".format(grad_name),
413  size=rendezvous['num_shards'],
414  rank=rendezvous['shard_id'],
415  engine=rendezvous['engine'],
416  )
417  model.net.Allreduce(
418  inputs=[comm_world] + grads,
419  outputs=grads,
420  engine=all_reduce_engine,
421  control_input=control_input,
422  )
423 
424  if rendezvous['engine'] == 'GLOO':
425  # With Gloo cross GPU and cross machine allreduce
426  # can be executed in a single operation
427  allreduce(grads_group)
428  else:
429  # Step 1: sum gradients from local GPUs to master GPU
430  with core.DeviceScope(master_device_opt):
431  model.ConstantFill(master_grad, reduced_grad, value=0.0)
432 
433  # Temp fix since NCCLReduce does not work
434  model.net.NCCLAllreduce(
435  grads_group,
436  grads_group,
437  control_input=nccl_control_blob,
438  )
439  nccl_control_blob = grads_group[0]
440  model.net.Copy(master_grad, reduced_grad)
441 
442  # Step 2: allreduce between all hosts, between master GPUs
443  allreduce([reduced_grad])
444 
445  with core.DeviceScope(master_device_opt):
446  model.net.Copy(reduced_grad, master_grad)
447 
448  # Step 3: broadcast locally
449  _Broadcast(devices, model, model.net, grad_name)
450 
451  if len(cyclical_controls) < num_controls:
452  cyclical_controls.append(reduced_grad)
453  else:
454  cyclical_controls[counter % num_controls] = reduced_grad
455 
456  counter += 1

459 def _AllReduceGradientsSingleHost(devices, model):
460  """Performs NCCL AllReduce to distribute gradients to all the GPUs."""
461 
462  if len(devices) == 1:
463  return
464 
465  # Gradients in reverse order
466  reverse_ordered_grads = _GetReverseOrderedGrads(model)
467  assert(len(reverse_ordered_grads) > 0)
468 
469  # Now we need to Allreduce gradients on all the GPUs.
470  # Pick GPU #0 as a master GPU.
471  master_device_opt = core.DeviceOption(caffe2_pb2.CUDA, devices[0])
472  last_out = None
473 
474  for grad_name in reverse_ordered_grads:
475  # Group by grads for reduce.
476  grads_group = model._device_grouped_blobs[grad_name].values()
477  assert len(grads_group) == len(devices), \
478  "Each GPU from {}, should have a copy of {}.".format(
479  devices, grad_name)
480 
481  if _IsGPUBlob(model, grad_name):
482  with core.DeviceScope(master_device_opt):
483  assert not isinstance(grads_group[0], core.GradientSlice), \
484  "GPU sync of gradient slices not implemented"
485  model.NCCLAllreduce(
486  grads_group,
487  grads_group,
488  control_input=last_out,
489  )
490  # last_out is used to serialize the execution of nccls
491  last_out = grads_group[0]
492  else:
493  assert not isinstance(grads_group[0], core.GradientSlice), \
494  "Synchronizing gradient slices not supported"
495  with core.DeviceScope(core.DeviceOption(caffe2_pb2.CPU)):
496  # Poor man's allreduce
497  model.Sum(grads_group, grads_group[0])
498  _Broadcast(devices, model, grad_name)

549 def _AnalyzeOperators(model):
550  '''
551  Look at all the operators and check that they do not cross device scopes
552  '''
553  for op in model.Proto().op:
554  if "NCCL" in op.type or "Copy" in op.type:
555  continue
556  if "Allreduce" in op.type and "GLOO" in op.engine:
557  continue
558 
559  op_dev = op.device_option
560  op_gpu = op_dev.cuda_gpu_id
561 
562  # This avoids failing on operators that are only for CPU
563  if op_dev.device_type == caffe2_pb2.CPU:
564  continue
565 
566  namescope = "gpu_{}/".format(op_gpu)
567  for inp in list(op.input) + list(op.output):
568  if inp.startswith("gpu_") and not inp.startswith(namescope):
569  raise Exception(
570  "Blob {} of op {}, should have namescope {}. Op: {}".format(
571  inp, op.type, "gpu_{}/".format(op_gpu), str(op),
572  ))

575 def _InferBlobDevice(model):
576  '''
577  Assign blob to device option based on the operator outputing it
578  '''
579  mapping = {}
580 
581  def map_ops(proto):
582  for op in proto.op:
583  for b in list(op.input) + list(op.output):
584  mapping[b] = op.device_option
585  if op.type.startswith('RecurrentNetwork'):
586  import google.protobuf.text_format as protobuftx
587  step_args = [a for a in op.arg if a.name.endswith("step_net")]
588  for step_arg in step_args:
589  step_proto = caffe2_pb2.NetDef()
590  protobuftx.Merge(step_arg.s, step_proto)
591  map_ops(step_proto)
592  map_ops(model.net.Proto())
593 
594  model._blob_to_device = mapping

607 def _GroupByDevice(devices, params, non_data_params):
608  '''
609  Groups blobs by device, returning a map of [blobname] = {0: BlobRef, 1: ..}.
610  Returns ordered dictionary, ensuring the original order.
611  '''
612  grouped = OrderedDict()
613  # Only consider params that were created to be "data parallel"
614  params = params[len(non_data_params):]
615  assert len(params) % len(devices) == 0,\
616  "There should be equal number of params per device"
617 
618  num_params_per_device = int(len(params) / len(devices))
619 
620  for i, p in enumerate(params):
621  assert isinstance(p, core.BlobReference) or \
622  isinstance(p, core.GradientSlice), \
623  "Param {} is not BlobReference or GradientSlice".format(p)
624 
625  name = stripParamName(p)
626  gpuid = devices[i // num_params_per_device]
627 
628  if isinstance(p, core.BlobReference):
629  assert "gpu_{}/".format(gpuid) in p.GetNameScope(),\
630  "Param {} expected to have namescope 'gpu_{}'".format(str(p), gpuid)
631  else:
632  assert "gpu_{}/".format(gpuid) in p.indices.GetNameScope(),\
633  "Indices {} expected to have namescope 'gpu_{}'".format(str(p), gpuid)
634  assert "gpu_{}/".format(gpuid) in p.values.GetNameScope(),\
635  "Values {} expected to have namescope 'gpu_{}'".format(str(p), gpuid)
636 
637  if name not in grouped:
638  grouped[name] = {}
639  grouped[name][gpuid] = p
640 
641  # Confirm consistency
642  for j, (p, ps) in enumerate(grouped.items()):
643  assert \
644  len(ps) == len(devices), \
