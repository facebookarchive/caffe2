144 // run on different devices. You should then implement the RunOnDevice()

148  public:

150  : OperatorBase(operator_def, ws),

152  // In the constructor, we switch to the device so that the child class
153  // constructors will run on that device.
154  context_.SwitchToDevice(0);

164  // The run function of Operator switches to the device, and then carries out
165  // the actual computation with RunOnDevice(). You should implement RunOnDevice
166  // instead of Run().
167  bool Run(int stream_id = 0) final {
168  try {
169  context_.SwitchToDevice(stream_id);
170  bool started = RunOnDevice();
171  bool finished = context_.FinishDeviceComputation();
172  if (!finished) {
173  // FinishDeviceComputation() returning error basically means that there
174  // is something wrong with the device (like CUDA) that usually cannot be
175  // recovered, so we should log FATAL.
176  LOG(FATAL) << "Computation on device returned error in operator\n"
177  << ProtoDebugString(this->def());
178  }
179  return (started && finished);
180  } catch (EnforceNotMet& err) {
181  err.AppendMessage("Error from operator: \n" + ProtoDebugString(def()));
182  throw;
183  }
184  }
185 
186  bool RunAsync(int stream_id = 0) final {
187  try {
188  context_.SwitchToDevice(stream_id);
189  return RunOnDevice();
190  } catch (EnforceNotMet& err) {
191  err.AppendMessage("Error from operator: \n" + ProtoDebugString(def()));
192  throw;
193  }
194  }
195 
196  virtual bool RunOnDevice() = 0;
197 
198  protected:
199  Context context_;
200 };

225 // Helpers to implement runtime op polymorphism. Often it's convenient to make
226 // an op work on different input types (e.g. i32 vs i64 indices) or special-case
227 // it for particular input size (e.g. ScatterWeightedSum for block size of 1
228 // doesn't need to call Eigen).
229 //
230 // DispatchHelper provides compile-time generation of nested "if" statements,
231 // e.g. `DispatchHelper<FixedValues<1, 4>>::call(this, block_size);`
232 // unrolls into:
233 // if (block_size == 1) {
234 // return DoRunWithValue<1>();
235 // } else if (block_size = 4) {
236 // return DoRunWithValue<4>();
237 // } else {
238 // return DoRunWithValue<-1>();
239 // }`
240 //
241 // DoRunWithValue implementation can use template arguments to do "if"
242 // statements
243 // or proxy to functions in math.h which often provide fixed size
244 // implementation.
245 //
246 // Similarly `TensorTypes<int32_t, int64_t>(this, Input(0))` provides branching
247 // based on type of the first input and calls DoRunWithType.
248 //
249 // Note, that the same instance of Op class is used as the method, not class is
250 // templated. We might consider adding static class-level polymorphism later.
251 //
252 // Convenient macro USE_DISPATCH_HELPER is provided for declaring friendship in
253 // case DoRunWithValue or DoRunWithType are declared non-public.
254 
255 #define USE_DISPATCH_HELPER \
256  template <typename Sizes, typename... ExtraArgs> \
257  friend struct DispatchHelper
