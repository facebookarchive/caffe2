321  def AddOperator(self, op_type, inputs, parameters, *args, **kwargs):
322  """
323  Adds an operator to a model. Use parameters list
324  to specify which operator inputs are model parameters to be
325  optimized.
326 
327  Example of usage:
328 
329  model.SparseLengthsSum(
330  [embedding, indices, lengths],
331  parameters=[embedding],

335  and lengths are not.

340  raise Exception("Some parameters are not inputs: {}".format(
341  map(str, extra_parameters)
342  ))

379 def ExtractPredictorNet(
380  net_proto,
381  input_blobs,
382  output_blobs,
383  device=None,
384  renames=None,
385  disabled_inputs=None
386 ):
387  '''
388  Takes a model net for training and returns a net which can be
389  used for prediction. For example, all gradient operators and
390  input operators are removed.
391  @param net_proto protobuf of the net you want to process (net.Proto())
392  @param input_blobs list/set of blob names that are the inputs of predictor
393  @param output_blobs list/set of blob names that are outputs of predictor
394  @param device optional device option that is assigned
395  @param renames dictionary of blob name to a new name (optional)
396  @param disabled_inputs optional set of blobs that are 'switched off'. This
397  will cause branches with those blobs as inputs to be removed
398  '''
399  predict_net = core.Net(net_proto.name + "_predict")
400  predict_proto = predict_net.Proto()
401 
402  orig_external_inputs = set(net_proto.external_input)
403  orig_external_outputs = set(net_proto.external_output)
404  input_blobs = {str(b) for b in input_blobs}
405  known_blobs = set(orig_external_inputs).union(input_blobs)
406  output_blobs = {str(b) for b in output_blobs}
407  external_inputs = set(input_blobs)
408  external_outputs = set(output_blobs)
409 
410  if disabled_inputs is not None:
411  known_blobs = known_blobs - set(disabled_inputs)
412 
413  ops = list(net_proto.op)
414 
415  # Find the range of ops that we should include
416  try:
417  first_op_with_input = min(
418  [
419  j for j in range(len(ops))
420  if input_blobs.intersection(ops[j].input) and ops[j].type !=
421  'StopGradient'
422  ]
423  )
424  except ValueError:
425  raise Exception("No ops with input={}".format(input_blobs))
426  try:
427  last_op_with_output = max(
428  [
429  j for j in range(len(ops))
430  if output_blobs.intersection(ops[j].output)
431  ]
432  )
433  except ValueError:
434  raise Exception("No ops with output={}".format(output_blobs))
435 
436  def validate_op(op):
437  # Check that the op does not have is_test = 0 set. This is a common
438  # pitfall with SpatialBN op, at lest.
439  for arg in op.arg:
440  if arg.name == "is_test" and arg.i == 0:
441  raise Exception(
442  "A operator had is_test=0, did you try to extract a " +
443  "predictor from a train model (instead of test model)?" +
444  " Op was: {}".format(str(op))
445  )
446 
447  # Iterate through the ops and only include those whose inputs
448  # we can satisfy.
449  for op in ops[first_op_with_input:(last_op_with_output + 1)]:
450  if known_blobs.issuperset(op.input):
451  if device is not None:
452  op.device_option.device_type = device.device_type
453  op.device_option.cuda_gpu_id = device.cuda_gpu_id
454  validate_op(op)
455  predict_proto.op.extend([op])
456  known_blobs.update(op.output)
457  external_inputs.update(
458  set(op.input).intersection(orig_external_inputs)
459  )
460  external_outputs.update(
461  set(op.output).intersection(orig_external_outputs)
462  )
463  else:
464  logging.debug(
465  "Op {} had unknown inputs: {}".format(
466  op.type, set(op.input).difference(known_blobs)
467  )
468  )

470  def rename_list(proto_list):
471  if renames is None:
472  return
473 
474  # proto lists don't support assignments
475  new_list = proto_list[:]
476  for j, b in enumerate(new_list):
477  if b in renames:
478  new_list[j] = renames[b]
479 
480  del proto_list[:]
481  proto_list.extend(new_list)
482 
483  # Predictor net's external inputs and outputs include only those
484  # that are part of this net.
485  predict_proto.external_input.extend(external_inputs)
486  predict_proto.external_output.extend(external_outputs)
487 
488  rename_list(predict_proto.external_input)
489  rename_list(predict_proto.external_output)
490 
491  for op in predict_proto.op:
492  rename_list(op.input)
493  rename_list(op.output)
494 
495  return predict_net
