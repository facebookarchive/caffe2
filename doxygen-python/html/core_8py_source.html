163  def __init__(self, name, net=None):

165

277 def _RectifyInputOutput(blobs, net=None):
278  """A helper function to rectify the input or output of the CreateOperator
279  interface.
280  """
281  if isinstance(blobs, string_types) or isinstance(blobs, binary_type):
282  # If blobs is a single string, prepend scope.CurrentNameScope()
283  # and put it as a list.
284  # TODO(jiayq): enforce using BlobReference instead of raw strings.
285  return [ScopedBlobReference(blobs, net=net)]
286  elif type(blobs) is BlobReference:
287  # If blob is a BlobReference, simply put it as a list.
288  return [blobs]
289  elif type(blobs) in (list, tuple):
290  # If blob is a list, we go through it and type check.
291  rectified = []
292  for blob in blobs:
293  if isinstance(blob, string_types) or isinstance(blob, binary_type):
294  rectified.append(ScopedBlobReference(blob, net=net))
295  elif type(blob) is BlobReference:
296  rectified.append(blob)
297  else:
298  raise TypeError(
299  "I/O blob #{} of unsupported type: {} of type {}"
300  .format(len(rectified), str(blob), type(blob)))
301  return rectified
302  else:
303  raise TypeError(
304  "Unknown input/output type: %s of type %s." %
305  (str(blobs), type(blobs))

309 def CreateOperator(
310  operator_type,
311  inputs,
312  outputs,
313  name='',
314  control_input=None,
315  device_option=None,
316  arg=None,
317  engine=None,
318  **kwargs
319 ):
320  """A function wrapper that allows one to create operators based on the
321  operator type. The type should be a string corresponding to an operator
322  registered with Caffe2.
323  """
324  operator = caffe2_pb2.OperatorDef()
325  stack = traceback.format_stack()
326  operator.debug_info = "".join(stack[:-1])
327 
328  operator.type = operator_type
329  operator.name = name
330  # Add rectified inputs and outputs
331  inputs = _RectifyInputOutput(inputs)
332  outputs = _RectifyInputOutput(outputs)
333  operator.input.extend([text_type(i) for i in inputs])
334  operator.output.extend([text_type(o) for o in outputs])
335  if control_input:
336  control_input = _RectifyInputOutput(control_input)
337  operator.control_input.extend([text_type(i) for i in control_input])
338  # Set device option:
339  # (1) If device_option is explicitly set, use device_option.
340  # (2) If not, but scope.CurrentDeviceScope() is set,
341  # then we use scope.CurrentDeviceScope().
342  # (3) Otherwise, do not set device option.
343  if device_option is not None:
344  operator.device_option.CopyFrom(device_option)
345  elif scope.CurrentDeviceScope() is not None:
346  operator.device_option.CopyFrom(scope.CurrentDeviceScope())
347  if engine is not None:
348  operator.engine = engine
349  # random seed is defined in the device option, so we need to do special
350  # care.
351 
352  if 'random_seed' in kwargs:
353  operator.device_option.random_seed = kwargs['random_seed']
354  del kwargs['random_seed']
355  # Add given arguments that do not need parsing
356  if arg is not None:
357  operator.arg.extend(arg)
358  # Add all other arguments
359  for key, value in viewitems(kwargs):
360  operator.arg.add().CopyFrom(utils.MakeArgument(key, value))
361 
362  if workspace.IsImmediate():
363  workspace.RunOperatorImmediate(operator)
364  return operator

430 class IR(object):
431  """A simple IR class to keep track of all intermediate representations used
432  in the gradient computation.
433  """
434 
435  def __init__(self, operators):
436  # The IR class holds multiple metadata from the forward pass:
437  # a) ssa: a list of [op, in_versions, out_versions] recording the
438  # input and the output version of each operator, similar
439  # to a normal SSA form.
440  # b) input_count: a dictionary specifying for each blob and
441  # each of its version, how many times it is used as input for another
442  # op.
443  # c) frontier: maintaining the current versions of the blobs
444  # we are having in the workspace, after the execution of all the ops
445  # added to the IR so far. This is useful because if a gradient is
446  # trying to access an earlier version of a blob, we can sanity check
447  # that it is no longer there, and thus throw an error.
448  # d) gradient_frontier: maps the names of blobs to its version that the
449  # gradient corresponds to.
450  # e) gradient_generators: for each blob and each of its version, maps to
451  # a list of operators that generates its gradient together with the
452  # gradient name.
453  self.ssa = []
454  self.input_usages = defaultdict(lambda: defaultdict(list))
455  self.frontier = defaultdict(int)
456  self.gradient_frontier = {}
457  self.gradient_generators = defaultdict(lambda: defaultdict(list))
458  self.out_version_history = defaultdict(list)
459  self.in_version_history = defaultdict(list)
460 
461  for op in operators:
462  self.Play(op)
463 
464  self.SanityCheck(operators)

466  def SanityCheck(self, operators):
467  # Validate StopGradient usage by checking that StopGradient's output
468  # is actually passed forward
469  for op in operators:
470  if op.type == 'StopGradient':
471  if op.output[0] not in self.input_usages:

473 You typically want to specify same input and output for

476  def Play(self, op):
477  """"Adds an op to the current IR, and update the internal states to
478  reflect the blobs and versions after the execution of the op.
479  """
480  # For input, they are the current version in the dict.
481  in_versions = {}
482  for s in op.input:
483  in_versions[s] = self.frontier[s]
484  self.input_usages[s][self.frontier[s]].append(len(self.ssa))
485  self.in_version_history[s].append((op, self.frontier[s]))
486  # For output, they are the current version plus one. If this is a
487  # newly created blob, its version starts with zero.
488  out_versions = {}
489  for s in op.output:
490  if s in self.frontier:
491  self.frontier[s] += 1
492  out_versions[s] = self.frontier[s]
493  self.out_version_history[s].append((op, self.frontier[s]))
494  # Add to SSA for bookkeeping.
495  self.ssa.append(OpSSA(op, in_versions, out_versions))

568  def AppendSparseGenerators(self, sparse_generators):
569  # merge indices and values generators for sparse gradients
570  for name, input_generators in viewitems(sparse_generators):
571  for version, generators in viewitems(input_generators):
572  if len(generators) == 1:
573  # either indices or values are generated (but not both)
574  generator = generators[0]
575  else:
576  # both indices and values are generated
577  assert(len(generators) == 2)
578  op1_i, idx1_i, op1_v, idx1_v, g1 = generators[0]
579  op2_i, idx2_i, op2_v, idx2_v, g2 = generators[1]
580  assert(g1 == g2)
581  assert(op1_i is None or op2_i is None)
582  assert(op1_v is None or op2_v is None)
583  assert(idx1_i == 0 or idx2_i == 0)
584  assert(idx1_v == 0 or idx2_v == 0)
585  generator = SparseGradGenMeta(
586  op1_i or op2_i, idx1_i + idx2_i,
587  op1_v or op2_v, idx1_v + idx2_v,
588  g1)

591  def BuildGradientGenerators( # NOQA
592  self, fwd_op_idx, gradient_ops, g_output, g_input):
593  """Updates gradient_generators and gradient_frontier"""
594  forward_op, in_versions, out_versions = self.ssa[fwd_op_idx]
595  locally_generated_blobs = []
596  sparse_generators = defaultdict(lambda: defaultdict(list))
597 
598  for grad_op in gradient_ops:
599  # (1) check that inputs are valid
600  for s in grad_op.input:
601  self.CheckGradientOperatorInput(
602  s, g_output, fwd_op_idx, locally_generated_blobs)
603 
604  # (2) add outputs to the locally generated blobs
605  # If an output corresponds to the gradient of an input, we also
606  # record it to gradient_generators
607  locally_generated_blobs.extend([str(s) for s in grad_op.output])
608  for i, output in enumerate(grad_op.output):
609  input_index = GetIndexFromGradientList(g_input, output)
610  if input_index is not None:
611  input_name = forward_op.input[input_index]
612  input_version = in_versions[input_name]
613  g = g_input[input_index]
614  if type(g) is GradientSlice:
615  # the output corresponds either to the indices or the
616  # values of the sparse gradient. In either case we
617  # create a (partial) SparseGradGenMeta. If necessary,
618  # we'll merge indices and values generators
619  # corresponding to the same gradient in step (3)
620  if g.indices == output:
621  m = SparseGradGenMeta(grad_op, i, None, 0, g)
622  else:
623  assert(g.values == output)
624  m = SparseGradGenMeta(None, 0, grad_op, i, g)
625  sparse_generators[input_name][input_version].append(m)
626  else:
627  self.gradient_generators[input_name][input_version] \
628  .append(GradGenMeta(
629  grad_op, i, g))
630 
631  # (3) merge indices and values generators for sparse gradients, and
632  # add them to gradient_generators

846  def DoGradientAccumulation(self, fwd_op_idx):
847  """For each input name in the forward op, check if we will need to
848  add gradient accumulation. If so, do gradient accumulation and return
849  the list of gradient operators.
850 
851  The criteria for doing gradient accumulation is:
852  (1) the specific input version has been used by multiple operators.
853  (2) the current fwd_op_idx is the first to use that input, i.e. in the
854  backward pass, is the last to optionally generate the gradient for
855  the op.
856  (3) For the operators that used the input, their gradient operators
857  have generated more than 1 gradient.
858 
859  When accumulating operators, our current solution is to rename all the
860  created gradients with an internal intermediate name, and then add a
861  Sum() operator that adds up all the gradients. This may use more memory
862  due to intermediate storage, but is usually the fastest approach as one
863  can do one single sum for multiple intermediate gradients.
864  """
865  forward_op, in_versions, out_versions = self.ssa[fwd_op_idx]
866  additional_sum_ops = []
867  grad_map = {}
868  for _i, input_name in enumerate(set(forward_op.input)):
869  input_version = in_versions[input_name]
870  input_usage = self.input_usages[input_name][input_version]
871  if (len(input_usage) <= 1 or fwd_op_idx != input_usage[0]):
872  # We do not need to do gradient accumulation yet.
873  continue
874  generator = self.gradient_generators[input_name][input_version]
875  try:
876  if not self._VerifyGradientGenerators(generator):
877  continue
878  except RuntimeError as err:
879  raise RuntimeError(
880  "Gradients for param ''{}'' failed to verify: {}".format(
881  input_name,
882  err
883  )
884  )
885 
886  # Finally, let's create the sum operator.
887  sum_ops, g = self._MakeSumOps(input_name, input_version)
888  additional_sum_ops.extend(sum_ops)
889  grad_map[input_name] = g
890  return additional_sum_ops, grad_map

956  def GetBackwardPass(self, ys):
957  """Gets the backward pass that computes the derivatives of given blobs.
958 
959  Inputs:
960  ys: a list or a dictionary specifying what blobs we want to compute
961  derivatives of. If the input is a list, we will automatically
962  generate their gradients with all-one values; if the input is a
963  dictionary, for any dictionary entries that are not None, we will
964  take the corresponding blobs as their gradients; for all those
965  that are None, we will auto-fill them with 1.
966  """
967  if isinstance(ys, list):
968  ys = dict((y, None) for y in ys)
969  elif not isinstance(ys, dict):
970  raise TypeError("ys should either be a list or a dict.")
971 
972  # Set the gradient frontier with the initialized external
973  # gradients.
974  for y in viewkeys(ys):
975  self.gradient_frontier[y] = self.frontier[y]
976  self.input_usages[str(y)][self.frontier[str(y)]].append(
977  len(self.ssa))
978 
979  all_input_to_grad, all_gradient_ops = self._GetInitGradients(ys)
980 
981  # (2) Now, after having the virtual play above, we now play the ops
982  # backwards, creating the gradients along the path. Note that although
983  # we are playing it backwards, we cannot refer to variables that are
984  # at a version older than current_versions because it is already been
985  # overwritten.
986  for forward_op_idx in reversed(range(len(self.ssa))):
987  input_to_grad, gradient_ops = self._GenerateGradientsForForwardOp(
988  forward_op_idx, all_input_to_grad)
989  all_input_to_grad.update(input_to_grad)
990  all_gradient_ops += gradient_ops
991 
992  # If there are multiple use blobs, do gradient accumulation.
993  additional_sum_ops, grad_map = self.DoGradientAccumulation(
994  forward_op_idx)
995  # This line is so that if in an accumulation some of the operators
996  # have not produced gradients, they still do not overwrite the
997  # general all_input_to_grad map.
998  all_input_to_grad.update(grad_map)
999  all_gradient_ops += additional_sum_ops
1000 
1001  # (3) Post-processing.
1002  # After we have done computation for each op, we now have the gradient
1003  # operators ready. For the output map, we will convert everything to
1004  # BlobReferences for easier handling in python.
1005  all_input_to_grad_out = {}
1006  for key, val in viewitems(all_input_to_grad):
1007  if val is not None:
1008  if (isinstance(val, string_types) or
1009  isinstance(val, binary_type)):
1010  grad_out = BlobReference(val)
1011  else:
1012  grad_out = GradientSlice(BlobReference(val[0]),
1013  BlobReference(val[1]))
1014  all_input_to_grad_out[BlobReference(key)] = grad_out
1015  return all_gradient_ops, all_input_to_grad_out

1124  output blob used by each operator.
1125
