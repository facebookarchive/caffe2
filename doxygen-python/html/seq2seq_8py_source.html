40 def prepare_batch(batch):
41  encoder_lengths = [len(entry[0]) for entry in batch]
42  max_encoder_length = max(encoder_lengths)
43  decoder_lengths = []
44  max_decoder_length = max([len(entry[1]) for entry in batch])
45 
46  batch_encoder_inputs = []
47  batch_decoder_inputs = []
48  batch_targets = []
49  batch_target_weights = []
50 
51  for source_seq, target_seq in batch:
52  encoder_pads = (
53  [_PAD_ID] * (max_encoder_length - len(source_seq))
54  )
55  batch_encoder_inputs.append(
56  list(reversed(source_seq)) + encoder_pads
57  )
58 
59  decoder_pads = (
60  [_PAD_ID] * (max_decoder_length - len(target_seq))
61  )
62  target_seq_with_go_token = [_GO_ID] + target_seq
63  decoder_lengths.append(len(target_seq_with_go_token))
64  batch_decoder_inputs.append(target_seq_with_go_token + decoder_pads)
65 
66  target_seq_with_eos = target_seq + [_EOS_ID]
67  targets = target_seq_with_eos + decoder_pads
68  batch_targets.append(targets)
69 
70  if len(source_seq) + len(target_seq) == 0:
71  target_weights = [0] * len(targets)
72  else:
73  target_weights = [
74  1 if target != _PAD_ID else 0
75  for target in targets
76  ]
77  batch_target_weights.append(target_weights)
78 
79  return Batch(
80  encoder_inputs=np.array(
81  batch_encoder_inputs,
82  dtype=np.int32,
83  ).transpose(),
84  encoder_lengths=np.array(encoder_lengths, dtype=np.int32),
85  decoder_inputs=np.array(
86  batch_decoder_inputs,
87  dtype=np.int32,
88  ).transpose(),
89  decoder_lengths=np.array(decoder_lengths, dtype=np.int32),
90  targets=np.array(
91  batch_targets,
92  dtype=np.int32,
93  ).transpose(),
94  target_weights=np.array(
95  batch_target_weights,
96  dtype=np.float32,
97  ).transpose(),
98  )

102  def _embedding_encoder(
103  self,
104  model,
105  encoder_type,
106  encoder_params,
107  inputs,
108  input_lengths,
109  vocab_size,
110  embedding_size,
111  use_attention,
112  ):
113  # Initialize the word embeddings that will be learned during training.
114  sqrt3 = math.sqrt(3)
115  encoder_embeddings = model.param_init_net.UniformFill(
116  [],
117  'encoder_embeddings',
118  shape=[vocab_size, embedding_size],
119  min=-sqrt3,
120  max=sqrt3,
121  )
122  model.params.append(encoder_embeddings)
123 
124  # Convert inputs to embedded inputs by the embeddings above.
125  embedded_encoder_inputs = model.net.Gather(
126  [encoder_embeddings, inputs],
127  ['embedded_encoder_inputs'],
128  )
129 
130  if encoder_type == 'rnn':
131  encoder_num_units = (
132  encoder_params['encoder_layer_configs'][0]['num_units']
133  )
134 
135  encoder_initial_cell_state = model.param_init_net.ConstantFill(
136  [],
137  ['encoder_initial_cell_state'],
138  shape=[encoder_num_units],
139  value=0.0,
140  )
141 
142  encoder_initial_hidden_state = model.param_init_net.ConstantFill(
143  [],
144  'encoder_initial_hidden_state',
145  shape=[encoder_num_units],
146  value=0.0,
147  )
148 
149  # Choose corresponding rnn encoder function
150  if encoder_params['use_bidirectional_encoder']:
151  rnn_encoder_func = seq2seq_util.rnn_bidirectional_encoder
152  else:
153  rnn_encoder_func = seq2seq_util.rnn_unidirectional_encoder
154

158  final_encoder_cell_state,
159  ) = rnn_encoder_func(
160  model,
161  embedded_encoder_inputs,
162  input_lengths,
163  encoder_initial_hidden_state,

166  encoder_num_units,
167  use_attention

221  def _build_model(
222  self,
223  init_params,
224  ):
225  model = seq2seq_util.ModelHelper(
226  init_params=init_params,
227  )

229  self.encoder_inputs = model.net.AddExternalInput('encoder_inputs')
230  self.encoder_lengths = model.net.AddExternalInput('encoder_lengths')
231  self.decoder_inputs = model.net.AddExternalInput('decoder_inputs')
232  self.decoder_lengths = model.net.AddExternalInput('decoder_lengths')
233  self.targets = model.net.AddExternalInput('targets')
234  self.target_weights = model.net.AddExternalInput('target_weights')
235 
236  optimizer_params = self.model_params['optimizer_params']
237  attention_type = self.model_params['attention']
238  assert attention_type in ['none', 'regular']
239 
240  self.learning_rate = model.AddParam(
241  name='learning_rate',
242  init_value=float(optimizer_params['learning_rate']),
243  trainable=False,
244  )
245  self.global_step = model.AddParam(
246  name='global_step',
247  init_value=0,
248  trainable=False,
249  )
250  self.start_time = model.AddParam(
251  name='start_time',
252  init_value=time.time(),
253  trainable=False,
254  )
255 
256  assert self.num_gpus < 2
257  assert len(self.encoder_params['encoder_layer_configs']) == 1
258  assert len(self.model_params['decoder_layer_configs']) == 1
259 
260  encoder_num_units = (
261  self.encoder_params['encoder_layer_configs'][0]['num_units']
262  )
263  decoder_num_units = (
264  self.model_params['decoder_layer_configs'][0]['num_units']
265  )

267  (
268  encoder_outputs,
269  final_encoder_hidden_state,
270  final_encoder_cell_state,
271  ) = self._embedding_encoder(
272  model=model,
273  encoder_type=self.encoder_type,
274  encoder_params=self.encoder_params,
275  inputs=self.encoder_inputs,
276  input_lengths=self.encoder_lengths,
277  vocab_size=self.source_vocab_size,
278  embedding_size=self.model_params['encoder_embedding_size'],
279  use_attention=(attention_type != 'none'),
280  )
281 
282  # For bidirectional RNN, the num of units doubles after encodeing
283  if (
284  self.encoder_type == 'rnn' and
285  self.encoder_params['use_bidirectional_encoder']
286  ):
287  encoder_num_units *= 2
288 
289  if attention_type == 'none':
290  decoder_initial_hidden_state = model.FC(
291  final_encoder_hidden_state,
292  'decoder_initial_hidden_state',
293  encoder_num_units,
294  decoder_num_units,
295  axis=2,
296  )
297  decoder_initial_cell_state = model.FC(
298  final_encoder_cell_state,
299  'decoder_initial_cell_state',
300  encoder_num_units,
301  decoder_num_units,
302  axis=2,
303  )
304  else:
305  decoder_initial_hidden_state = model.param_init_net.ConstantFill(
306  [],
307  'decoder_initial_hidden_state',
308  shape=[decoder_num_units],
309  value=0.0,
310  )
311  decoder_initial_cell_state = model.param_init_net.ConstantFill(
312  [],
313  'decoder_initial_cell_state',
314  shape=[decoder_num_units],
315  value=0.0,
316  )
317  initial_attention_weighted_encoder_context = (
318  model.param_init_net.ConstantFill(
319  [],
320  'initial_attention_weighted_encoder_context',
321  shape=[encoder_num_units],
322  value=0.0,
323  )
324  )
325 
326  sqrt3 = math.sqrt(3)
327  decoder_embeddings = model.AddParam(
328  name='decoder_embeddings',
329  init=('UniformFill', dict(
330  shape=[
331  self.target_vocab_size,
332  self.model_params['decoder_embedding_size'],
333  ],
334  min=-sqrt3,
335  max=sqrt3,
336  )),
337  )
338 
339  embedded_decoder_inputs = model.net.Gather(
340  [decoder_embeddings, self.decoder_inputs],
341  ['embedded_decoder_inputs'],
342  )
343  # seq_len x batch_size x decoder_embedding_size
344  with core.NameScope('', reset=True):
345  if attention_type == 'none':
346  decoder_outputs, _, _, _ = recurrent.LSTM(
347  model=model,
348  input_blob=embedded_decoder_inputs,
349  seq_lengths=self.decoder_lengths,
350  initial_states=(
351  decoder_initial_hidden_state,
352  decoder_initial_cell_state,
353  ),
354  dim_in=self.model_params['decoder_embedding_size'],
355  dim_out=decoder_num_units,
356  scope='decoder',
357  outputs_with_grads=[0],
358  )
359  decoder_output_size = decoder_num_units
360  else:
361  (
362  decoder_outputs, _, _, _,
363  attention_weighted_encoder_contexts, _
364  ) = recurrent.LSTMWithAttention(
365  model=model,
366  decoder_inputs=embedded_decoder_inputs,
367  decoder_input_lengths=self.decoder_lengths,
368  initial_decoder_hidden_state=decoder_initial_hidden_state,
369  initial_decoder_cell_state=decoder_initial_cell_state,
370  initial_attention_weighted_encoder_context=(
371  initial_attention_weighted_encoder_context
372  ),
373  encoder_output_dim=encoder_num_units,
374  encoder_outputs=encoder_outputs,
375  decoder_input_dim=self.model_params['decoder_embedding_size'],
376  decoder_state_dim=decoder_num_units,
377  scope='decoder',
378  outputs_with_grads=[0, 4],
379  )
380  decoder_outputs, _ = model.net.Concat(
381  [decoder_outputs, attention_weighted_encoder_contexts],
382  [
383  'states_and_context_combination',
384  '_states_and_context_combination_concat_dims',
385  ],
386  axis=2,
387  )
388  decoder_output_size = decoder_num_units + encoder_num_units
389 
390  # we do softmax over the whole sequence
391  # (max_length in the batch * batch_size) x decoder embedding size
392  # -1 because we don't know max_length yet
393  decoder_outputs_flattened, _ = model.net.Reshape(
394  [decoder_outputs],
395  [
396  'decoder_outputs_flattened',
397  'decoder_outputs_and_contexts_combination_old_shape',
398  ],
399  shape=[-1, decoder_output_size],
400  )
401  output_logits = self.output_projection(
402  model=model,
403  decoder_outputs=decoder_outputs_flattened,
404  decoder_output_size=decoder_output_size,
405  target_vocab_size=self.target_vocab_size,
406  decoder_softmax_size=self.model_params['decoder_softmax_size'],
407  )
408  targets, _ = model.net.Reshape(
409  [self.targets],
410  ['targets', 'targets_old_shape'],
411  shape=[-1],
412  )
413  target_weights, _ = model.net.Reshape(
414  [self.target_weights],
415  ['target_weights', 'target_weights_old_shape'],
416  shape=[-1],
417  )
418 
419  output_probs, loss_per_word = model.net.SoftmaxWithLoss(
420  [output_logits, targets, target_weights],
421  ['OutputProbs', 'loss_per_word'],
422  )
423 
424  num_words = model.net.ReduceFrontSum(
425  target_weights,
426  'num_words',
427  )
428  self.total_loss_scalar = model.net.Mul(
429  [loss_per_word, num_words],
430  'total_loss_scalar',
431  )
432  self.forward_net = model.net.Clone(
433  name=model.net.Name() + '_forward_only',
434  )
435  # print loss only in the forward net which evaluates loss after every
436  # epoch
437  self.forward_net.Print([self.total_loss_scalar], [])
438 
439  # Note: average over batch.
440  # It is tricky because of two problems:
441  # 1. ReduceFrontSum from 1-D tensor returns 0-D tensor
442  # 2. If you want to multiply 0-D by 1-D tensor
443  # (by scalar batch_size_inverse_tensor),
444  # you need to use broadcasting. But gradient propogation
445  # is broken for op with broadcasting.
446  # total_loss_scalar, _ = model.net.Reshape(
447  # [total_loss_scalar],
448  # [total_loss_scalar, 'total_loss_scalar_old_shape'],
449  # shape=[1],
450  # )
451  batch_size_inverse_tensor = (
452  model.param_init_net.ConstantFill(
453  [],
454  'batch_size_tensor',
455  shape=[],
456  value=1.0 / self.batch_size,
457  )
458  )
459  total_loss_scalar_average = model.net.Mul(
460  [self.total_loss_scalar, batch_size_inverse_tensor],
461  ['total_loss_scalar_average'],
462  )
463 
464  model.AddGradientOperators([
465  total_loss_scalar_average,
466  ])
467  ONE = model.param_init_net.ConstantFill(
468  [],
469  'ONE',
470  shape=[1],
471  value=1.0,
472  )
473  logger.info('All trainable variables: ')
474 
475  for param in model.params:
476  param_grad = model.param_to_grad[param]
477  if param in model.param_to_grad:
478  if isinstance(param_grad, core.GradientSlice):
479  param_grad_values = param_grad.values
480  param_grad_values = model.net.Clip(
481  [param_grad_values],
482  [param_grad_values],
483  min=0.0,
484  max=float(self.model_params['max_grad_value']),
485  )
486  model.net.ScatterWeightedSum(
487  [
488  param,
489  ONE,
490  param_grad.indices,
491  param_grad_values,
492  model.net.Negative(
493  [self.learning_rate],
494  'negative_learning_rate',
495  ),
496  ],
497  param,
498  )
499  else:
500  param_grad = model.net.Clip(
501  [param_grad],
502  [param_grad],
503  min=0.0,
504  max=float(self.model_params['max_grad_value']),
505  )
506  model.net.WeightedSum(
507  [
508  param,
509  ONE,
510  param_grad,
511  model.net.Negative(
512  [self.learning_rate],
513  'negative_learning_rate',
514  ),
515  ],
516  param,
517  )
518  self.model = model

599 def gen_vocab(corpus, unk_threshold):
600  vocab = collections.defaultdict(lambda: len(vocab))
601  freqs = collections.defaultdict(lambda: 0)
602  # Adding padding tokens to the vocabulary to maintain consistency with IDs
603  vocab[PAD]
604  vocab[GO]
605  vocab[EOS]
606  vocab[UNK]
607 
608  with open(corpus) as f:
609  for sentence in f:
610  tokens = sentence.strip().split()
611  for token in tokens:
612  freqs[token] += 1
613  for token, freq in freqs.items():
614  if freq > unk_threshold:
615  # TODO: Add reverse lookup dict when it becomes necessary
616  vocab[token]
617 
618  return vocab

621 def gen_batches(source_corpus, target_corpus, source_vocab, target_vocab,
622  batch_size):
623  batches = []
624  with open(source_corpus) as source, open(target_corpus) as target:
625  elems = 0
626  batch = []
627  for source_sentence, target_sentence in zip(source, target):
628 
629  # Convert sentence into list of vocabulary indices
630  def get_numberized_sentence(sentence, vocab):
631  sentence_with_id = []
632  for token in sentence.strip().split():
633  if token in vocab:
634  sentence_with_id.append(vocab[token])
635  else:
636  sentence_with_id.append(vocab[UNK])
637  sentence_with_id.append(vocab[EOS])
638  return sentence_with_id
639 
640  batch.append(tuple([get_numberized_sentence(sentence, vocab)
641  for vocab, sentence in
642  zip([source_vocab, target_vocab],
643  [source_sentence, target_sentence])
644  ]))
645  elems += 1
646  if elems >= batch_size:
647  batches.append(batch)
648  batch = []
649  elems = 0
650  if len(batch) > 0:
651  while len(batch) < batch_size:
652  batch.append(batch[-1])
653  assert len(batch) == batch_size
654  batches.append(batch)
655  return batches

658 def run_seq2seq_model(args, model_params=None):
659  source_vocab = gen_vocab(args.source_corpus, args.unk_threshold)
660  target_vocab = gen_vocab(args.target_corpus, args.unk_threshold)
661  batches = gen_batches(args.source_corpus, args.target_corpus, source_vocab,
662  target_vocab, model_params['batch_size'])
663 
664  logger.info('Source vocab size', len(source_vocab))
665  logger.info('Target vocab size', len(target_vocab))
666 
667  batches_eval = gen_batches(args.source_corpus_eval, args.target_corpus_eval,
668  source_vocab, target_vocab,
669  model_params['batch_size'])
670 
671  with Seq2SeqModelCaffe2(
672  model_params=model_params,
673  source_vocab_size=len(source_vocab),
674  target_vocab_size=len(target_vocab),
675  num_gpus=0,
676  ) as model_obj:
677  model_obj.initialize_from_scratch()
678  for _ in range(args.epochs):
679  for batch in batches:
680  model_obj.step(
681  batch=batch,
682  forward_only=False,
683  )
684  model_obj.step(
685  # merge all eval batches for scalar loss computation
686  batch=[entry for batch in batches_eval for entry in batch],
687  forward_only=True,
