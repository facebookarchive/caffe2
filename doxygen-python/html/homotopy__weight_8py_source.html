29 '''
30 Homotopy Weighting between two weights x, y by doing: alpha x + (1-alpha) y
31 where x is a decreasing scalar parameter ranging from [min, max] (default, [0,
32 1]);
33 
34 Homotopy methods first solves an "easy" problem (one to which the solution is
35 well known), and is gradually transformed into the target problem
36 '''

39 class HomotopyWeight(ModelLayer):
40  def __init__(
41  self,
42  model,
43  input_record,
44  name='homotopy_weight',
45  min_weight=0.,
46  max_weight=1.,
47  half_life=1e6,
48  quad_life=3e6,
49  atomic_iter=None,
50  **kwargs
51  ):
52  super(HomotopyWeight,
53  self).__init__(model, name, input_record, **kwargs)
54  self.output_schema = schema.Scalar(
55  np.float32, self.get_next_blob_reference('homotopy_weight')
56  )
57  data = self.input_record.field_blobs()
58  assert len(data) == 2
59  self.x = data[0]
60  self.y = data[1]
61  # TODO: currently model building does not have access to iter counter or
62  # learning rate; it's added at optimization time;
63  self.use_external_iter = (atomic_iter is not None)
64  self.atomic_iter = (
65  atomic_iter if self.use_external_iter else self.create_atomic_iter()
66  )
67  # to map lr to [min, max]; alpha = scale * lr + offset
68  assert max_weight > min_weight
69  self.scale = float(max_weight - min_weight)
70  self.offset = self.model.add_global_constant(
71  '%s_offset_1dfloat' % self.name, float(min_weight)
72  )
73  self.gamma, self.power = self.solve_inv_lr_params(half_life, quad_life)
74 
75  def solve_inv_lr_params(self, half_life, quad_life):
76  # ensure that the gamma, power is solvable
77  assert half_life > 0
78  # convex monotonically decreasing
79  assert quad_life > 2 * half_life
80  t = float(quad_life) / float(half_life)
81  x = t * (1.0 + np.sqrt(2.0)) / 2.0 - np.sqrt(2.0)
82  gamma = (x - 1.0) / float(half_life)
83  power = np.log(2.0) / np.log(x)
84  logger.info(
85  'homotopy_weighting: found lr param: gamma=%g, power=%g' %
86  (gamma, power)
87  )
88  return gamma, power

93  shape=None,

96  )
97  self.atomic_iter = self.create_param(
98  param_name=('%s_atomic_iter' % self.name),
99  shape=[1],
100  initializer=(
101  'ConstantFill', {
102  'value': 0,
103  'dtype': core.DataType.INT64
104  }
105  ),
106  optimizer=self.model.NoOptim,
107  )

110  def update_weight(self, net):
111  alpha = net.NextScopedBlob('alpha')
112  beta = net.NextScopedBlob('beta')
113  lr = net.NextScopedBlob('lr')
114  comp_lr = net.NextScopedBlob('complementary_lr')
115  scaled_lr = net.NextScopedBlob('scaled_lr')
116  scaled_comp_lr = net.NextScopedBlob('scaled_complementary_lr')
117  if not self.use_external_iter:
118  net.AtomicIter([self.mutex, self.atomic_iter], [self.atomic_iter])
119  net.LearningRate(
120  [self.atomic_iter],
121  [lr],
122  policy='inv',
123  gamma=self.gamma,
124  power=self.power,
125  base_lr=1.0,
126  )
127  net.Sub([self.model.global_constants['ONE'], lr], [comp_lr])
128  net.Scale([lr], [scaled_lr], scale=self.scale)
129  net.Scale([comp_lr], [scaled_comp_lr], scale=self.scale)
130  net.Add([lr, self.offset], [alpha])
131  net.Add([comp_lr, self.offset], beta)
132  return alpha, beta

134  def add_ops(self, net):
135  alpha, beta = self.update_weight(net)
136  # alpha x + beta y
137  net.WeightedSum([self.x, alpha, self.y, beta], self.output_schema())
