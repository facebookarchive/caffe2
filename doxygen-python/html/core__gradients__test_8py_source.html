21 def NeedAll(op, g_output):
22  """A sanity check to make sure that all the gradient are given."""
23  for name, g in zip(op.output, g_output):
24  if g is None:
25  raise RuntimeError(
26  'Need gradient for "%s" but it is not provided.' % name)
27  return g_output
