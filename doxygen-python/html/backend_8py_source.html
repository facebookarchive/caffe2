84 def convertAttributeProto(onnx_arg):
85  """
86  Convert an ONNX AttributeProto into an appropriate Python object
87  for the type.
88 
89  NB: Tensor attribute gets returned as the straight proto.
90  """
91  if onnx_arg.HasField('f'):
92  return onnx_arg.f
93  elif onnx_arg.HasField('i'):
94  return onnx_arg.i
95  elif onnx_arg.HasField('s'):
96  return onnx_arg.s
97  elif onnx_arg.HasField('t'):
98  return onnx_arg.t # this is a proto!
99  elif len(onnx_arg.floats):
100  return list(onnx_arg.floats)
101  elif len(onnx_arg.ints):
102  return list(onnx_arg.ints)
103  elif len(onnx_arg.strings):
104  return list(onnx_arg.strings)
105  else:
106  raise ValueError("Unsupported ONNX attribute: {}".format(onnx_arg))

110 class OnnxNode(object):
111  """
112  Reimplementation of NodeProto from ONNX, but in a form
113  more convenient to work with from Python.
114 
115  We may temporarily edit these nodes to get them into Caffe2 form,
116  before actually translating into the Caffe2 protobuf, since this
117  is easier than decomposing everything, and putting it back together
118  when we're ready.
119  """

123  self.attrs = OnnxAttributes.from_onnx(node.attribute)
124  self.inputs = list(node.input)
125  self.outputs = list(node.output)

133  # The greatest version of the ONNX operator set which we are aware of.
134  # Models whose version is larger than this will cause us to emit a warning
135  # that we are attempting to translate on a "best effort" basis.
136  #
137  # If you increase this, make SURE you cross-reference all BC-breaking
138  # changes from one version to the next, and any that you did not
139  # implement, mark as broken in _broken_operators
140  _known_opset_version = 6
141 
142  # This dictionary will record operators which are KNOWN to be
143  # broken, so we give a good error message rather than do something
144  # bogus and then fail.
145  _broken_operators = {
146  # 'BrokenOp': version_it_was_broken_in
147  }
148 
149  # Operators that are different between Caffe2 and
150  # ONNX but only in their name.
151  # In most cases, this should be empty - as the effort of ONNX is
152  # to unify the operator definitions.
153  _renamed_operators = {
154  'Caffe2ConvTranspose': 'ConvTranspose',
155  'GlobalMaxPool': 'MaxPool',
156  'GlobalAveragePool': 'AveragePool',
157  'Pad': 'PadImage',
158  'Neg': 'Negative',
159  'BatchNormalization': 'SpatialBN',
160  'InstanceNormalization': 'InstanceNorm',
161  'MatMul': 'BatchMatMul',
162  'Upsample': 'ResizeNearest',
163  'Identity': 'Copy',
164  'InstanceNormalization': 'InstanceNorm',
165  'Equal': 'EQ',
166  'Less': 'LT',
167  'Greater': 'GT',
168  'Unsqueeze': 'ExpandDims',
169  }
170 
171  _global_renamed_attrs = {'kernel_shape': 'kernels'}
172  _per_op_renamed_attrs = {
173  'Squeeze': {'axes': 'dims'},
174  'Unsqueeze': {'axes': 'dims'},
175  'Transpose': {'perm': 'axes'},
176  'Upsample': {'mode': ''},
177  'ConvTranspose': {'output_padding': 'adjs'},
178  'Selu': {'gamma': 'scale'},
179  }

181  # operators whose behavior is different beyond renaming
182  # the value is an attribute of this class that is a
183  # function from ToffeIR node_def to caffe2 op_def

190  # NB: By default, you will use the LATEST definition of the operator,
191  # so this interface MAY make BC-breaking changes. Specify an
192  # opset_version if you don't want this to version.
193  @classmethod
194  def run_node(cls, node, inputs, device='CPU', opset_version=_known_opset_version, outputs_info=None):
195  super(Caffe2Backend, cls).run_node(node, inputs, device=device, outputs_info=outputs_info)
196 
197  device_option = get_device_option(Device(device))
198  with Workspace(), core.DeviceScope(device_option): # temporary!
199  if isinstance(inputs, dict):
200  for key, value in inputs.items():
201  workspace.FeedBlob(key, value)
202  else:
203  assert len(node.input) == len(inputs), "{}: expected {} but got {}".format(
204  node.op_type, len(node.input), len(inputs))
205  for key, value in zip(node.input, inputs):
206  workspace.FeedBlob(key, value)
207 
208  ops = []
209  cbackend = C.Caffe2Backend()
210  ops_str = cbackend.convert_node(node.SerializeToString(), opset_version)
211  for s in ops_str[0] + ops_str[1]:
212  op = caffe2_pb2.OperatorDef()
213  op.ParseFromString(s)
214  op.device_option.CopyFrom(device_option)
215  ops.append(op)
216  # For testing
217  if "ONNX_CAFFE2_DEBUG" in os.environ:
218  init_ops, ops2, _ = cls._onnx_node_to_caffe2_op(
219  None, None, node, opset_version or cls._known_opset_version)
220  ops2 = init_ops + ops2
221  for op in ops2:
222  op.device_option.CopyFrom(device_option)
223  print("\nC++:\n{}\nPython:\n{}".format(ops, ops2))
224  workspace.RunOperatorsOnce(ops)
225  output_values = [workspace.FetchBlob(name) for name in node.output]
226  return namedtupledict('Outputs', node.output)(*output_values)

228  @classmethod
229  def _create_tensor_filling_op(cls, onnx_tensor, name=None):
230  """
231  Given an Onnx TensorProto, translate it into a Caffe2 operator
232  which produces the given tensor filling op.
233  """
234  assert name or onnx_tensor.name
235  name = name or onnx_tensor.name
236 
237  c2_op = caffe2_pb2.OperatorDef()
238 
239  c2_values = c2_op.arg.add()
240  c2_values.name = "values"
241 
242  def tensor2list(onnx_tensor):
243  # Use the onnx.numpy_helper because the data may be raw
244  return onnx.numpy_helper.to_array(onnx_tensor).flatten().tolist()
245 
246  if onnx_tensor.data_type in [TensorProto.FLOAT]:
247  c2_op.type = 'GivenTensorFill'
248  c2_values.floats.extend(tensor2list(onnx_tensor))
249  elif onnx_tensor.data_type in [TensorProto.DOUBLE]:
250  c2_op.type = 'GivenTensorDoubleFill'
251  c2_values.floats.extend(tensor2list(onnx_tensor))
252  elif onnx_tensor.data_type in [TensorProto.INT64,
253  TensorProto.UINT32]:
254  c2_op.type = 'GivenTensorInt64Fill'
255  c2_values.ints.extend(tensor2list(onnx_tensor))
256  elif onnx_tensor.data_type in [TensorProto.UINT8,
257  TensorProto.INT8,
258  TensorProto.UINT16,
259  TensorProto.INT16,
260  TensorProto.INT32]:
261  c2_op.type = 'GivenTensorIntFill'
262  c2_values.ints.extend(tensor2list(onnx_tensor))
263  elif onnx_tensor.data_type == TensorProto.BOOL:
264  c2_op.type = 'GivenTensorBoolFill'
265  c2_values.ints.extend(tensor2list(onnx_tensor))
266  elif onnx_tensor.data_type == TensorProto.STRING:
267  c2_op.type = 'GivenTensorStringFill'
268  c2_values.strings.extend(onnx_tensor.string_data)
269  else:
270  raise RuntimeError(
271  "unrecognized tensor type {}".format(onnx_tensor.data_type))
272 
273  c2_shape = c2_op.arg.add()
274  c2_shape.name = "shape"
275  c2_shape.ints.extend(onnx_tensor.dims)
276 
277  c2_op.output.append(name)
278 
279  return c2_op

282  def _rnn_shape_inference(cls, init_model, pred_model, n, input_blob, W):
283  # ad-hoc, informally-specified, bug-ridden, slow
284  # implementation of shape inference
285 
286  # if the weight matrices are directly provided as
287  # initializers, their dimensions should be available in the
288  # init net model.
289  for x in init_model.graph.input:
290  if x.name == W:
291  return x.type.tensor_type.shape.dim[1].dim_value
292 
293  # otherwise, assume that the input_blob is either a direct
294  # graph input, or another rnn op of the same type. This
295  # matches the pattern produced by exporting from pytorch
296  # (where the weight matrices are unusable for this purpose due
297  # to reshaping operations that lose shape information).
298  for x in pred_model.graph.input:
299  if x.name == input_blob:
300  return x.type.tensor_type.shape.dim[2].dim_value
301 
302  curr = n
303  while True:
304  for x in pred_model.graph.input:
305  if x.name == curr.inputs[0] and curr.op_type == 'Gather':
306  return x.type.tensor_type.shape.dim[1].dim_value
307  prev = [x for x in map(OnnxNode, pred_model.graph.node) if x.outputs[0] == curr.inputs[0]]
308  if len(prev) != 1:
309  return
310  prev = prev[0]
311  if prev.op_type == n.op_type:
312  return prev.attrs['hidden_size']
313  if prev.op_type == 'Transpose':
314  for x in pred_model.graph.input:
315  if x.name == prev.inputs[0]:
316  return x.type.tensor_type.shape.dim[2].dim_value
317  curr = prev

343  def make_rnn(direction_offset):
344  name = dummy_name()
345 
346  # input and recurrence biases are squashed together in
347  # onnx but not in caffe2
348

358  init_net.Slice(W, name + '/i2h_w',
359  starts=[weight_offset + 0 * hidden_size, 0],
360  ends =[weight_offset + 1 * hidden_size,-1])
361  init_net.Slice(R, name + '/gates_t_w',
362  starts=[weight_offset + 0 * hidden_size, 0],
363  ends =[weight_offset + 1 * hidden_size,-1])

366  init_net.Slice(initial_h, initial_h_sliced,
367  starts=[direction_offset + 0, 0, 0],
368  ends =[direction_offset + 1,-1,-1])

372  [input_blob, sequence_lens], name + "/input-reversed")
373  else:
374  input = input_blob
375

377  pred_mh,
378  input,
379  sequence_lens,
380  [initial_h_sliced],
381  input_size,
382  hidden_size,
383  name,
384  drop_states=False,
385  forward_only=True,
386  activation=activation
387  )
388 
389  if direction_offset == 1:
390  hidden_t_all = pred_mh.net.ReversePackedSegs(
391  [hidden_t_all, sequence_lens], name + "/output-reversed")
392 
393  return hidden_t_all, hidden_t_last

395  if direction == 'forward':
396  hidden_t_all, hidden_t_last = make_rnn(0)
397

399  # outputs. We need to decouple them so that the
400  # VariableLengthSequencePadding only mutates n.outputs[0]
401  pred_mh.net.Copy(hidden_t_last, n.outputs[1])

413  [n.outputs[1], dummy_name()], axis=0)

417  [n.outputs[0], sequence_lens], [n.outputs[0]])

477  dim0 = i * hidden_size, (i+1) * hidden_size

489  ends =[direction_offset + 1,-1,-1])

493  [input_blob, sequence_lens], name + "/input-reversed")

611  ends =[direction_offset + 1,-1,-1])

615  [input_blob, sequence_lens], name + "/input-reversed")

621  input,
622  sequence_lens,

634  [hidden_t_all, sequence_lens], name + "/output-reversed")

815  @classmethod
816  def _common_onnx_node_to_caffe2_op(cls, init_model, pred_model, onnx_node, opset_version):
817  """
818  This translator performs the basic translation of ONNX nodes into
819  Caffe2 operators. Besides doing a straightforward marshalling from
820  one format to another, it also does these extra things:
821 
822  - Renames operators based on '_renamed_operators'
823  - Renames attributes based on '_global_renamed_attrs' and
824  '_per_op_renamed_attrs'
825 
826  If you're writing a custom translator, consider calling this first,
827  and then fixing things up further.
828  """
829  c2_op = caffe2_pb2.OperatorDef()
830 
831  c2_op.input.extend(onnx_node.inputs)
832  c2_op.output.extend(onnx_node.outputs)
833  c2_op.name = onnx_node.name
834 
835  onnx_op_type = onnx_node.op_type
836  broken_version = cls._broken_operators.get(onnx_op_type, float('Inf'))
837  if broken_version <= opset_version:
838  raise ValueError(
839  "Don't know how to translate op {} in ONNX operator set v{} (I only support prior to v{})".format(onnx_op_type, opset_version, broken_version))
840  c2_op.type = cls._renamed_operators.get(onnx_op_type, onnx_op_type)
841  if not core.IsOperator(c2_op.type):
842  raise ValueError(
843  "Don't know how to translate op {}".format(onnx_op_type))
844 
845  def kmap(k):
846  if (onnx_op_type in cls._per_op_renamed_attrs and
847  k in cls._per_op_renamed_attrs[onnx_op_type]):
848  return cls._per_op_renamed_attrs[onnx_op_type][k]
849  if k in cls._global_renamed_attrs:
850  return cls._global_renamed_attrs[k]
851  return k
852  c2_op.arg.extend(onnx_node.attrs.caffe2(kmap=kmap))
853 
854  return c2_op
