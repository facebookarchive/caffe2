188  template <class ContextForCopy>
189  void Extend(TIndex num, float growthPct, ContextForCopy* context) {
190  CAFFE_ENFORCE_GE(dims_.size(), 1);
191  auto newDims = dims_;
192  newDims[0] += num;
193  if (!data_) {
194  Resize(newDims);
195  return;
196  }
197  auto newSize = std::accumulate(
198  newDims.begin(),
199  newDims.end(),
200  static_cast<TIndex>(1),
201  std::multiplies<TIndex>());
202  if (newSize * meta_.itemsize() <= capacity_) {
203  dims_ = newDims;
204  size_ = newSize;
205  return;
206  }
207  auto newCapacity = dims_;
208  newCapacity[0] = std::max<size_t>(
209  newDims[0], std::ceil(dims_[0] * (growthPct + 100) / 100));
210  Reserve(newCapacity, context);
211  dims_ = newDims;
212  size_ = newSize;
213  }

215  template <class T, class ContextForCopy>
216  void Reserve(const std::vector<T>& newCapacity, ContextForCopy* context) {
217  auto newSize = std::accumulate(
218  newCapacity.begin(),
219  newCapacity.end(),
220  static_cast<TIndex>(1),
221  std::multiplies<TIndex>());
222  if (newSize * meta_.itemsize() <= capacity_) {
223  return;
224  }
225  auto oldData = std::move(data_);
226  auto oldSize = size_;
227  auto oldDims = dims_;
228  Resize(newCapacity);
229  auto* newData = raw_mutable_data(meta_);
230  context->template CopyItems<ContextForCopy, ContextForCopy>(
231  meta_, oldSize, oldData.get(), newData);
232  dims_ = oldDims;
233  size_ = oldSize;
234  }

438  inline void* raw_mutable_data(const TypeMeta& meta) {
439  // For 0-size tensors it's fine to return any pointer (including nullptr)
440  if (meta_ == meta && (data_.get() || size_ == 0)) {
441  return data_.get();
442  } else {
443  meta_ = meta;
444  CAFFE_ENFORCE_GE(
445  size_,
446  0,
447  "Tensor is not initialized. You probably need to call Resize() "
448  "before calling mutable_data()");
449  if (size_ == 0) {
450  return data_.get();
451  }
452  if (meta.ctor()) {
453  // For types that need placement new, we will call it, as well as
454  // making sure that when the data is freed, it calls the right
455  // destruction procedure.
456  auto size = size_;
457  auto dtor = meta_.dtor();
458  data_.reset(
459  static_cast<void*>(Context::New(size_ * meta_.itemsize())),
460  [size, dtor](void* ptr) -> void {
461  dtor(ptr, size);
462  Context::Delete(ptr);
463  });
464  meta_.ctor()(data_.get(), size_);
465  } else {
466  // For fundamental type, new and delete is easier.
467  data_.reset(static_cast<void*>(Context::New(size_ * meta_.itemsize())),
468  Context::Delete);
469  }
470  capacity_ = size_ * meta_.itemsize();
471  return data_.get();
472  }
473  }

718 template <class T>
719 void TensorPrinter::Print(const Tensor<CPUContext>& tensor) {
720  std::stringstream values_stream;
721  // One most likely doesn't want to print int64-number of items for visual
722  // inspection, so we cast down to int here.
723  int total_count = std::min(tensor.size(), TIndex(limit_));
724  const T* tensor_data = tensor.template data<T>();
725  for (int i = 0; i < total_count - 1; ++i) {
726  values_stream << tensor_data[i] << ",";
727  }
728  // We do not add a comma after the last item.
729  values_stream << tensor_data[total_count - 1];
730  if (to_file_) {
731  (*log_file_) << MetaStr(tensor) << values_stream.str() << std::endl;
732  } else {
733  // Log to console.
734  LOG(INFO) << MetaStr(tensor) << values_stream.str();
735  }
