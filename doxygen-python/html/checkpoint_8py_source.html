33 @context.define_context()
34 class Job(object):
35  """
36  A Job defines three TaskGroups: the `init_group`, the `epoch_group` and the
37  `exit_group` which will be run by a JobRunner.
38 
39  The `init_group` will be run only once at startup. Its role is to
40  initialize globally persistent blobs such as model weights, accumulators
41  and data file lists.
42 
43  The `epoch_group` will be run in a loop after init_group. The loop will
44  exit when any of the stop signals added with `add_stop_signal` is True
45  at the end of an epoch.
46 
47  The download_group will be run only once, after all the executions of
48  epoch_group finish. Its role is to collect the distribute scattered
49  parameters back after training.
50 
51  The `exit_group` will be run only once at the very end of the job, the
52  role of this group is to save the results of training in the end of the job.
53 
54  Jobs are context-driven, so that Tasks can be added to the active Job
55  without having to explicitly pass the job object around.
56 
57  Example of usage:
58 
59  def build_reader(partitions):
60  with Job.current().init_group:
61  reader = HiveReader(init_reader, ..., partitions)
62  Task(step=init_reader)
63  with Job.current().epoch_group:
64  limited_reader = ReaderWithLimit(reader, num_iter=10000)
65  data_queue = pipe(limited_reader, num_threads=8)
66  Job.current().add_stop_signal(limited_reader.data_finished())
67  return data_queue
68 
69  def build_hogwild_trainer(reader, model):
70  with Job.current().init_group:
71  Task(step=model.param_init_net)
72  with Job.current().epoch_group:
73  pipe(reader, processor=model, num_threads=8)
74  with Job.current().exit_group:
75  Task(step=model.save_model_net)
76 
77  with Job() as job:
78  reader = build_reader(partitions)
79  model = build_model(params)
80  build_hogwild_trainer(reader, model)
81  """
82  def __init__(self,
83  init_group=None, epoch_group=None,
84  download_group=None, exit_group=None,
85  stop_signals=None, nodes_to_checkpoint=None):
86  self.init_group = init_group or TaskGroup(
87  workspace_type=WorkspaceType.GLOBAL)
88  self.epoch_group = epoch_group or TaskGroup()
89  self.download_group = download_group or TaskGroup()
90  self.exit_group = exit_group or TaskGroup()
91  self.stop_signals = stop_signals or []
92  self._nodes_to_checkpoint = nodes_to_checkpoint
93 
94  def nodes_to_checkpoint(self):
95  if self._nodes_to_checkpoint:
96  return self._nodes_to_checkpoint
97  else:
98  return self.init_group.used_nodes()
99 
100  def compile(self, session_class):
101  return Job(
102  init_group=session_class.compile(self.init_group),
103  epoch_group=session_class.compile(self.epoch_group),
104  download_group=session_class.compile(self.download_group),
105  exit_group=session_class.compile(self.exit_group),
106  stop_signals=self.stop_signals,
107  nodes_to_checkpoint=self.nodes_to_checkpoint())

116  def add_stop_signal(self, output):
117  if isinstance(output, core.BlobReference):
118  t = Task(outputs=[output], group=self.epoch_group)
119  output = t.outputs()[0]
120  assert isinstance(output, TaskOutput)
121  self.stop_signals.append(output)

158 class CheckpointManager(object):
159  """
160  Controls saving and loading of workspaces on every epoch boundary of a job.
161  If a CheckpointManager instance is passed to JobRunner, then JobRunner will
162  call `init`, `read` and `save` at different moments in between epoch runs.
163 
164  Args:
165  db_prefix: The prefix used to construct full db name. Since `absolute_path`
166  is set to True, this will be used as db_name in SaveOp.
167  node_name: Name of the node where this checkpoint_manager is used.
168  db_type: Type of database to use for storing checkpoint.
169  metadata_handler: An optional object capable of reading/writing
170  checkpoint info in storage of choice.
171  """
172  def __init__(self, db_prefix, node_name, db_type, metadata_handler=None):
173  self._db_prefix = db_prefix
174  self._node_name = node_name
175  self._db_type = db_type
176  self._metadata_handler = metadata_handler
177  # make sure these blobs are the first in the checkpoint file.
178  self._net = core.Net('!!checkpoint_mngr')
179  self._blob_names = self._net.AddExternalInput('blob_names')
180  self._names_output = None
181 
182  """
183  Initialize the checkpoint manager. Determines all blobs that need to be saved
184  or loads from a checkpoint.
185 
186  Args:
187  nodes: An array of nodes where this checkpoint manager is running. Should
188  only contain a single node.
189  retrieve_from_epoch: Set to a number to load blobs from this epoch.
190  path_prefix: Used to construct db name or path where checkpoint files are
191  stored.
192  path_type: Indicate the type of path where checkpoint files are stored.
193  """
194  def init(
195  self,
196  nodes=None,
197  retrieve_from_epoch=None,
198  path_prefix=None,
199  path_type=None
200  ):
201  """
202  Build a Task that will be run once after the job's `init_group` is run.
203  This task will determine which blobs need to be checkpointed.
204  If retrieve_from_epoch is not None, then the checkpoint metadata is
205  retrieved from a previously saved checkpoint.
206  """
207  assert nodes is None or len(nodes) == 1, (
208  'CheckpointManager only supports single node.')
209 
210  self._path_prefix = path_prefix
211  self._path_type = path_type
212  if self._metadata_handler:
213  self._metadata_handler.init(
214  db_prefix=self._db_prefix,
215  db_type=self._db_type,
216  node_names=[str(self._node_name)],
217  path_prefix=self._path_prefix,
218  path_type=self._path_type)
219 
220  with Task(outputs=[self._blob_names]) as task:
221  if retrieve_from_epoch is None:
222  ops.GetAllBlobNames(
223  [],
224  self._blob_names,
225  include_shared=False)
226  else:
227  full_db_name = db_name(retrieve_from_epoch,
228  self._node_name, self._db_prefix, path_prefix)
229  db_type = path_type or self._db_type
230  logger.info("Initializing checkpoints from = %s"
231  % full_db_name)
232  ops.Load(
233  [], self._blob_names,
234  db=full_db_name,
235  db_type=db_type,
236  absolute_path=True)
237  self._names_output = task.outputs()[0]
238  return task
239 
240  def blob_list(self):
241  assert self._names_output
242  return self._names_output.fetch().tolist()
243 
244  def load(self, epoch, path_prefix=None, path_type=None):
245  """
246  Build a Task that will be run by JobRunner when the job is to be
247  resumed from a given epoch. This task will run a Load op that will
248  load and deserialize all relevant blobs from a persistent storage.
249  """
250  full_db_name = db_name(epoch, self._node_name, self._db_prefix, path_prefix)
251  db_type = path_type or self._db_type
252  logger.info("Loading checkpoints from = %s" % full_db_name)
253  with Task() as task:
254  ops.Load(
255  [],
256  self.blob_list(),
257  db=full_db_name,
258  db_type=db_type,
259  absolute_path=True)
260  return task
261 
262  def load_blobs_from_checkpoint(self, blob_names, epoch):
263  """
264  Builds a Task that loads only the necessary blobs from a checkpoint of
265  the given epoch. The necessary blobs are given in the blob_names
266  argument.
267 
268  Args:
269  blob_names: A list of strings. Each string is the name of a
270  blob.
271  epoch: The checkpoint epoch to load from.
272 
273  Returns:
274  A Task which loads the specified blobs from the checkpoint of the
275  given epoch.
276  """
277  logger.info('Load from %s' % db_name(epoch, self._node_name, self._db_prefix))
278  with Task() as task:
279  ops.Load(
280  [],
281  blob_names,
282  db=db_name(epoch, self._node_name, self._db_prefix),
283  db_type=self._db_type,
284  absolute_path=True,
285  allow_incomplete=True)
286  return task
287 
288  def check_db_exists(self, epoch):
289  logger.info('Check existence of %s' %
290  db_name(epoch, self._node_name, self._db_prefix))
291  with Task() as task:
292  existence = ops.Const(False)
293  ops.DBExists(
294  [],
295  [existence],
296  db_name=db_name(epoch, self._node_name, self._db_prefix),
297  db_type=self._db_type,
298  absolute_path=True)
299  task.add_output(existence)
300  return task

302  def save(self, epoch):
303  """
304  Build a Task that is run once after `init_group` and after each
305  epoch is run. This will execute a Save ops to serialize and persist
306  blobs present in the global workspace.
307  """
308  logger.info('Saving to %s' % db_name(epoch, self._node_name, self._db_prefix))
309  with Task() as task:
310  ops.Save(
311  self.blob_list(), [],
312  db=db_name(epoch, self._node_name, self._db_prefix),
313  db_type=self._db_type, absolute_path=True)
314  return task

316  def write_checkpoint_metadata(self, epoch):
317  if self._metadata_handler is not None:
318  self._metadata_handler.write(epoch=epoch)
319 
320  def get_resume_from_epoch_id(self, user_epoch=None):
321  last_epoch = user_epoch
322  if self._metadata_handler is not None:
323  last_epoch = self._metadata_handler.last_epoch(user_epoch=user_epoch)
324  return last_epoch

327 class MultiNodeCheckpointManager(object):
328  """
329  Coordinates checkpointing and checkpointing across multiple nodes.
330  Each of `init`, `load` and `save` will build TaskGroups which will
331  trigger checkpointing on each of the nodes involved in a distributed job.
332 
333  Args:
334  db_prefix: The prefix used to construct full db name. Since `absolute_path`
335  is set to True, this will be used as db_name in SaveOp.
336  db_type: Type of database to use for storing checkpoint.
337  metadata_handler: An optional object capable of reading/writing
338  checkpoint info in storage of choice.
339  """
340  def __init__(self, db_prefix, db_type, metadata_handler=None):
341  self._node_managers = None
342  self._db_prefix = db_prefix
343  self._db_type = db_type
344  self._metadata_handler = metadata_handler
345 
346  def _task_group(self, func, *args, **kw):
347  assert self._node_managers is not None, 'init must be called first.'
348  with TaskGroup(WorkspaceType.GLOBAL) as task_group:
349  for node, manager in self._node_managers:
350  with Node(node):
351  func(manager, *args, **kw)
352  return task_group
353 
354  """
355  Args:
356  nodes: An array of nodes where this checkpoint manager is running.
357  retrieve_from_epoch: Set to a number to load blobs from this epoch.
358  path_prefix: Used to construct db name or path where checkpoint files are
359  stored.
360  path_type: Indicate the type of path where checkpoint files are stored.
361  """
362  def init(
363  self, nodes, retrieve_from_epoch=None, path_prefix=None, path_type=None
364  ):
365  if self._node_managers is not None:
366  assert [node for node, _ in self._node_managers] == nodes
367  return TaskGroup(WorkspaceType.GLOBAL)
368  self._node_managers = []
369  self._path_prefix = path_prefix
370  self._path_type = path_type
371  self._node_names = [str(node) for node in nodes]
372  if self._metadata_handler:
373  self._metadata_handler.init(
374  db_prefix=self._db_prefix,
375  db_type=self._db_type,
376  node_names=self._node_names,
377  path_prefix=self._path_prefix,
378  path_type=self._path_type)
379  for node in nodes:
380  with Node(node):
381  manager = CheckpointManager(
382  db_prefix=self._db_prefix,
383  node_name=str(node),
384  db_type=self._db_type)
385  self._node_managers.append((node, manager))
386  return self._task_group(
387  CheckpointManager.init,
388  nodes=[node],
389  retrieve_from_epoch=retrieve_from_epoch,
390  path_prefix=path_prefix,
391  path_type=path_type)
392 
393  def load(self, epoch, path_prefix=None, path_type=None):
394  return self._task_group(
395  CheckpointManager.load,
396  epoch,
397  path_prefix=path_prefix,
398  path_type=path_type)

400  def load_blobs_locally(self, nodes, blob_names, epoch, session):
401  """Loads the necessary blobs from the checkpoints to the current node.
402 
403  Args:
404  blob_names: A list of strings. Each string is the name of a
405  blob.
406  epoch: An integer. The checkpoint epoch to load from.
407  session: A Session object to execute the Load ops.
408  """
409  if self._node_managers is not None:
410  assert [node for node, _ in self._node_managers] == nodes
411  else:
412  self._node_managers = []
413  for node in nodes:
414  with Node(node):
415  manager = CheckpointManager(
416  db_prefix=self._db_prefix,
417  node_name=str(node),
418  db_type=self._db_type)
419  self._node_managers.append((node, manager))
420  assert self._node_managers is not None, 'must initialize node managers'
421  for _, manager in self._node_managers:
422  existence_task = manager.check_db_exists(epoch)
423  session.run(existence_task)
424  existence = existence_task.outputs()[0].fetch()
425  if not existence:
426  logger.info('DB %s does not exist!' %
427  db_name(epoch, manager._node_name, manager._db_prefix))
428  return False
429  load_task = manager.load_blobs_from_checkpoint(blob_names, epoch)
430  session.run(load_task)
431  logger.info('Successfully loaded from checkpoints.')
432  return True

484 class JobRunner(object):
485  """
486  Implement the runtime logic for jobs with checkpointing at the level of
487  epoch. Can be used to run either single-host or distributed jobs. Job
488  runner is a callable to be called once from the master, passing a session
489  as an argument. This call will block until the Job execution is complete.
490 
491  If a checkpoint_manager is passed, checkpoints will be taken after
492  initialization and after each epoch execution. If, in addition,
493  `resume_from_epoch` is an epoch number, the corresponding checkpoint will
494  be loaded and job execution will continue from the given epoch. In
495  this case, the job's init_group will not be run.
496 
497  Refer to checkpoint_test.py for an example.
498  """
499  def __init__(self, job, checkpoint_manager=None, resume_from_epoch=None,
500  upload_task_group_builder=None):
501  """Initializes the JobRunner.
502 
503  Args:
504  job: A Job object. The job to be executed.
505  checkpoint_manager: Can be a CheckpointManager for single machine
506  or a MultiNodeCheckpointManager for multi-machine. The manager
507  that initializes/saves/loads checkpoints.
508  resume_from_epoch: An integer. The epoch to resume from.
509  upload_task_group_builder: A subclass of the
510  UploadTaskGroupBuilder. Creates a task group to upload
511  checkpoints.
512  """
513  self.resume_from_epoch = resume_from_epoch
514  self.checkpoint_manager = checkpoint_manager
515  self.job = job
516  self.upload_task_group_builder = upload_task_group_builder
517 
518  def __call__(self, session):
519  """Runs the training flow.
520 
521  Args:
522  session: A Session object. Valid choises are: LocalSession,
523  LocalHostScheduler, and DistributedSession. It is used to
524  execute one TaskGroup a time.
525  """
526  # identify the epoch we must resume from
527  if self.checkpoint_manager:
528  self.resume_from_epoch = self.checkpoint_manager.\
529  get_resume_from_epoch_id(self.resume_from_epoch)
530  if self.resume_from_epoch is not None:
531  logger.info('Resuming from epoch {}'.format(self.resume_from_epoch))
532 
533  # Initialize all the nodes.
534  from_scratch = self.resume_from_epoch is None
535  if from_scratch:
536  session.run(self.job.init_group)
537 
538  if self.checkpoint_manager:
539  logger.info('Preparing checkpoints ...')
540  session.run(self.checkpoint_manager.init(
541  self.job.nodes_to_checkpoint(),
542  retrieve_from_epoch=self.resume_from_epoch))
543  # Save the first checkpoint before training starts, or resume from
544  # a previously saved checkpoint.
545  if from_scratch:
546  self.save_checkpoints(0, session)
547  else:
548  logger.info('Loading checkpoints for epoch {} ...'.format(
549  self.resume_from_epoch))
550  session.run(
551  self.checkpoint_manager.load(self.resume_from_epoch))
552  logger.info('Checkpoint loaded')
553 
554  logger.info("Finished initializing")
555 
556  # Start training.
557  epoch = 1 if from_scratch else self.resume_from_epoch + 1
558  while True:
559  logger.info('Starting epoch %d' % epoch)
560  session.run(self.job.epoch_group)
561  logger.info('Finished epoch %d' % epoch)
562  stop_signals = [o.fetch() for o in self.job.stop_signals]
563 
564  if self.checkpoint_manager:
565  self.save_checkpoints(epoch, session)
566 
567  if any(stop_signals):
568  logger.info('Stopping')
569  break
570  epoch += 1
571  logger.info('Finished training')
572  # Upload the checkpoints.
573  if (self.upload_task_group_builder):
574  upload_task_group = self.upload_task_group_builder.build(
575  epoch, self.checkpoint_manager)
576  session.run(upload_task_group)
577  logger.info('Finished uploading the checkpoints')
578 
579  # Download the parameters to save
580  session.run(self.job.download_group)
581  logger.info('Finished downloading the parameters')
582 
583  # Finally run the exit step to save nets
584  session.run(self.job.exit_group)
585  logger.info('Finished running the exit group')
586  return epoch

588  def load_blobs_from_checkpoints(self, blob_names, epoch, session):
589  """Loads the necessary blobs from the checkpoints.
590 
591  Checkpoints store the snapshots of the workspace in each node.
592  Sometimes we only need to load a subset of the blobs from the
593  checkpoints. One common scenario is to load only the model blobs from
594  the checkpoints for evaluation purpose. Given the names of the
595  necessary blobs, this function goes over all the checkpoints of all the
596  nodes, but only loads the blobs specified in the blob_names to the
597  current workspace.
598 
599  Args:
600  blob_names: A list of strings. Each string is the name of a
601  blob.
602  epoch: An integer. The checkpoint epoch to load from.
603  session: A Session object to execute the load ops.
604 
605  Raises:
606  ValueError: When the checkpoint manager is invalid.
607  """
608  if not self.checkpoint_manager:
609  raise ValueError('Checkpoint manager is None')
610  logger.info('Loading checkpoint for epoch {} ...'.format(epoch))
611  return self.checkpoint_manager.load_blobs_locally(
612  self.job.nodes_to_checkpoint(), blob_names, epoch, session)
613 
614  def save_checkpoints(self, epoch, session):
615  """Triggers operation to save checkpoints
616 
617  This method will trigger the Save ops to serialize and persist the
618  blobs present in the global workspaace.
619 
620  Args:
621  epoch: An integer. The checkpoint epoch-id that we are saving.
622  session: A Session object to execute the save ops.
623 
624  Raises:
625  ValueError: When the checkpoint manager is invalid.
626  """
627  if not self.checkpoint_manager:
628  raise ValueError('Checkpoint manager is None')
629  try:
630  logger.info('Saving checkpoints for epoch {}'.format(epoch))
631  session.run(self.checkpoint_manager.save(epoch))
632  self.checkpoint_manager.write_checkpoint_metadata(epoch)
633  logger.info('Checkpoints saved')
634  except Exception as ex:
635  logger.warning("Unable to write checkpoint for epoch {}. Error={}".
636  format(epoch, ex))

639 def epoch_limiter(num_epochs):
640  """
641  Creates a task that will output True when a given
642  number of epochs has finished.
