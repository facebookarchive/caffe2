102  def _common_caffe2_arg_to_onnx_attr(cls, op_def, arg):
103  # name
104  op_type = op_def.type
105  if op_type in cls._per_op_renamed_args:
106  name = cls._per_op_renamed_args[op_type].get(
107  arg.name, arg.name)
108  else:
109  name = cls._global_renamed_args.get(arg.name, arg.name)
110 
111  # value
112  if arg.HasField('f'):
113  value = arg.f
114  elif arg.HasField('i'):
115  value = arg.i
116  elif arg.HasField('s'):
117  value = arg.s
118  elif arg.floats:
119  value = arg.floats
120  elif arg.ints:
121  value = arg.ints
122  elif arg.strings:
123  value = arg.strings
124  else:
125  raise ValueError('Could not find data field in arg: {}'.format(arg))
126 
127  if name in cls._blacklist_caffe2_args:
128  assert value in cls._blacklist_caffe2_args[arg.name]
129  return None
130 
131  return helper.make_attribute(name, value)

188  @classmethod
189  def _create_conv_pool_op(cls, op_def, shapes):
190  node = cls._common_caffe2_op_to_onnx_node(op_def, shapes)
191 
192  if node.op_type in ['MaxPool', 'AveragePool']:
193  for i, attr in enumerate(node.attribute):
194  if attr.name == 'global_pooling' and attr.i:
195  node.op_type = 'Global{}'.format(node.op_type)
196  del node.attribute[i]
197  break
198 
199  attrs = {attr.name: attr for attr in node.attribute}
200 
201  def apply_trans(k, dim=2, ks=None):
202  ks = ks or (k + 's')
203  if dim == 2:
204  k_h, k_w = k + '_h', k + '_w'
205  else:
206  k_t, k_l, k_b, k_r = k + '_t', k + '_l', k + '_b', k + '_r'
207 
208  vals = None
209  if (dim == 2 and k_h in attrs and k_w in attrs):
210  vals = [attrs[k_h].i, attrs[k_w].i]
211  del attrs[k_h]
212  del attrs[k_w]
213  elif (dim == 4 and
214  k_t in attrs and k_l in attrs and k_b in attrs and k_r in attrs):
215  vals = [attrs[k_t].i,
216  attrs[k_l].i,
217  attrs[k_b].i,
218  attrs[k_r].i]
219  del attrs[k_t]
220  del attrs[k_l]
221  del attrs[k_b]
222  del attrs[k_r]
223  elif k in attrs:
224  vals = [attrs[k].i] * dim
225  del attrs[k]
226 
227  if vals and not node.op_type.startswith('Global'):
228  attrs[ks] = helper.make_attribute(ks, vals)
229 
230  apply_trans('kernel', ks='kernel_shape')
231  apply_trans('stride')
232  apply_trans('dilation')
233  apply_trans('adj')
234  apply_trans('pad', dim=4)
235 
236  legacy_pad_attr = attrs.pop('legacy_pad', None)
237  if legacy_pad_attr:
238  assert node.op_type.endswith('Pool')
239  input_size = shapes[node.input[0]]
240  output_size = shapes[node.output[0]]
241  assert len(output_size) == 4
242  if node.op_type.startswith('Global'):
243  pass
244  elif legacy_pad_attr.i == caffe2_legacy_pb2.NOTSET:
245  pass
246  elif legacy_pad_attr.i == caffe2_legacy_pb2.VALID:
247  assert not 'pads' in attrs
248  new_attr = make_attribute('auto_pad', 'VALID')
249  attrs[new_attr.name] = new_attr
250  elif legacy_pad_attr.i == caffe2_legacy_pb2.SAME:
251  assert not 'pads' in attrs
252  # default behavior in Caffe2 is SAME_UPPER
253  # https://github.com/caffe2/caffe2/blob/master/caffe2/operators/conv_pool_op_base.h#L39
254  new_attr = make_attribute('auto_pad', 'SAME_UPPER')
255  attrs[new_attr.name] = new_attr
256  elif legacy_pad_attr.i == caffe2_legacy_pb2.CAFFE_LEGACY_POOLING:
257  # The problem here is that, Pool op in Caffe may add an additional pixel, if the last part is smaller than stride.
258  # So we use the explicit padding to replace legacy_pad.
259  # pad[end] = output_size[start + 2] * stride[start] - pad[start] - 1 + kernel[start] - input[start + 2]
260  # end = start + len(pad) / 2
261  logger.warning('Converting legacy padding to explicit padding.')
262  for i in range(2):
263  attrs['pads'].ints[i + 2] = (output_size[i + 2] * attrs['strides'].ints[i] - attrs['pads'].ints[i]
264  - 1 + attrs['kernel_shape'].ints[i] - input_size[i + 2])
265  else:
266  logger.error('Don\'t know how to handle the legacy_pad, while processing operator:\n{}'.format(op_def))
267  raise
268 
269  del node.attribute[:]
270  node.attribute.extend(attrs.values())
271  return node

274  def _create_gemm(cls, op_def, shapes):
275  x, w, b = op_def.input
276  args = {arg.name: arg for arg in op_def.arg}
277  y, = op_def.output
278  x_shape = list(shapes[x])
279 
280  nodes = []
281  const_tensors = []
282  if 'axis' in args:
283  axis = args['axis'].i
284  outer = np.prod(x_shape[:axis]).astype(int)
285  inner = np.prod(x_shape[axis:]).astype(int)
286  reshaped_x = dummy_name()
287  shape_tensor = cls._create_shape_tensor([outer, inner])
288  const_tensors.append(shape_tensor)
289  nodes.append(helper.make_node(
290  'Reshape',
291  inputs=[x, shape_tensor.name],
292  outputs=[reshaped_x],
293  ))
294  x = reshaped_x
295 
296  if 'axis_w' in args:
297  axis_w = args['axis_w'].i
298  w_shape = shapes[w]
299  outer = np.prod(w_shape[:axis_w]).astype(int).item()
300  inner = np.prod(w_shape[axis_w:]).astype(int).item()
301  reshaped_w = dummy_name()
302  shape_tensor = cls._create_shape_tensor([outer, inner])
303  const_tensors.append(shape_tensor)
304  nodes.append(helper.make_node(
305  'Reshape',
306  inputs=[w, shape_tensor.name],
307  outputs=[reshaped_w],
308  ))
309  w = reshaped_w
310 
311  gemm_y_output = dummy_name() if 'axis' in args else y
312  nodes.append(helper.make_node(
313  'Gemm',
314  inputs=[x, w, b],
315  outputs=[gemm_y_output],
316  name=op_def.name,
317  transB=1,
318  broadcast=1,
319  ))
320 
321  if 'axis' in args:
322  axis = args['axis'].i
323  shape_tensor = cls._create_shape_tensor(x_shape[:axis] + [-1])
324  const_tensors.append(shape_tensor)
325  nodes.append(helper.make_node(
326  'Reshape',
327  inputs=[gemm_y_output, shape_tensor.name],
328  outputs=[y],
329  ))
330 
331  return nodes, const_tensors

438  @classmethod
439  def caffe2_net_to_onnx_graph(cls,
440  predict_net,
441  init_net=None,
442  value_info=None):
443  if value_info is None:
444  value_info = {}
445  if not isinstance(value_info, dict):
446  raise ValueError('Please pass value_info as a '
447  'name -> (type, shape) dictionary')
448 
449  cls._filter_fake_init(init_net, value_info)
450  cls._ssa_rewrite(predict_net, init_net, value_info)
451 
452  if init_net:
453  initializer = cls.caffe2_init_net_to_initializer(init_net)
454  value_info.update({init.name: (init.data_type, init.dims)
455  for init in initializer})
456  else:
457  initializer = []
458 
459  # Check whether we have got type shape info of all input
460  missing = (set(list(predict_net.external_input)) -
461  set(value_info.keys()))
462  if missing:
463  raise RuntimeError('Could not find value info of inputs: {}'.format(
464  ', '.join(missing)))
465 
466  inputs = {}
467  for name in predict_net.external_input:
468  elem_type, shape = value_info[name]
469  inputs[name] = np.random.randn(*shape).astype(
470  mapping.TENSOR_TYPE_TO_NP_TYPE[elem_type])
471 
472  ws, outputs = c2_native_run_net(
473  init_net,
474  predict_net,
475  inputs)
476 
477  for name in predict_net.external_output:
478  output = outputs[name]
479  elem_type = mapping.NP_TYPE_TO_TENSOR_TYPE[output.dtype]
480  shape = output.shape
481  value_info[name] = (elem_type, shape)
482 
483  graph_def = GraphProto()
484  graph_def.name = predict_net.name
485  graph_def.initializer.extend(initializer)
486  # This is a mapping from Caffe2 names to ONNX names
487  graph_def.input.extend(
488  make_tensor_value_info(
489  name=name,
490  elem_type=value_info[name][0],
491  shape=value_info[name][1])
492  for name in predict_net.external_input)
493 
494  dummy_name(cls._all_names_in_net(predict_net) |
495  cls._all_names_in_net(init_net))
496 
497  for op in predict_net.op:
498  shapes = {}
499  for name in itertools.chain(op.input, op.output):
500  blob = ws.FetchBlob(name)
501  if hasattr(blob, 'shape'):
502  shapes[name] = blob.shape
503  nodes, const_tensors = cls.caffe2_op_to_onnx_node(op, shapes=shapes)
504  graph_def.node.extend(nodes)
505  graph_def.initializer.extend(const_tensors)
506  graph_def.input.extend([cls._extract_value_info(tensor) for tensor in const_tensors])
507 
508  all_output = set(sum((list(node.output) for node in graph_def.node),
509  [init.name for init in graph_def.initializer]))
510  redundant_output = set(vi.name for vi in graph_def.output) - all_output
511  if redundant_output:
512  logger.warning(
513  'There are graph output not produced by any node or initializer: {}'
514  '! Will drop them.'.format(', '.join(redundant_output)))
515  graph_def.output.extend(
516  make_tensor_value_info(
517  name=name,
518  elem_type=value_info[name][0],
519  shape=value_info[name][1])
520  for name in predict_net.external_output
521  if name in all_output)
522 
523  cls._annotate_consumed(graph_def)
524  checker.check_graph(graph_def)
525  return graph_def

527  @classmethod
528  def caffe2_init_net_to_initializer(cls, init_net):
529  initializer = []
530  for op in init_net.op:
531  assert not op.input
532  try:
533  data_type, field_name = {
534  'GivenTensorFill': (TensorProto.FLOAT, 'floats'),
535  'GivenTensorInt64Fill': (TensorProto.INT64, 'ints'),
536  'GivenTensorIntFill': (TensorProto.INT32, 'ints'),
537  'GivenTensorBoolFill': (TensorProto.BOOL, 'ints'),
538  'GivenTensorStringFill': (TensorProto.STRING, 'strings'),
539  }[op.type]
540  except KeyError:
541  raise RuntimeError(
542  "Can not translate init_net with operator '{}' "
543  "to initializer".format(op.type)
544  )
545  raw = (data_type != TensorProto.STRING)
546  args = {a.name: a for a in op.arg}
547  vals = getattr(args['values'], field_name)
548  if raw:
549  vals = np.asarray(
550  vals,
551  dtype=mapping.TENSOR_TYPE_TO_NP_TYPE[data_type]).tobytes()
552  initializer.append(make_tensor(
553  name=op.output[0],
554  data_type=data_type,
555  dims=args['shape'].ints,
556  vals=vals,
557  raw=raw,
558  ))
559  return initializer

590  @classmethod
591  def _ssa_rewrite(cls, net, init_net, value_info):
592  def ssa_name(name, version):
593  return '{}_{}'.format(name, version)
594 
595  if init_net:
596  for op in init_net.op:
597  assert re.match('GivenTensor.*Fill', op.type), "type is {}, \n{}".format(op.type, op)
598  assert len(op.output) == 1
599  op.output[0] = ssa_name(op.output[0], 0)
600  init_net.external_input[:] = [ssa_name(name, 0)
601  for name in init_net.external_input]
602  init_net.external_output[:] = [ssa_name(name, 0)
603  for name in init_net.external_output]
604  if value_info:
605  ssa_value_info = {ssa_name(name, 0): value
606  for name, value in value_info.items()}
607  value_info.clear()
608  value_info.update(ssa_value_info)
609  net.external_input[:] = [ssa_name(name, 0)
610  for name in net.external_input]
611  ssa, blob_versions = caffe2_core.get_ssa(net)
612  assert len(net.op) == len(ssa)
613  for op, (versioned_inputs, versioned_outputs) in zip(net.op, ssa):
614  op.input[:] = [ssa_name(name, version)
615  for name, version in versioned_inputs]
616  op.output[:] = [ssa_name(name, version)
