85  def assertBlobsEqual(self, spec_blobs, op_blobs):
86  """
87  spec_blobs can either be None or a list of blob names. If it's None,
88  then no assertion is performed. The elements of the list can be None,
89  in that case, it means that position will not be checked.
90  """
91  if spec_blobs is None:
92  return
93  self.assertEqual(len(spec_blobs), len(op_blobs))
94  for spec_blob, op_blob in zip(spec_blobs, op_blobs):
95  if spec_blob is None:
96  continue
97  self.assertEqual(spec_blob, op_blob)

99  def assertNetContainOps(self, net, op_specs):
100  """
101  Given a net and a list of OpSpec's, check that the net match the spec
102  """
103  ops = net.Proto().op
104  self.assertEqual(len(op_specs), len(ops))
105  for op, op_spec in zip(ops, op_specs):
106  self.assertEqual(op_spec.type, op.type)
107  self.assertBlobsEqual(op_spec.input, op.input)
108  self.assertBlobsEqual(op_spec.output, op.output)
109  return ops
