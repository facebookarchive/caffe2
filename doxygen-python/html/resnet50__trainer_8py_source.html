22 '''
23 Parallelized multi-GPU distributed trainer for Resnet 50. Can be used to train
24 on imagenet data, for example.
25 
26 To run the trainer in single-machine multi-gpu mode by setting num_shards = 1.
27 
28 To run the trainer in multi-machine multi-gpu mode with M machines,
29 run the same program on all machines, specifying num_shards = M, and
30 shard_id = a unique integer in the set [0, M-1].
31 
32 For rendezvous (the trainer processes have to know about each other),
33 you can either use a directory path that is visible to all processes
34 (e.g. NFS directory), or use a Redis instance. Use the former by
35 passing the `file_store_path` argument. Use the latter by passing the
36 `redis_host` and `redis_port` arguments.
37 '''

67 def AddMomentumParameterUpdate(train_model, LR):
68  '''
69  Add the momentum-SGD update.
70  '''
71  params = train_model.GetParams()
72  assert (len(params) > 0)
73 
74  for param in params:
75  param_grad = train_model.param_to_grad[param]
76  param_momentum = train_model.param_init_net.ConstantFill(
77  [param], param + '_momentum', value=0.0
78  )
79 
80  # Update param_grad and param_momentum in place
81  train_model.net.MomentumSGDUpdate(
82  [param_grad, param_momentum, LR, param],
83  [param_grad, param_momentum, param],
84  momentum=0.9,
85  nesterov=1,
86  )

136 def RunEpoch(
137  args,
138  epoch,
139  train_model,
140  test_model,
141  total_batch_size,
142  num_shards,
143  expname,
144  explog,
145 ):
146  '''
147  Run one epoch of the trainer.
148  TODO: add checkpointing here.
149  '''
150  # TODO: add loading from checkpoint
151  log.info("Starting epoch {}/{}".format(epoch, args.num_epochs))
152  epoch_iters = int(args.epoch_size / total_batch_size / num_shards)
153  for i in range(epoch_iters):
154  # This timeout is required (temporarily) since CUDA-NCCL
155  # operators might deadlock when synchronizing between GPUs.
156  timeout = 600.0 if i == 0 else 60.0
157  with timeout_guard.CompleteInTimeOrDie(timeout):
158  t1 = time.time()
159  workspace.RunNet(train_model.net.Proto().name)
160  t2 = time.time()
161  dt = t2 - t1
162 
163  fmt = "Finished iteration {}/{} of epoch {} ({:.2f} images/sec)"
164  log.info(fmt.format(i + 1, epoch_iters, epoch, total_batch_size / dt))
165  prefix = "gpu_{}".format(train_model._devices[0])
166  accuracy = workspace.FetchBlob(prefix + '/accuracy')
167  loss = workspace.FetchBlob(prefix + '/loss')
168  train_fmt = "Training loss: {}, accuracy: {}"
169  log.info(train_fmt.format(loss, accuracy))
170 
171  num_images = epoch * epoch_iters * total_batch_size
172  prefix = "gpu_{}".format(train_model._devices[0])
173  accuracy = workspace.FetchBlob(prefix + '/accuracy')
174  loss = workspace.FetchBlob(prefix + '/loss')
175  learning_rate = workspace.FetchBlob(prefix + '/LR')
176  test_accuracy = 0
177  if (test_model is not None):
178  # Run 100 iters of testing
179  ntests = 0
180  for _ in range(0, 100):
181  workspace.RunNet(test_model.net.Proto().name)
182  for g in test_model._devices:
183  test_accuracy += np.asscalar(workspace.FetchBlob(
184  "gpu_{}".format(g) + '/accuracy'
185  ))
186  ntests += 1
187  test_accuracy /= ntests
188  else:
189  test_accuracy = (-1)
190 
191  explog.log(
192  input_count=num_images,
193  batch_count=(i + epoch * epoch_iters),
194  additional_values={
195  'accuracy': accuracy,
196  'loss': loss,
197  'learning_rate': learning_rate,
198  'epoch': epoch,
199  'test_accuracy': test_accuracy,
200  }
201  )
202  assert loss < 40, "Exploded gradients :("
203 
204  # TODO: add checkpointing
205  return epoch + 1

208 def Train(args):
209  # Either use specified device list or generate one
210  if args.gpus is not None:
211  gpus = [int(x) for x in args.gpus.split(',')]
212  num_gpus = len(gpus)
213  else:
214  gpus = range(args.num_gpus)
215  num_gpus = args.num_gpus
216 
217  log.info("Running on GPUs: {}".format(gpus))
218 
219  # Verify valid batch size
220  total_batch_size = args.batch_size
221  batch_per_device = total_batch_size // num_gpus
222  assert \
223  total_batch_size % num_gpus == 0, \
224  "Number of GPUs must divide batch size"
225 
226  # Round down epoch size to closest multiple of batch size across machines
227  global_batch_size = total_batch_size * args.num_shards
228  epoch_iters = int(args.epoch_size / global_batch_size)
229  args.epoch_size = epoch_iters * global_batch_size
230  log.info("Using epoch size: {}".format(args.epoch_size))
231 
232  # Create ModelHelper object
233  train_arg_scope = {
234  'order': 'NCHW',
235  'use_cudnn': True,
236  'cudnn_exhaustice_search': True,
237  'ws_nbytes_limit': (args.cudnn_workspace_limit_mb * 1024 * 1024),
238  }
239  train_model = model_helper.ModelHelper(
240  name="resnet50", arg_scope=train_arg_scope
241  )
242 
243  num_shards = args.num_shards
244  shard_id = args.shard_id
245  if num_shards > 1:
246  # Create rendezvous for distributed computation
247  store_handler = "store_handler"
248  if args.redis_host is not None:
249  # Use Redis for rendezvous if Redis host is specified
250  workspace.RunOperatorOnce(
251  core.CreateOperator(
252  "RedisStoreHandlerCreate", [], [store_handler],
253  host=args.redis_host,
254  port=args.redis_port,
255  prefix=args.run_id,
256  )
257  )
258  else:
259  # Use filesystem for rendezvous otherwise
260  workspace.RunOperatorOnce(
261  core.CreateOperator(
262  "FileStoreHandlerCreate", [], [store_handler],
263  path=args.file_store_path,
264  )
265  )
266  rendezvous = dict(
267  kv_handler=store_handler,
268  shard_id=shard_id,
269  num_shards=num_shards,
270  engine="GLOO",
271  exit_nets=None)
272  else:
273  rendezvous = None

276  def create_resnet50_model_ops(model, loss_scale):
277  [softmax, loss] = resnet.create_resnet50(
278  model,
279  "data",
280  num_input_channels=args.num_channels,
281  num_labels=args.num_labels,

352  AddImageInput(
353  model,
354  test_reader,
355  batch_size=batch_per_device,
356  img_size=args.image_size,
357  )
358 
359  data_parallel_model.Parallelize_GPU(
360  test_model,
361  input_builder_fun=test_input_fn,
362  forward_pass_builder_fun=create_resnet50_model_ops,
363  param_update_builder_fun=None,
364  devices=gpus,
365  )
366  workspace.RunNetOnce(test_model.param_init_net)
367  workspace.CreateNet(test_model.net)
368 
369  workspace.RunNetOnce(train_model.param_init_net)
370  workspace.CreateNet(train_model.net)
371 
372  epoch = 0
373  # load the pre-trained model and reset epoch
374  if args.load_model_path is not None:
375  LoadModel(args.load_model_path, train_model)
376 
377  # Sync the model params
378  data_parallel_model.FinalizeAfterCheckpoint(train_model)
379 
380  # reset epoch. load_model_path should end with *_X.mdl,
381  # where X is the epoch number
382  last_str = args.load_model_path.split('_')[-1]
383  if last_str.endswith('.mdl'):
384  epoch = int(last_str[:-4])
385  log.info("Reset epoch to {}".format(epoch))
386  else:
387  log.warning("The format of load_model_path doesn't match!")
388 
389  expname = "resnet50_gpu%d_b%d_L%d_lr%.2f_v2" % (
390  args.num_gpus,
391  total_batch_size,
392  args.num_labels,
393  args.base_learning_rate,
394  )
395  explog = experiment_util.ModelTrainerLog(expname, args)
396 
397  # Run the training one epoch a time
398  while epoch < args.num_epochs:
399  epoch = RunEpoch(
400  args,
401  epoch,
402  train_model,
403  test_model,
404  total_batch_size,
405  num_shards,
406  expname,
407  explog
408  )
409 
410  # Save the model for each epoch
411  SaveModel(args, train_model, epoch)
412 
413  model_path = "%s/%s_" % (
414  args.file_store_path,
415  args.save_model_name
416  )
417  # remove the saved model from the previous epoch if it exists
418  if os.path.isfile(model_path + str(epoch - 1) + ".mdl"):
419  os.remove(model_path + str(epoch - 1) + ".mdl")
