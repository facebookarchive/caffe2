49  return workspace.blobs[grad_blob]

87 class GradientChecker:
88  """A gradient checker in Python.
89 
90  This is not the most efficient way to check gradients, as the Python
91  interface will involve a lot of copy back and forth operations. Use at your
92  own risk.
93  """
94 
95  def __init__(
96  self,
97  stepsize,
98  threshold,
99  device_option=caffe2_pb2.DeviceOption(),
100  workspace_name="gradient_check"
101  ):
102  self._stepsize = stepsize
103  self._threshold = threshold
104  self._device_option = device_option
105  self._workspace_name = workspace_name
106 
107  def GetLossAndGrad(
108  self, op, grad_ops, x, input_name, grad_name, outputs_with_grads
109  ):
110  # First, feed in the current input. Note that we are not changing
111  # anything else, so we don't need to feed in others.
112  workspace.FeedBlob(input_name, x, self._device_option)
113  # Run.
114  workspace.RunOperatorOnce(op)
115  loss = 0.
116  # Get Loss and feed in the gradients, run gradient ops.
117  for idx in outputs_with_grads:
118  name = op.output[idx]
119  arr = workspace.FetchBlob(name)
120  loss += (arr**2).sum()
121  workspace.FeedBlob(name + '_grad', arr, self._device_option)
122  loss /= 2.
123  # Run gradient ops
124  workspace.RunOperatorsOnce(grad_ops)
125  # Get gradients
126  if isinstance(grad_name, core.GradientSlice):
127  workspace.FeedBlob('zeros', np.zeros_like(x, dtype=np.float32))
128  workspace.FeedBlob('ones', np.ones(1, dtype=np.float32))
129  gv_cpu_op = core.CreateOperator(
130  'EnsureCPUOutput', grad_name.values, grad_name.values + '_cpu',
131  device_option=self._device_option
132  )
133  gi_cpu_op = core.CreateOperator(
134  'EnsureCPUOutput', grad_name.indices, grad_name.indices + '_cpu',
135  device_option=self._device_option
136  )
137  sparse_to_dense_op = core.CreateOperator(
138  'ScatterWeightedSum',
139  [
140  'zeros', 'ones', grad_name.indices + '_cpu',
141  grad_name.values + '_cpu', 'ones'
142  ],
143  'zeros',
144  )
145  workspace.RunOperatorOnce(gv_cpu_op)
146  workspace.RunOperatorOnce(gi_cpu_op)
147  workspace.RunOperatorOnce(sparse_to_dense_op)
148  grad = workspace.FetchBlob('zeros')
149  else:
150  grad = workspace.FetchBlob(grad_name)
151  return loss, grad

153  def CheckSimple(
154  self,
155  op,
156  inputs,
157  input_to_check,
158  outputs_with_grads,
159  grad_ops=None,
160  input_device_options=None
161  ):
162  """Checks the operator in a very simple fashion by stacking a sum of
163  squares on the top.
164 
165  Inputs:
166  op: the operator to be checked.
167  inputs: the input data in numpy arrays.
168  input_to_check: an index specifying which input blob we should
169  check.
170  outputs_with_grads: indices specifying which output blobs will we
171  need to check gradients with. For these outputs, we will collect a
172  squared sum and also feed in their gradients.
173  grad_operator: the gradient operator. If not given, we will get the
174  gradient operator from the gradient registry.
175  input_device_options: an optional mapping from input names to
176  DeviceOptions (to override the default DeviceOption)
177  Outputs:
178  boolean: True if it passes, False if it does not pass.
179  """
180  if input_device_options is None:
181  input_device_options = {}
182  # Entering the checker workspace
183  old_ws_name = workspace.CurrentWorkspace()
184  if self._workspace_name != old_ws_name:
185  workspace.SwitchWorkspace(self._workspace_name, True)
186 
187  op.device_option.CopyFrom(self._device_option)
188  if grad_ops is None:
189  # TODO(jiayq): use the gradient registration instead of the old
190  # hack.
191  grad_ops, g_input = core.GradientRegistry.GetGradientForOp(
192  op, [s + '_grad' for s in op.output])
193 
194  dims_to_check = inputs[input_to_check].size
195  # First, feed in the input.
196  for i, arr in enumerate(inputs):
197  workspace.FeedBlob(
198  op.input[i], arr,
199  input_device_options.get(
200  op.input[i], self._device_option))
201 
202  # Get the loss and gradient for the original.
203  input_name = op.input[input_to_check]
204  grad_name = g_input[input_to_check]
205  loss, grad = self.GetLossAndGrad(
206  op, grad_ops, inputs[input_to_check], input_name, grad_name,
207  outputs_with_grads
208  )
209  grad_estimate = np.zeros_like(inputs[input_to_check])
210  if grad_estimate.shape != grad.shape:
211  raise Exception(
212  "Mismatched gradient shapes: estimated ({}), grad ({})".format(
213  grad_estimate.shape, grad.shape))
214 
215  for current_dim in range(dims_to_check):
216  # Positive gradient
217  inputs[input_to_check].flat[current_dim] += self._stepsize
218  pos_loss, _ = self.GetLossAndGrad(
219  op, grad_ops, inputs[input_to_check], input_name,
220  grad_name, outputs_with_grads
221  )
222  # Negative gradient
223  inputs[input_to_check].flat[current_dim] -= self._stepsize * 2
224  neg_loss, _ = self.GetLossAndGrad(
225  op, grad_ops, inputs[input_to_check], input_name,
226  grad_name, outputs_with_grads
227  )
228  # Recover the value
229  inputs[input_to_check].flat[current_dim] += self._stepsize
230  grad_estimate.flat[current_dim] = (
231  pos_loss - neg_loss) / self._stepsize / 2
232  # Now, check correctness
233  fail_mat = ~np.isclose(
234  grad, grad_estimate, atol=self._threshold, rtol=self._threshold)
235  if np.any(fail_mat):
236  idx = np.flatnonzero(fail_mat)
237  print('Failed. [idx, grad, grad_estimate] are:')
238  print(np.vstack([idx, grad.flat[idx], grad_estimate.flat[idx]]).T)
239  ret = False
240  else:
241  ret = True
242  # After finishing, cleaning up things.
243  if self._workspace_name != old_ws_name:
244  # We reset the workspace to make sure everything intermediate is
245  # cleaned up. Note that there is no need to delete a workspace -
246  # when empty it takes a very limited amount of memory.
247  workspace.ResetWorkspace()
248  workspace.SwitchWorkspace(old_ws_name)
249  return ret, grad, grad_estimate
