159  def _gather_wrapper(self, net, version, in_indices, out):
160  # Gather can work on all kinds of input data types, and output
161  # data with the same type. Convert the output of Gather to float,
162  # because the follow-up Ops expect fp32.
163  if version == 'fp32':
164  return net.Gather([self.w, in_indices], out)
165  elif version == 'fp16':
166  gathered_w = net.Gather([self.w, in_indices], 'gathered_w')
167 
168  return net.HalfToFloat(gathered_w, out)
169  elif version == 'uint8rowwise':
170  gathered_w = net.Gather([self.w, in_indices], 'gathered_w')
171  gathered_scale_bias = net.Gather(
172  [self.scale_bias, in_indices],
173  'gathered_scale_bias'
174  )
175 
176  return net.Rowwise8BitQuantizedToFloat(
177  [gathered_w, gathered_scale_bias], out)
178  elif version == 'fused_uint8rowwise':
179  gathered_w = net.Gather([self.w, in_indices], 'gathered_w')
180  return net.Fused8BitRowwiseQuantizedToFloat(gathered_w, out)
181  else:
182  raise "Unsupported version of operators in SparseLookup " +\
183  "layer: {0}".format(version)

186  self, in_indices, weights, reducer,
187  net, version, grad_on_weights=0):
188  op_input = [
189  self.w,
190  weights,
191  in_indices,
192  self.input_record.lengths()

194  layer_name = 'SparseLengths' + reducer

216  def _add_ops_id_list(self, net, version):

218  "Unsupported reducer: {} for ID_LIST".format(self.reducer)
219  )

221  op_input = [self.w,
222  self.input_record.items(),

226  # 'WeightedSum' are identical, since we can regard the weight on each
227  # id as 1. Similarly, for 'Mean' and 'WeightedMean'.

230  elif self.reducer == 'WeightedMean':

234  if version in ['fp32', 'fp16']:
235  # SparseLengths* Ops will accept either fp16 or fp32 embedding
236  # matrix and output fp32 pooled embedding
237  net.__getattr__(layer_name)(
238  op_input,
239  self.output_schema.field_blobs(),
240  )
241  elif version == 'uint8rowwise':
242  op_input.insert(len(op_input), self.scale_bias)
243  net.__getattr__(layer_name + '8BitsRowwise')(
244  op_input, self.output_schema.field_blobs())
245  elif version == 'fused_uint8rowwise':
246  net.__getattr__(layer_name + 'Fused8BitRowwise')(
247  op_input, self.output_schema.field_blobs())
248  else:
249  raise "Unsupported version of operator in SparseLookUp " +\
250  "layer: {0}".format(version)
251 
252  elif self.reducer == 'Sqrt':
