48  '''
49  Adds optimization operators to the net for given parameter and its gradient
50  Parameter is specified by either 'param' being a ParameterInfo object.
51  In this case param.grad has to be set
52 
53  Or by 'param' being a BlobReference and 'grad' being a BlobReference for its
54  gradient.
55  '''
56  def __call__(self, net, param_init_net, param, grad=None):
57  if grad is None:
58  assert isinstance(param, parameter_info.ParameterInfo)
59  assert param.grad is not None
60  else:
61  if isinstance(param, basestring):
62  param = core.BlobReference(param)
63  param = parameter_info.ParameterInfo(
64  param_id=None, param=param, grad=grad)
65 
66  self._run(net, param_init_net, param)

97  def build_lr(self, net, param_init_net, base_learning_rate,
98  learning_rate_blob=None, policy="fixed",
99  iter_val=0, **kwargs):
100  if learning_rate_blob is None:
101  learning_rate_blob = self.make_unique_blob_name('lr')
102 
103  optimization_iter_blob = _OPTIMIZER_ITERATION_NAME
104  if not param_init_net.BlobIsDefined(optimization_iter_blob):
105  # Add training operators.
106  with core.DeviceScope(core.DeviceOption(caffe2_pb2.CPU)):
107  iteration = param_init_net.ConstantFill(
108  [], optimization_iter_blob, shape=[1],
109  value=iter_val,
110  dtype=core.DataType.INT64)
111  iter_mutex = param_init_net.CreateMutex(
112  [], ["iteration_mutex"]
113  )
114  net.AtomicIter([iter_mutex, iteration], [iteration])
115  else:
116  iteration = param_init_net.GetBlobRef(optimization_iter_blob)
117 
118  if not net.BlobIsDefined(learning_rate_blob):
119  # There is one interesting thing here: since we are minimizing, we are
120  # doing "descent" so the learning rate is set to be negative.
121  lr = net.LearningRate(
122  [iteration],
123  learning_rate_blob,
124  base_lr=-base_learning_rate,
125  policy=policy,
126  **kwargs
127  )
128  else:
129  lr = net.GetBlobRef(learning_rate_blob)
130 
131  if self._lr_multiplier is not None:
132  current_scope = scope.CurrentDeviceScope()
133  if (current_scope.device_type == caffe2_pb2.CUDA
134  and not self._lr_multiplier_on_gpu):
135  lr_multiplier = net.CopyFromCPUInput(
136  self._lr_multiplier,
137  self.make_unique_blob_name('lr_multiplier')
138  )
139  else:
140  lr_multiplier = self._lr_multiplier
141 
142  scaled_lr = net.Mul(
143  [lr, lr_multiplier],
144  self.make_unique_blob_name('scaled_lr'),
145  broadcast=1,
146  )
147  lr = scaled_lr
148 
149  return lr, iteration

164  def get_auxiliary_parameters(self):
165  """Returns a list of auxiliary parameters.
166 
167  Returns:
168  aux_params: A namedtuple, AuxParams.
169 
170  aux_params.local stores a list of blobs. Each blob is a local
171  auxiliary parameter. A local auxiliary parameter is a parameter in
172  parallel to a learning rate parameter. Take adagrad as an example,
173  the local auxiliary parameter is the squared sum parameter, because
174  every learning rate has a squared sum associated with it.
175 
176  aux_params.shared also stores a list of blobs. Each blob is a shared
177  auxiliary parameter. A shared auxiliary parameter is a parameter
178  that is shared across all the learning rate parameters. Take adam as
179  an example, the iteration parameter is a shared parameter, because
180  all the learning rates share the same iteration parameter.
181  """
182  return self._aux_params

197 class SgdOptimizer(Optimizer):
198  def __init__(self, base_learning_rate=0.01, policy='fixed',
199  momentum=0.0, nesterov=1, sparse_dedup_aggregator=None,
200  lars=None, **kwargs):
201  super(SgdOptimizer, self).__init__()
202  self.base_learning_rate = base_learning_rate
203  self.policy = policy
204  self.momentum = momentum
205  self.nesterov = nesterov
206  self.sparse_dedup_aggregator = sparse_dedup_aggregator
207  self.lars = lars
208  self.init_kwargs = kwargs
209 
210  def _run(self, net, param_init_net, param_info):
211  param = param_info.blob
212  grad = param_info.grad
213  if self.base_learning_rate == 0:
214  return
215  assert self.base_learning_rate > 0
216 
217  # TODO(zqq): support LARS for sparse parameters
218  if self.lars is not None and not isinstance(grad, core.GradientSlice):
219  assert self.lars >= 0, 'Lars offset must be nonnegative.'
220  lr_lars_multiplier = net.Lars(
221  [param, grad],
222  self.make_unique_blob_name(str(param) + "_lars"),
223  offset=self.lars)
224  current_scope = scope.CurrentDeviceScope()
225  self.add_lr_multiplier(
226  lr_lars_multiplier,
227  is_gpu_blob=(current_scope.device_type == caffe2_pb2.CUDA),
228  )
229 
230  # We need negative sign for LR when used directly with WeightedSum
231  # below.
232  lr_sign = -1 if self.momentum else 1
233  lr, _ = self.build_lr(
234  net, param_init_net,
235  base_learning_rate=self.base_learning_rate * lr_sign,
236  policy=self.policy,
237  **(self.init_kwargs)
238  )
239 
240  dev = scope.CurrentDeviceScope()
241  if dev is None:
242  dev = core.DeviceOption(caffe2_pb2.CPU)
243 
244  # Each GPU/CPU must have its own ONE blob, thus modify the name
245  # to include device information.
246  ONE = param_init_net.ConstantFill(
247  [],
248  "ONE_{}_{}{}".format(dev.device_type, dev.cuda_gpu_id, dev.node_name),
249  shape=[1],
250  value=1.0
251  )
252 
253  self._aux_params.shared.append(ONE)
254 
255  if self.momentum > 0:
256  momentum_data = param_init_net.ConstantFill(
257  param, str(param) + "_momentum", value=0.)
258  self._aux_params.local.append(momentum_data)
259 
260  if isinstance(grad, core.GradientSlice):
261  grad = self.dedup(net, self.sparse_dedup_aggregator, grad)
262  if self.momentum > 0.:
263  net.SparseMomentumSGDUpdate(
264  [grad.values, momentum_data, lr, param, grad.indices],
265  [grad.values, momentum_data, param],
266  momentum=self.momentum,
267  nesterov=self.nesterov)
268  else:
269  net.ScatterWeightedSum(
270  [param, ONE, grad.indices, grad.values, lr],
271  param
272  )
273  else:
274  if self.momentum > 0.:
275  net.MomentumSGDUpdate(
276  [grad, momentum_data, lr, param],
277  [grad, momentum_data, param],
278  momentum=self.momentum,
279  nesterov=self.nesterov)
280  else:
281  coeff = lr
282 
283  net.WeightedSum(
284  [param, ONE, grad, coeff],
285  param
286  )
287 
288  def scale_learning_rate(self, scale):
289  self.base_learning_rate *= scale
290  return

309  if param_info.blob_copy is not None else None

363  def _run(self, net, param_init_net, param_info, fp32_update=False):
364 
365  fp32_update_flag = 0
366  param_name = str(param_info.blob)
367 
368  # should only be triggered in FP16 training by SpatialBN, which
369  # requires FP32 params in CuDNN.
370  if param_name.find("spatbn") != -1:
371  fp32_update = True
372 
373  if fp32_update:
374  # doing a 32bit update
375  # Have to assume param_info.blob is FP32 as there is no way
376  # (that i currently know of) to query a blob's type in python
377  fp32_update_flag = 1
378  param = param_info.blob
379  param_fp32 = param_info.blob
380  else:
381  if param_info.blob_copy is None:
382  # doing a 32bit update
383  # Have to assume param_info.blob is FP32 as there is no way
384  # (that i currently know of) to query a blob's type in python
385  fp32_update_flag = 1
386  param = param_info.blob
387  param_fp32 = param_info.blob
388  else:
389  if core.DataType.FLOAT in param_info.blob_copy:
390  param = param_info.blob
391  param_fp32 = param_info.blob_copy[core.DataType.FLOAT]
392  elif core.DataType.FLOAT16 in param_info.blob_copy:
393  param = param_info.blob_copy[core.DataType.FLOAT16]
394  param_fp32 = param_info.blob
395  else:
396  assert (False), (
397  "Unrecognized parameter format to be updated "
398  "by FP16 Optimizer. Parameter: {}".format(param_info.name)
399  )
400 
401  grad = param_info.grad
402 
403  if self.base_learning_rate == 0:
404  return
405  assert self.base_learning_rate > 0
406 
407  lr, _ = self.build_lr(
408  net, param_init_net,
409  base_learning_rate=-self.base_learning_rate,
410  policy=self.policy,
411  **(self.init_kwargs)
412  )
413 
414  momentum_data_fp32 = param_init_net.ConstantFill(
415  param_fp32, str(param) + "_momentum_fp32", value=0.)
416 
417  momentum_data = param_init_net.FloatToHalf(
418  momentum_data_fp32, str(param) + "_momentum")
419 
420  self._aux_params.local.append(momentum_data)
421 
422  assert not isinstance(grad, core.GradientSlice), \
423  "Doesn't support sparse gradients"
424 
425  if fp32_update_flag == 0:
426  net.FP16MomentumSGDUpdate(
427  [grad, momentum_data, lr, param],
428  [grad, momentum_data, param],
429  momentum=self.momentum,
430  nesterov=self.nesterov,
431  weight_decay=self.weight_decay)
432  else:
433  # flag set to 1, therefore doing FP32 update
434  net.FP32MomentumSGDUpdate(
435  [grad, momentum_data_fp32, lr, param],
436  [grad, momentum_data_fp32, param],
437  momentum=self.momentum,
438  nesterov=self.nesterov,
439  weight_decay=self.weight_decay)

471 class AdagradOptimizer(Optimizer):
472  def __init__(self, alpha=0.01, epsilon=1e-4, decay=1, policy="fixed",
473  sparse_dedup_aggregator=None, rowWise=False, engine='',
474  lars=None, **kwargs):
475  super(AdagradOptimizer, self).__init__()
476  self.alpha = alpha
477  self.epsilon = epsilon
478  self.decay = decay
479  self.policy = policy
480  self.sparse_dedup_aggregator = sparse_dedup_aggregator
481  self.rowWise = rowWise
482  self.engine = engine

484  self.init_kwargs = kwargs
485 
486  def _run(self, net, param_init_net, param_info):
487  param = param_info.blob
488  grad = param_info.grad
489 
490  if self.alpha <= 0:
491  return
492 
493  if self.lars is not None and not isinstance(grad, core.GradientSlice):
494  assert self.lars >= 0, 'Lars offset must be nonnegative.'
495  lr_lars_multiplier = net.Lars(
496  [param, grad],
497  self.make_unique_blob_name(str(param) + "_lars"),
498  offset=self.lars)
499  current_scope = scope.CurrentDeviceScope()
500  self.add_lr_multiplier(
501  lr_lars_multiplier,
502  is_gpu_blob=(current_scope.device_type == caffe2_pb2.CUDA),
503  )
504 
505  lr, _ = self.build_lr(
506  net, param_init_net,
507  base_learning_rate=self.alpha,
508  policy=self.policy,
509  **(self.init_kwargs)
510  )
511 
512  if self.rowWise:
513  shapes, types = workspace.InferShapesAndTypes([param_init_net])
514  if str(param) not in shapes:
515  # Type/shape inference is not available for this param, fallback
516  # on Shape/Slice logic
517  shape = param_init_net.Shape(param, str(param) + "_shape")
518  num_rows = param_init_net.Slice(
519  [shape],
520  str(shape) + "_numrows",
521  starts=[0], ends=[1]
522  )
523  param_squared_sum = param_init_net.ConstantFill(
524  num_rows,
525  str(param) + "_avg_squared_sum",
526  input_as_shape=1,
527  value=0.0
528  )
529  else:
530  param_squared_sum = param_init_net.ConstantFill(
531  [],
532  str(param) + "_avg_squared_sum",
533  shape=[shapes[str(param)][0]],
534  value=0.0
535  )
536 
537  else:
538  param_squared_sum = param_init_net.ConstantFill(
539  [param],
540  str(param) + "_squared_sum",
541  value=0.0
542  )
543 
544  self._aux_params.local.append(param_squared_sum)
545 
546  if self.rowWise:
547  assert isinstance(grad, core.GradientSlice),\
548  'If SparseAdagrad with rowWise=True, gradient must be '\
549  'a gradientslice. PLease ensure that rowWise is not enabled '\
550  'for the dense Adagrad optimizer, as it is not supported.'
551  if isinstance(grad, core.GradientSlice):
552  assert self.decay == 1.,\
553  'Decay is not implemented for SparseAdagrad and must be set to 1'
554  grad = self.dedup(net, self.sparse_dedup_aggregator, grad)
555  if self.rowWise:
556  op = 'RowWiseSparseAdagrad'
557  else:
558  op = 'SparseAdagrad'
559  net.__getattr__(op)(
560  [param, param_squared_sum, grad.indices, grad.values, lr],
561  [param, param_squared_sum],
562  epsilon=self.epsilon,
563  engine=self.engine
564  )
565  else:
566  net.Adagrad(
567  [param, param_squared_sum, grad, lr],
568  [param, param_squared_sum],
569  epsilon=self.epsilon,
570  decay=float(self.decay),
571  engine=self.engine
572  )
573

575  self.alpha *= scale
576  return

992  model, params, name_scope, param_to_device, max_gradient_norm

1008  )

1011  grad_squared_sum = model.net.SumSqrElements(
1012  grad,
1013  grad_squared_sum_name,

1015  grad_squared_sum_cpu = model.net.EnsureCPUOutput(
1016  grad_squared_sum

1023  'grad_squared_full_sum',

1027  'global_norm',

1037  [global_norm, clip_norm],
1038  'max_norm',
