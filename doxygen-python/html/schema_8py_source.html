393  def __add__(self, other):
394  """
395  Allows to merge fields of two schema.Struct using '+' operator.
396  If two Struct have common field names, the merge is conducted
397  recursively. Here are examples:
398 
399  Example 1
400  s1 = Struct(('a', Scalar()))
401  s2 = Struct(('b', Scalar()))
402  s1 + s2 == Struct(
403  ('a', Scalar()),
404  ('b', Scalar()),
405  )
406 
407  Example 2
408  s1 = Struct(
409  ('a', Scalar()),
410  ('b', Struct(('c', Scalar()))),
411  )
412  s2 = Struct(('b', Struct(('d', Scalar()))))
413  s1 + s2 == Struct(
414  ('a', Scalar()),
415  ('b', Struct(
416  ('c', Scalar()),
417  ('d', Scalar()),
418  )),
419  )
420  """
421  if not isinstance(other, Struct):
422  return NotImplemented
423 
424  children = OrderedDict(self.get_children())
425  for name, right_field in other.get_children():
426  if name not in children:
427  children[name] = right_field
428  continue
429  left_field = children[name]
430  children[name] = left_field + right_field
431 
432  return Struct(*(children.items()))

435 class Scalar(Field):
436  """Represents a typed scalar or tensor of fixed shape.
437 
438  A Scalar is a leaf in a schema tree, translating to exactly one tensor in
439  the dataset's underlying storage.
440 
441  Usually, the tensor storing the actual values of this field is a 1D tensor,
442  representing a series of values in its domain. It is possible however to
443  have higher rank values stored as a Scalar, as long as all entries have
444  the same shape.
445 
446  E.g.:
447 
448  Scalar(np.float64)
449 
450  Scalar field of type float32. Caffe2 will expect readers and
451  datasets to expose it as a 1D tensor of doubles (vector), where
452  the size of the vector is determined by this fields' domain.
453 
454  Scalar((np.int32, 5))
455 
456  Tensor field of type int32. Caffe2 will expect readers and
457  datasets to implement it as a 2D tensor (matrix) of shape (L, 5),
458  where L is determined by this fields' domain.
459 
460  Scalar((str, (10, 20)))
461 
462  Tensor field of type str. Caffe2 will expect readers and
463  datasets to implement it as a 3D tensor of shape (L, 10, 20),
464  where L is determined by this fields' domain.
465 
466  If the field type is unknown at construction time, call Scalar(), that will
467  default to np.void as its dtype.
468 
469  It is an error to pass a structured dtype to Scalar, since it would contain
470  more than one field. Instead, use from_dtype, which will construct
471  a nested `Struct` field reflecting the given dtype's structure.
472 
473  A Scalar can also contain a blob, which represents the value of this
474  Scalar. A blob can be either a numpy.ndarray, in which case it contain the
475  actual contents of the Scalar, or a BlobReference, which represents a
476  blob living in a caffe2 Workspace. If blob of different types are passed,
477  a conversion to numpy.ndarray is attempted.
478  """

660 def from_dtype(dtype, _outer_shape=()):
661  """Constructs a Caffe2 schema from the given numpy's dtype.
662 
663  Numpy supports scalar, array-like and structured datatypes, as long as
664  all the shapes are fixed. This function breaks down the given dtype into
665  a Caffe2 schema containing `Struct` and `Scalar` types.
666 
667  Fields containing byte offsets are not currently supported.
668  """
669  if not isinstance(dtype, np.dtype):
670  # wrap into a ndtype
671  shape = _outer_shape
672  dtype = np.dtype((dtype, _outer_shape))
673  else:
674  # concatenate shapes if necessary
675  shape = _outer_shape + dtype.shape
676  if shape != dtype.shape:
677  dtype = np.dtype((dtype.base, shape))
678 
679  if not dtype.fields:
680  return Scalar(dtype)
681 
682  struct_fields = []
683  for name, (fdtype, offset) in dtype.fields:
684  assert offset == 0, ('Fields with byte offsets are not supported.')
685  struct_fields += (name, from_dtype(fdtype, _outer_shape=shape))
686  return Struct(*struct_fields)

930  if not schema_or_record.has_blobs():
