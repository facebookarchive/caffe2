101 def get_setup_nets(key, steps_or_nets, target):
102  init_net = core.Net(key + '/init')
103  exit_net = core.Net(key + '/exit')
104  init_nets = []
105  exit_nets = []
106  objs = []
107  for step_or_net in steps_or_nets:
108  if hasattr(step_or_net, 'get_all_attributes'):
109  objs += step_or_net.get_all_attributes(key)
110  elif hasattr(step_or_net, 'get_attributes'):
111  objs += step_or_net.get_attributes(key)
112  for obj in objs:
113  # these are needed in order to allow nesting of TaskGroup, which
114  # is a feature not yet implemented.
115  if hasattr(obj, '_setup_used') and obj._setup_used:
116  continue
117  if hasattr(obj, '_setup_target') and obj._setup_target != target:
118  continue
119  if hasattr(obj, 'setup'):
120  nets = obj.setup(init_net)
121  if isinstance(nets, (list, tuple)):
122  init_nets += nets
123  elif isinstance(nets, (core.Net, core.ExecutionStep)):
124  init_nets.append(nets)
125  elif nets is not None:
126  raise TypeError('Unsupported type for setup: %s' % type(nets))
127  obj._setup_used = True
128  if hasattr(obj, 'exit'):
129  nets = obj.exit(exit_net)
130  if isinstance(nets, (list, tuple)):
131  exit_nets += nets
132  elif isinstance(nets, (core.Net, core.ExecutionStep)):
133  exit_nets.append(nets)
134  elif nets is not None:
135  raise TypeError('Unsupported type for setup: %s' % type(nets))
136  obj._setup_used = True
137 
138  if len(init_net.Proto().op) > 0:
139  init_nets.insert(0, init_net)
140  if len(exit_net.Proto().op) > 0:
141  exit_nets.insert(0, exit_net)
142  return init_nets, exit_nets

145 @context.define_context(allow_default=False)
146 class TaskGroup(object):
147  """
148  Context that gathers tasks which will run concurrently, potentially on
149  multiple nodes. All tasks in the same node will share the same workspace
150  and thus can share blobs, while tasks running in different nodes won't
151  be able to directly share data.
152 
153  All tasks of the task group will start concurrently, and the task group
154  will finish execution when the last task of the group finishes.
155 
156  Example:
157  # supose that s1 ... s5 are execution steps or nets.
158  with TaskGroup() as tg:
159  # these tasks go to default node 'local'
160  Task(step=s1)
161  Task(step=s2)
162 
163  with Node('n2'):
164  Task(step=s3)
165  with Node('n1'):
166  Task(step=s4)
167  with Node('n2'):
168  Task(step=s5)
169 
170  # this will run all steps in parallel.
171  # s1 and s2 will run at default node 'local'
172  # s3 and s5 will run at node 'n2'
173  # s4 will run at node 'n1'
174  session.run(tg)
175  """

549 class SetupNets(object):

551  Allow to register a list of nets to be run at initialization
552  and finalization of Tasks or TaskGroups.
553  For example, let's say you have the following:
554 
555  init_net = core.Net('init')
556  my_val = init_net.ConstantFill([], 'my_val', value=0)
557 
558  net = core.Net('counter')

562  with Node('trainer'):
563  my_task = Task(step=[net])

566  first time, you can do one of the following:
567 
568  net.add_object(Task.TASK_SETUP, SetupNets([init_net]))
569 
570  or
571 
572  net.add_object(TaskGroup.LOCAL_SETUP, SetupNets([init_net]))
573 
574  - With Task.TASK_SETUP, init_net will run once at my_task startup.
575  - With TaskGroup.LOCAL_SETUP, init_net will run once on node 'trainer',
576  before any task of the task group is run on that node.
577

579  run once per Task/TaskGroup run.
