121 // Waits until *var != initial_value.
122 //
123 // Returns the new value of *var. The guarantee here is that
124 // the return value is different from initial_value, and that that
125 // new value has been taken by *var at some point during the
126 // execution of this function. There is no guarantee that this is
127 // still the value of *var when this function returns, since *var is
128 // not assumed to be guarded by any lock.
129 //
130 // First does some busy-waiting for a fixed number of no-op cycles,
131 // then falls back to passive waiting for the given condvar, guarded
132 // by the given mutex.
133 //
134 // The idea of doing some initial busy-waiting is to help get
135 // better and more consistent multithreading benefits for small GEMM sizes.
136 // Busy-waiting help ensuring that if we need to wake up soon after having
137 // started waiting, then we can wake up quickly (as opposed to, say,
138 // having to wait to be scheduled again by the OS). On the other hand,
139 // we must still eventually revert to passive waiting for longer waits
140 // (e.g. worker threads having finished a GEMM and waiting until the next GEMM)
141 // so as to avoid permanently spinning.
142 //
143 template <typename T>
144 T WaitForVariableChange(std::atomic<T>* var,
145  T initial_value,
146  std::condition_variable* cond,
147  std::mutex* mutex) {
148  // If we are on a platform that supports it, spin for some time.
149  {
150  int nops = 0;
151  // First, trivial case where the variable already changed value.
152  T new_value = var->load(std::memory_order_relaxed);
153  if (new_value != initial_value) {
154  std::atomic_thread_fence(std::memory_order_acquire);
155  return new_value;
156  }
157  // Then try busy-waiting.
158  while (nops < kMaxBusyWaitNOPs) {
159  nops += Do256NOPs();
160  new_value = var->load(std::memory_order_relaxed);
161  if (new_value != initial_value) {
162  std::atomic_thread_fence(std::memory_order_acquire);
163  return new_value;
164  }
165  }
166  }
167 
168  // Finally, do real passive waiting.
169  {
170  std::unique_lock<std::mutex> g(*mutex);
171  T new_value = var->load(std::memory_order_relaxed);
172  // Handle spurious wakeups.
173  cond->wait(g, [&]() {
174  new_value = var->load(std::memory_order_relaxed);
175  return new_value != initial_value;
176  });
177  DCHECK_NE(static_cast<size_t>(new_value), static_cast<size_t>(initial_value));
178  return new_value;
179  }
180 }

182 // A BlockingCounter lets one thread to wait for N events to occur.
183 // This is how the master thread waits for all the worker threads
184 // to have finished working.
185 class BlockingCounter {
186  public:
187  // Sets/resets the counter; initial_count is the number of
188  // decrementing events that the Wait() call will be waiting for.
189  void Reset(std::size_t initial_count) {
190  std::lock_guard<std::mutex> g(mutex_);
191  DCHECK_EQ(count_, 0);
192  count_ = initial_count;
193  }
194 
195  // Decrements the counter; if the counter hits zero, signals
196  // the thread that was waiting for that, and returns true.
197  // Otherwise (if the decremented count is still nonzero),
198  // returns false.
199  bool DecrementCount() {
200  const auto count_value = count_.fetch_sub(1, std::memory_order_relaxed) - 1;
201  DCHECK_GE(count_value, 0);
202  if (count_value == 0) {
203  std::lock_guard<std::mutex> g(mutex_);
204  cond_.notify_one();
205  }
206  bool retval = count_value == 0;
207  return retval;
208  }
209 
210  // Waits for the N other threads (N having been set by Reset())
211  // to hit the BlockingCounter.
212  void Wait() {
213  while (size_t count_value = count_.load(std::memory_order_relaxed)) {
214  WaitForVariableChange(&count_, count_value, &cond_, &mutex_);
215  }
216  }
217 
218  private:
219  std::condition_variable cond_;
220  std::mutex mutex_;
221  std::atomic<std::size_t> count_{0};
222 };

231 // A worker thread.
232 class alignas(kGEMMLOWPCacheLineSize) Worker {
233  public:
234  enum class State : uint8_t {
235  ThreadStartup, // The initial state before the thread main loop runs.
236  Ready, // Is not working, has not yet received new work to do.
237  HasWork, // Has work to do.
238  ExitAsSoonAsPossible // Should exit at earliest convenience.
239  };
240 
241  explicit Worker(BlockingCounter* counter_to_decrement_when_ready)
242  : task_(nullptr),
243  state_(State::ThreadStartup),
244  counter_to_decrement_when_ready_(counter_to_decrement_when_ready) {
245  thread_ = caffe2::make_unique<std::thread>([this]() { this->ThreadFunc(); });
246  }
247 
248  ~Worker() {
249  ChangeState(State::ExitAsSoonAsPossible);
250  thread_->join();
251  }
252 
253  // Changes State; may be called from either the worker thread
254  // or the master thread; however, not all state transitions are legal,
255  // which is guarded by assertions.
256  void ChangeState(State new_state) {
257  std::lock_guard<std::mutex> g(state_mutex_);
258  DCHECK(new_state != state_.load(std::memory_order_relaxed));
259  switch (state_.load(std::memory_order_relaxed)) {
260  case State::ThreadStartup:
261  DCHECK(new_state == State::Ready);
262  break;
263  case State::Ready:
264  DCHECK(new_state == State::HasWork || new_state == State::ExitAsSoonAsPossible);
265  break;
266  case State::HasWork:
267  DCHECK(new_state == State::Ready || new_state == State::ExitAsSoonAsPossible);
268  break;
269  default:
270  abort();
271  }
272  state_.store(new_state, std::memory_order_relaxed);
273  state_cond_.notify_one();
274  if (new_state == State::Ready) {
275  counter_to_decrement_when_ready_->DecrementCount();
276  }
277  }
278 
279  // Thread entry point.
280  void ThreadFunc() {
281  ChangeState(State::Ready);
282 
283  // Thread main loop
284  while (true) {
285  // Get a state to act on
286  // In the 'Ready' state, we have nothing to do but to wait until
287  // we switch to another state.
288  State state_to_act_upon =
289  WaitForVariableChange(&state_, State::Ready, &state_cond_, &state_mutex_);
290 
291  // We now have a state to act on, so act.
292  switch (state_to_act_upon) {
293  case State::HasWork:
294  // Got work to do! So do it, and then revert to 'Ready' state.
295  DCHECK(task_);
296  task_->Run();
297  task_ = nullptr;
298  ChangeState(State::Ready);
299  break;
300  case State::ExitAsSoonAsPossible:
301  return;
302  default:
303  abort();
304  }
305  }
306  }
307 
308  static void* ThreadFunc(void* arg) {
309  static_cast<Worker*>(arg)->ThreadFunc();
310  return nullptr;
311  }

313  // Called by the master thead to give this worker work to do.
314  // It is only legal to call this if the worker
315  void StartWork(Task* task) {
316  DCHECK(!task_);
317  task_ = task;
318  DCHECK(state_.load(std::memory_order_acquire) == State::Ready);
319  ChangeState(State::HasWork);
320  }
321 
322  private:
323  // The underlying thread.
324  std::unique_ptr<std::thread> thread_;
325 
326  // The task to be worked on.
327  // Visibility of writes to task_ guarded by state_mutex_.
328  Task* task_;
329 
330  // The condition variable and mutex guarding state changes.
331  std::condition_variable state_cond_;
332  std::mutex state_mutex_;
333 
334  // The state enum tells if we're currently working, waiting for work, etc.
335  std::atomic<State> state_;
336 
337  // pointer to the master's thread BlockingCounter object, to notify the
338  // master thread of when this worker switches to the 'Ready' state.
339  BlockingCounter* const counter_to_decrement_when_ready_;
340 };

346  void Execute(const std::vector<std::shared_ptr<Task>>& tasks) {
347  CAFFE_ENFORCE_GE(tasks.size(), 1);
348  // One of the tasks will be run on the current thread.
349  int workers_count = tasks.size() - 1;
350  CreateWorkers(workers_count);
351  DCHECK_LE(workers_count, workers_.size());
352  counter_to_decrement_when_ready_.Reset(workers_count);
353  for (auto task = 1; task < tasks.size(); ++task) {
354  workers_[task - 1]->StartWork(tasks[task].get());
355  }
356  // Execute the remaining workload immediately on the current thread.
357  auto& task = tasks.front();
358  task->Run();
359  // Wait for the workers submitted above to finish.
360  counter_to_decrement_when_ready_.Wait();
361  }
362 
363  private:
364  // Ensures that the pool has at least the given count of workers.
365  // If any new worker has to be created, this function waits for it to
366  // be ready.
367  void CreateWorkers(std::size_t workers_count) {
368  if (workers_.size() >= workers_count) {
369  return;
370  }
371  counter_to_decrement_when_ready_.Reset(workers_count - workers_.size());
372  while (workers_.size() < workers_count) {
373  workers_.push_back(MakeAligned<Worker>::make(&counter_to_decrement_when_ready_));
374  }
375  counter_to_decrement_when_ready_.Wait();
376  }
377 
378  DISABLE_COPY_AND_ASSIGN(WorkersPool);
379  std::vector<std::unique_ptr<Worker, AlignedDeleter<Worker>>> workers_;
380  // The BlockingCounter used to wait for the workers.
381  BlockingCounter counter_to_decrement_when_ready_;
382 };
