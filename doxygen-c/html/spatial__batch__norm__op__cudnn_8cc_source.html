83 template <typename T>
84 bool CudnnSpatialBNOp<T>::RunOnDevice() {
85  const auto& X = Input(INPUT);
86  const auto& scale = Input(SCALE);
87  const auto& bias = Input(BIAS);
88 
89  DCHECK_EQ(X.ndim(), 4);
90  const int N = X.dim32(0);
91  const int C = (order_ == StorageOrder::NCHW ? X.dim32(1) : X.dim32(3));
92  const int H = (order_ == StorageOrder::NCHW ? X.dim32(2) : X.dim32(1));
93  const int W = (order_ == StorageOrder::NCHW ? X.dim32(3) : X.dim32(2));
94  DCHECK_EQ(scale.ndim(), 1);
95  DCHECK_EQ(bias.ndim(), 1);
96  DCHECK_EQ(scale.dim32(0), C);
97  DCHECK_EQ(bias.dim32(0), C);
98  // See if we need to reshape.
99  if (X.dims() != cudnn_input_dims_) {
100  VLOG(1) << "Setting descriptors.";
101  cudnn_input_dims_ = X.dims();
102  CUDNN_ENFORCE(cudnnSetTensor4dDescriptor(
103  data_desc_,
104  GetCudnnTensorFormat(order_),
105  cudnnTypeWrapper<T>::type,
106  N,
107  C,
108  H,
109  W));
110  CUDNN_ENFORCE(cudnnDeriveBNTensorDescriptor(
111  bn_param_desc_, data_desc_, kSpatialBNMode));
112  }
113 
114  // Now, depending on whether we are running test or not, we have two paths.
115  if (is_test_) {
116  // Run inference mode.
117  const auto& est_mean = Input(EST_MEAN);
118  const auto& est_var = Input(EST_VAR);
119  DCHECK_EQ(est_mean.ndim(), 1);
120  DCHECK_EQ(est_var.ndim(), 1);
121  DCHECK_EQ(est_mean.dim32(0), C);
122  DCHECK_EQ(est_var.dim32(0), C);
123 
124  auto* Y = Output(OUTPUT);
125  Y->ResizeLike(X);
126  CUDNN_ENFORCE(cudnnBatchNormalizationForwardInference(
127  cudnn_wrapper_.inline_cudnn_handle(),
128  kSpatialBNMode,
129  cudnnTypeWrapper<T>::kOne(),
130  cudnnTypeWrapper<T>::kZero(),
131  data_desc_,
132  X.template data<T>(),
133  data_desc_,
134  Y->template mutable_data<T>(),
135  bn_param_desc_,
136  scale.template data<T>(),
137  bias.template data<T>(),
138  est_mean.template data<T>(),
139  est_var.template data<T>(),
140  epsilon_));
141  } else {
142  // Run training mode.
143  auto* Y = Output(OUTPUT);
144  Y->ResizeLike(X);
145  // obtain running mean and running inv var, and see if we need to
146  // initialize them.
147  auto* running_mean = Output(RUNNING_MEAN);
148  auto* running_var = Output(RUNNING_VAR);
149  double this_factor = 1. - momentum_;
150  T* running_mean_data = nullptr;
151  T* running_var_data = nullptr;
152  if (!running_mean->size()) {
153  // If the input mean and var are not initialized yet, this is the first
154  // run and we will initialize the storage.
155  VLOG(1) << "Initializing running mean and var.";
156  // Need to do initialization
157  running_mean->Resize(C);
158  running_var->Resize(C);
159  running_mean_data = running_mean->template mutable_data<T>();
160  running_var_data = running_var->template mutable_data<T>();
161  // In principle, setting this_momentum to 1 will wipe existing data.
162  // This has a caveat that if cudnn does not deal with 0*NaN cases we
163  // will be having an issue. Thus we choose a safe path by explicitly
164  // setting zero.
165  math::Set<T, CUDAContext>(C, 0, running_mean_data, &context_);
166  math::Set<T, CUDAContext>(C, 0, running_var_data, &context_);
167  } else {
168  // Does not need to do initialization.
169  DCHECK_EQ(running_mean->ndim(), 1);
170  DCHECK_EQ(running_var->ndim(), 1);
171  DCHECK_EQ(running_mean->dim32(0), C);
172  DCHECK_EQ(running_var->dim32(0), C);
173  running_mean_data = running_mean->template mutable_data<T>();
174  running_var_data = running_var->template mutable_data<T>();
175  }
176  // Save the mean and inv var results.
177  auto* save_mean = Output(SAVED_MEAN);
178  auto* save_var = Output(SAVED_INV_VAR);
179  save_mean->Resize(C);
180  save_var->Resize(C);
181  void* save_mean_data = save_mean->template mutable_data<T>();
182  void* save_var_data = save_var->template mutable_data<T>();
183 
184  CUDNN_ENFORCE(cudnnBatchNormalizationForwardTraining(
185  cudnn_wrapper_.inline_cudnn_handle(),
186  kSpatialBNMode,
187  cudnnTypeWrapper<T>::kOne(),
188  cudnnTypeWrapper<T>::kZero(),
189  data_desc_,
190  X.template data<T>(),
191  data_desc_,
192  Y->template mutable_data<T>(),
193  bn_param_desc_,
194  scale.template data<T>(),
195  bias.template data<T>(),
196  this_factor,
197  running_mean_data,
198  running_var_data,
199  epsilon_,
200  save_mean_data,
201  save_var_data));
202  }
203  return true;
204 }
