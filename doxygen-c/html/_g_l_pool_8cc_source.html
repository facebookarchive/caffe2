81  void pool(const GLImageVector<float16_t>& input_images,
82  const GLImageVector<float16_t>& output_images) {
83  for (int i = 0; i < input_images.size(); i++) {
84  auto input_image = input_images[i];
85  auto output_image = output_images[i];
86  int input_slices = input_image->slices;
87  int output_slices = output_image->slices;
88 
89  for (int is = 0; is < input_slices; is++) {
90  run({{input_image->textures[is], inputData}},
91  {output_image->textures[is]},
92  [&]() {
93  glUniform2i(outputSize->location, output_image->texture_width, output_image->texture_height);
94  glUniform2i(kernelSize->location, geometry.kernel_size.x, geometry.kernel_size.y);
95  },
96  output_image->texture_width,
97  output_image->texture_height);
98  }
99  }
100  }

106  * @retval 0 if bounds check can be skipped

304  bool RunOnDeviceWithOrderNCHW() override {
305  const GLImageVector<T>& input = OperatorBase::Inputs()[0]->template Get<GLImageVector<T>>();
306  const int num_images = input.size();

308  const int input_width = input.width();
309  const int input_height = input.height();

311  int output_height;

313  const int output_channels = input_channels;

315  computeOutputHW(this, input_height, input_width, &output_height, &output_width);

318

320  const int output_tile_x = input_tile_x, output_tile_y = input_tile_y;
321 
322  GLImageVector<T>* output = ImageAllocator<T>::newImage(
323  num_images, output_width, output_height, output_channels, output_tile_x, output_tile_y, is_last);
324 
325  GLPool::descriptor geometry{input_channels,
326  {kernel_w(), kernel_h()},
327  {pad_l(), pad_t()},
328  {stride_w(), stride_h()},
329  {input_width, input_height},
330  {output_height, output_width}};
331 
332  if (!glPool_) {
