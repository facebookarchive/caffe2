68 // Caffe2 gemm provides a simpler interface to the gemm functions, with the
69 // limitation that the data has to be contiguous in memory.
70 //
71 // The gemm call implements the following operation:
72 //
73 // C = alpha * op(A) * op(B) + beta * C
74 //
75 // where op(A) has size M x K, op(B) has size K x N, and C has size M x N. Each
76 // of A, B, and C are matrices and alpha and beta are scalars. Note that the
77 // most common use case of gemm will involve setting alpha to 1 and beta to 0.
78 //
79 // op(A) and op(B) represent the transformations that are done to A and B before
80 // the matrix multiply; depending on the flags set, op(A) is equal to A or A^T
81 // (transpose) if the argument TransA or TransB is set to CblasNoTrans or
82 // CblasTrans, respectively, for each of A and B.
83 template <>
84 void Gemm<float, CPUContext>(
85  const CBLAS_TRANSPOSE TransA,
86  const CBLAS_TRANSPOSE TransB,
87  const int M,
88  const int N,
89  const int K,
90  const float alpha,
91  const float* A,
92  const float* B,
93  const float beta,
94  float* C,
95  CPUContext* context,
96  TensorProto::DataType math_type) {
97  auto C_mat = EigenMatrixMap<float>(C, N, M);
98  if (beta == 0) {
99  C_mat.setZero();
100  } else {
101  C_mat *= beta;
102  }
103  switch (TransA) {
104  case CblasNoTrans: {
105  switch (TransB) {
106  case CblasNoTrans:
107  C_mat.noalias() += alpha * (
108  ConstEigenMatrixMap<float>(B, N, K) *
109  ConstEigenMatrixMap<float>(A, K, M));
110  return;
111  case CblasTrans:
112  C_mat.noalias() += alpha * (
113  ConstEigenMatrixMap<float>(B, K, N).transpose() *
114  ConstEigenMatrixMap<float>(A, K, M));
115  return;
116  default:
117  LOG(FATAL) << "Unexpected CBLAS_TRANSPOSE for TransB";
118  }
119  }
120  case CblasTrans: {
121  switch (TransB) {
122  case CblasNoTrans:
123  C_mat.noalias() += alpha * (
124  ConstEigenMatrixMap<float>(B, N, K) *
125  ConstEigenMatrixMap<float>(A, M, K).transpose());
126  return;
127  case CblasTrans:
128  C_mat.noalias() += alpha * (
129  ConstEigenMatrixMap<float>(B, K, N).transpose() *
130  ConstEigenMatrixMap<float>(A, M, K).transpose());
131  return;
132  default:
133  LOG(FATAL) << "Unexpected CBLAS_TRANSPOSE for TransB";
134  }
135  }
136  default:
137  LOG(FATAL) << "Unexpected CBLAS_TRANSPOSE for TransA";
138  }

232  y_vec.noalias() += alpha * (
233  ConstEigenMatrixMap<float>(A, N, M) *
234  ConstEigenVectorMap<float>(x, M));
235  return;
236  }

959 void Im2colNd<float, CPUContext, StorageOrder::NCHW>(
960  const float* data_img,
961  const int* im_shape,
962  const int* col_shape,
963  const int /* img_size*/,
964  const int /* col_size*/,
965  const int* kernel_shape,
966  const int* stride,
967  const int* dilation,
968  const int* pad,
969  const int N,
970  float* data_col,
971  CPUContext* /* context */,
972  bool accumulate_output) {
973  int kernel_size = 1;
974  for (int i = 0; i < N; ++i) {
975  kernel_size *= kernel_shape[i];
976  }
977  const int channels_col = col_shape[0];
978  vector<int> d_offset(N, 0);
979  vector<int> d_iter(N, 0);
980  for (int c_col = 0; c_col < channels_col; ++c_col) {
981  // Loop over spatial axes in reverse order to compute a per-axis offset.
982  int offset = c_col;
983  for (int d_i = N - 1; d_i >= 0; --d_i) {
984  if (d_i < N - 1) {
985  offset /= kernel_shape[d_i + 1];
986  }
987  d_offset[d_i] = offset % kernel_shape[d_i];
988  }
989  for (bool incremented = true; incremented;) {
990  // Loop over spatial axes in forward order to compute the indices in the
991  // image and column, and whether the index lies in the padding.
992  int index_col = c_col;
993  int index_im = c_col / kernel_size;
994  bool is_padding = false;
995  for (int d_i = 0; d_i < N; ++d_i) {
996  const int d = d_iter[d_i];
997  const int d_im =
998  d * stride[d_i] - pad[d_i] + d_offset[d_i] * dilation[d_i];
999  is_padding |= d_im < 0 || d_im >= im_shape[d_i + 1];
1000  index_col *= col_shape[d_i + 1];
1001  index_col += d;
1002  index_im *= im_shape[d_i + 1];
1003  index_im += d_im;
1004  }
1005  if (!accumulate_output) {
1006  if (is_padding) {
1007  data_col[index_col] = 0;
1008  } else {
1009  data_col[index_col] = data_img[index_im];
1010  }
1011  } else if (!is_padding) { // col2im
1012  data_col[index_im] += data_img[index_col];
1013  }
1014  // Loop over spatial axes in reverse order to choose an index,
1015  // like counting.
1016  incremented = false;
1017  for (int d_i = N - 1; d_i >= 0; --d_i) {
1018  const int d_max = col_shape[d_i + 1];
1019  DCHECK_LT(d_iter[d_i], d_max);
1020  if (d_iter[d_i] == d_max - 1) {
1021  d_iter[d_i] = 0;
1022  } else { // d_iter[d_i] < d_max - 1
1023  ++d_iter[d_i];
1024  incremented = true;
1025  break;
1026  }
1027  }
1028  } // while(incremented) {
1029  } // for (int c = 0; c < channels_col; ++c) {

1082  (height + pad_b + pad_t - (dilation_h * (kernel_h - 1) + 1)) / stride_h +
1083  1;
1084  const int output_w =
1085  (width + pad_l + pad_r - (dilation_w * (kernel_w - 1) + 1)) / stride_w +
1086  1;

1089  // From Torch, THNN_(unfolded_copy)
1090  if (dilation_h == 1 && dilation_w == 1 && pad_l == 0 && pad_r == 0 &&
1091  pad_t == 0 && pad_b == 0) {
1092  for (auto k = 0; k < channels * kernel_h * kernel_w; k++) {
1093  const auto nip = k / (kernel_h * kernel_w);
1094  const auto rest = k % (kernel_h * kernel_w);
1095  const auto kh = rest / kernel_w;
1096  const auto kw = rest % kernel_w;
1097  auto* dst = data_col + nip * (kernel_h * kernel_w * output_h * output_w) +
1098  kh * (kernel_w * output_h * output_w) + kw * (output_h * output_w);

1100  for (auto y = 0; y < output_h; y++) {
1101  const auto iy = y * stride_h + kh;
1102  const auto ix = kw;
1103  if (stride_w == 1) {
1104  memcpy(
1105  dst + (y * output_w),
1106  src + (iy * width + ix),
1107  sizeof(float) * output_w);
1108  } else {
1109  for (auto x = 0; x < output_w; x++) {
1110  memcpy(
1111  dst + (y * output_w + x),
1112  src + (iy * width + ix + x * stride_w),
1113  sizeof(float));
1114  }
1115  }
1116  }
1117  }
1118  return;
1119  }

1122  if (pad_l == pad_r && pad_t == pad_b) {

1129  for (int kernel_col = 0; kernel_col < kernel_w; kernel_col++) {

1134  *(data_col++) = 0;

1136  } else {
1137  int input_col = -pad_w + kernel_col * dilation_w;
1138  for (int output_col = output_w; output_col; output_col--) {
1139  if (is_a_ge_zero_and_a_lt_b(input_col, width)) {
1140  *(data_col++) = data_im[input_row * width + input_col];
1141  } else {
1142  *(data_col++) = 0;
1143  }
1144  input_col += stride_w;
1145  }
1146  }
1147  input_row += stride_h;
1148  }
1149  }

1156  const int dkernel_h = dilation_h * (kernel_h - 1) + 1;
1157  const int dkernel_w = dilation_w * (kernel_w - 1) + 1;

1160  int width_col = (width + pad_l + pad_r - dkernel_w) / stride_w + 1;
1161

1163  for (int c = 0; c < channels_col; ++c) {

1165  int h_offset = (c / kernel_w) % kernel_h;
1166  int c_im = c / kernel_h / kernel_w;
1167  for (int h = 0; h < height_col; ++h) {
1168  for (int w = 0; w < width_col; ++w) {
