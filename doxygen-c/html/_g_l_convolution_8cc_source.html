462 template <typename T>
463 void GLConvolution::run_batched_conv(
464  const GLImageVector<T>& input_images,
465  const GLImageVector<T>& output_images) {
466  for (int i = 0; i < input_images.size(); i++) {
467  GLImage<T>* input_image = input_images[i];
468  GLImage<T>* output_image = output_images[i];
469  int input_slices = input_image->slices;
470  int output_slices = output_image->slices;
471 
472  for (int is = 0; is < input_slices; is += input_batch_size) {
473  for (int os = 0; os < output_slices; os += output_batch_size) {
474  const int output_channels_per_batch =
475  std::min(4 * output_batch_size, geometry.output_channels - 4 * os);
476 
477  gl_log(
478  GL_VERBOSE,
479  "GLConvolution::convolution - is: %d, os: %d\n",
480  is,
481  os);
482 
483  // Note the order of the binding point needs to be the same as in the
484  // constructor
485  int binding_point = 0;
486 
487  // bias
488  attach_uniform_buffer<float16_t>(
489  bias_block, binding_point++, [&](float16_t* data, size_t size) {
490  CAFFE_ENFORCE_GE(
491  size,
492  output_channels_per_batch * sizeof(float16_t),
493  "Bias buffer size too small");
494  for (int ob = 0; ob < output_channels_per_batch; ob++) {
495  data[ob] = bias[4 * os + ob];
496  }
497  });
498 
499  // kernel weights
500  for (int ib = 0; ib < input_batch_size; ib++) {
501  attach_uniform_buffer<float16_t>(
502  kernel_block[ib],
503  binding_point++,
504  [&](float16_t* data, size_t size) {
505  CAFFE_ENFORCE_EQ(
506  size,
507  4 * (4 * output_batch_size) * geometry.kernel_size.y *
508  geometry.kernel_size.x * sizeof(float16_t),
509  "Kernel size mismatch");
510  pack_kernel_data_for_bached_conv(
511  data,
512  size,
513  input_image->channels,
514  output_image->channels,
515  is,
516  os,
517  ib);
518  });
519  }
520 
521  // PRelu scale
522  if (prelu_scale != nullptr && is == input_slices - input_batch_size) {
523  attach_uniform_buffer<float16_t>(
524  prelu_scale_block,
525  binding_point++,
526  [&](float16_t* data, size_t size) {
527  CAFFE_ENFORCE_GE(
528  size,
529  output_channels_per_batch * sizeof(float16_t),
530  "PRelu buffer size too small");
531  for (int ob = 0; ob < output_channels_per_batch; ob++) {
532  data[ob] = prelu_scale_size == geometry.output_channels
533  ? prelu_scale[4 * os + ob]
534  : prelu_scale[0];
535  }
536  });
537  }
538 
539  std::vector<texture_attachment> input_attachments;
540  for (int ib = 0; ib < input_batch_size; ib++) {
541  input_attachments.push_back(
542  {input_image->textures[is + ib], inputData[ib]});
543  }
544  for (int ob = 0; ob < output_batch_size; ob++) {
545  input_attachments.push_back(
546  {output_image->textures[os + ob], previousData[ob]});
547  }
548 
549  run(input_attachments,
550  {output_image->textures.begin() + os,
551  output_image->textures.begin() + os + output_batch_size},
552  [&]() {
553  glUniform2i(
554  outputSize->location,
555  output_image->texture_width,
556  output_image->texture_height);
557  glUniform2i(inputTileRange->location, 0, 1);
558  glUniform1i(accumulate->location, is != 0);
559  glUniform1i(
560  fusePRelu->location,
561  prelu_scale != nullptr &&
562  (is == input_slices - input_batch_size));
563  },
564  output_image->texture_width,
565  output_image->texture_height);
566  }
567  }
568  }

580  int input_tile_x = input_image->tile_x;
581  int input_tile_y = input_image->tile_y;
582  int input_tiles = input_image->tile_x * input_image->tile_y;
583  int output_tiles = output_image->tile_x * output_image->tile_y;
584

586  ib++, it += input_tile_chunk_size) {

588  // constructor
589  int binding_point = 0;
590

593  bias_block, binding_point++, [&](float16_t* data, size_t size) {

595  size,

597  "Bias buffer size too small");
598  for (int ob = 0; ob < geometry.output_channels; ob++) {
599  data[ob] = bias[ob];
600  }
601  });

603  // kernel weights
604  for (int ob = 0, ot = 0; ot < output_tiles;
605  ob++, ot += output_tile_chunk_size) {

607  kernel_block[ob],
608  binding_point++,
609  [&](float16_t* data, size_t size) {
610  CAFFE_ENFORCE_EQ(
611  size,
612  (4 * input_tile_chunk_size) * (4 * output_tile_chunk_size) *
613  geometry.kernel_size.y * geometry.kernel_size.x *
614  sizeof(float16_t),
615  "Kernel size mismatch");

617  data,
618  size,
619  input_image->channels,
620  output_image->channels,
621  {it, std::min(it + input_tile_chunk_size, input_tiles)},
622  {ot, std::min(ot + output_tile_chunk_size, output_tiles)});
623  });
624  }
625 
626  // PRelu scale
627  if (prelu_scale != nullptr && ib == input_tile_batch_size - 1) {
628  attach_uniform_buffer<float16_t>(
629  prelu_scale_block,
630  binding_point++,
631  [&](float16_t* data, size_t size) {
632  CAFFE_ENFORCE_GE(
633  size,
634  geometry.output_channels * sizeof(float16_t),
635  "PRelu buffer size too small");
636  for (int ob = 0; ob < geometry.output_channels; ob++) {
637  data[ob] = prelu_scale_size == geometry.output_channels
638  ? prelu_scale[ob]
639  : prelu_scale[0];
640  }
641  });
642  }
643 
644  std::vector<texture_attachment> input_attachments(
645  {{input_image->textures[0], inputData[0]},
646  {output_image->textures[0], previousData[0]}});
647 
648  run(input_attachments,
649  {output_image->textures[0]},
650  [&]() {
651  glUniform2i(
652  outputSize->location,
653  output_image->texture_width,
654  output_image->texture_height);
655  // [inputTileFrom, inputTileTo)
656  glUniform2i(
657  inputTileRange->location,
658  it,
659  std::min(it + input_tile_chunk_size, input_tiles));
660 
661  glUniform1i(accumulate->location, it != 0);
662  glUniform1i(
663  fusePRelu->location,
664  prelu_scale != nullptr && (ib == input_tile_batch_size - 1));
665  },
666  output_image->texture_width,
