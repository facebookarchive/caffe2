31 template <class T>
32 bool TryConvertingTensorRawValues(const TensorProto &onnx_tensor,
33  ::google::protobuf::RepeatedField<T> *field) {
34  if (!onnx_tensor.has_raw_data()) {
35  return false;
36  }
37 
38  size_t raw_size = onnx_tensor.raw_data().size();
39  assert(raw_size % sizeof(T) == 0);
40 
41  size_t num_elements = raw_size / sizeof(T);
42  const void *src_ptr =
43  static_cast<const void *>(onnx_tensor.raw_data().data());
44  field->Resize(num_elements, 0);
45  void *target_ptr = static_cast<void *>(field->mutable_data());
46  memcpy(target_ptr, src_ptr, raw_size);
47 
48  return true;
49 }

152  if (attr.has_f()) {

239  std::function<std::string(const std::string &)> mapper) const {
240  ::google::protobuf::RepeatedPtrField<caffe2::Argument> args;
241  for (const auto& kv: onnx_attrs_) {
242  // If the attribute was rewritten, we use it instead. Note that the
243  // rewritten attibute still has the unmapped name
244  const auto &attr = rewritten_onnx_attrs_.count(kv.first)
245  ? rewritten_onnx_attrs_.at(kv.first)
246  : (*kv.second);
247  auto* arg = args.Add();
248  arg->set_name(mapper(attr.name()));
249  CopyOnnxAttrValueToCaffe2Arg(arg, attr);
250  }
251  for (const auto& kv:rewritten_onnx_attrs_) {
252  // If rewritten attribute doesn't appear in the original attributes, this is
253  // a newlly added one and we need to add this to argument too
254  if (!onnx_attrs_.count(kv.first)) {
255  const auto& attr = kv.second;
256  auto* arg = args.Add();
257  arg->set_name(mapper(attr.name()));
258  CopyOnnxAttrValueToCaffe2Arg(arg, attr);
259  }
260  }
261 
262  return args;
263 }

371 // Note [Caffe2 ConvPoolOpBase]
372 // ~~~~~~~~~~~~~~~~~~~~~~~~~~~~
373 // To understand what is going on here, we have to talk a little bit about
374 // Caffe2's internals.
375 //
376 // First, it's important to know that all of Caffe2's pooling and convolution
377 // operators inherit from "ConvPoolOpBase", which is an abstract class that
378 // defines all of the attributes (kernels, dilations, strides, etc) which one
379 // sees on these operators. Unfortunately, Caffe2's documentation generator
380 // doesn't know how to handle cases like this, so for example, if you look at
381 // the docs for MaxPool at
382 // <https://caffe2.ai/docs/operators-catalogue.html#maxpool> you won't see any
383 // of the attributes. You have to go source diving to find the information; in
384 // particular, you want to look at:
385 // https://github.com/caffe2/caffe2/blob/master/caffe2/operators/conv_pool_op_base.h
386 // This class handles *global* pooling as well.
387 //
388 // Second, it's important to know what Caffe2 expects for padding, which can
389 // be somewhat difficult to understand from the code because Caffe2 handles
390 // both singular/pluralized spellings of padding, and there is also legacy
391 // padding business. The short version of the story is that, for NON-legacy
392 // padding (which is what we want to output), padding is expected to be
393 // *twice* the size of kernels. So if you have a 2D convolution, Caffe2
394 // will accept two values in 'kernels', but FOUR values in 'pads';
395 // furthermore, this is *mandatory.*
396 //
397 // Finally, ConvPoolOpBase is not the only class of it's kind; there is
398 // also ConvTransposeUnpoolBase, which backs ConvTranspose. So don't
399 // be tricked by the fact that Conv and ConvTranspose have similar
400 // parameters; they exercise different codepaths and need to be handled
401 // differently.
402 Caffe2Ops Caffe2Backend::CreateConvePoolOpBase(const ModelProto &init_model,
403  const ModelProto &pred_model,
404  OnnxNode *onnx_node,
405  int opset_version) {
406  const auto& node = onnx_node->node;
407  auto& attributes = onnx_node->attributes;
408  if (node.op_type().find("Global") == 0) {
409  auto* attr = attributes.AddRewrittenAttibute("global_pooling");
410  attr->set_i(1);
411  }
412 
413  if (attributes.HasAttribute("kernel_shape") && attributes.HasAttribute("pads")) {
414  auto kernel_shape =
415  attributes
416  .get<::google::protobuf::RepeatedField<::google::protobuf::int64>>(
417  "kernel_shape");
418  auto pads =
419  attributes
420  .get<::google::protobuf::RepeatedField<::google::protobuf::int64>>(
421  "pads");
422  if (kernel_shape.size() == pads.size()) {
423  // Caffe2 requires pads to be twice the size of kernels.
424  auto* attr = attributes.AddRewrittenAttibute("pads");
425  attr->mutable_ints()->CopyFrom(pads);
426  attr->mutable_ints()->MergeFrom(pads);
427  }
428  }
429 
430  return CommonOnnxNodeToCaffe2Ops(init_model, pred_model, onnx_node,
431  opset_version);
432 }

519 Caffe2Ops Caffe2Backend::CreateGemm(const ModelProto &init_model,
520  const ModelProto &pred_model,
521  OnnxNode *onnx_node, int opset_version) {
522  const auto& node = onnx_node->node;
523  if (node.input_size() < 3 || node.output_size() < 1) {
524  throw std::runtime_error("Caffe2 Gemm should have 3 inputs and 1 output");
525  }
526 
527  Caffe2Ops ret;
528  auto input_a = node.input(0);
529  auto input_b = node.input(1);
530  auto input_c = node.input(2);
531  auto output = node.output(0);
532 
533  auto alpha = onnx_node->attributes.get<float>("alpha", 1.0);
534  auto beta = onnx_node->attributes.get<float>("beta", 1.0);
535  if (!AlmostEqual(alpha, 1)) {
536  auto scaled_a = DummyName::NewDummyName();
537  caffe2::Argument scale;
538  scale.set_name("scale");
539  scale.set_f(alpha);
540 
541  auto *c2_op = ret.ops.Add();
542  BuildOperator(c2_op, "Scale", {input_a}, {scaled_a}, {scale});
543  input_a = scaled_a;
544  }
545  if (!AlmostEqual(beta, 1)) {
546  auto scaled_c = DummyName::NewDummyName();
547  caffe2::Argument scale;
548  scale.set_name("scale");
549  scale.set_f(beta);
550 
551  auto *c2_op = ret.ops.Add();
552  BuildOperator(c2_op, "Scale", {input_c}, {scaled_c}, {scale});
553  input_c = scaled_c;
554  }
555 
556  auto trans_a = onnx_node->attributes.get<int64_t>("transA", 0L);
557  auto trans_b = onnx_node->attributes.get<int64_t>("transB", 0L);
558  auto broadcast = onnx_node->attributes.get<int64_t>("broadcast", 0L);
559  if ( (!trans_a) && trans_b && broadcast) {
560  auto *c2_op = ret.ops.Add();
561  BuildOperator(c2_op, "FC", {input_a, input_b, input_c}, {output});
562  } else {
563  auto ab = DummyName::NewDummyName();
564  caffe2::Argument arg_trans_a;
565  arg_trans_a.set_name("trans_a");
566  arg_trans_a.set_i(trans_a);
567  caffe2::Argument arg_trans_b;
568  arg_trans_b.set_name("trans_b");
569  arg_trans_b.set_i(trans_b);
570  caffe2::Argument arg_broadcast;
571  arg_broadcast.set_name("broadcast");
572  arg_broadcast.set_i(broadcast);
573 
574  auto *c2_op = ret.ops.Add();
575  BuildOperator(c2_op, "MatMul", {input_a, input_b}, {ab},
576  {arg_trans_a, arg_trans_b});
577  c2_op = ret.ops.Add();
578  BuildOperator(c2_op, "Add", {ab, input_c}, {output},
579  {arg_broadcast});
580  }
581 
582  return ret;
583 }

589  auto& attributes = onnx_node->attributes;
590  ::google::protobuf::RepeatedField<::google::protobuf::int64> pads;
591  std::string pad_name = opset_version < 2 ? "paddings" : "pads";
592  pads = attributes
593  .get<::google::protobuf::RepeatedField<::google::protobuf::int64>>(
594  pad_name);
595  std::string str;
596  std::stringstream ss;
597  ss << "[";
598  for (const auto& i: pads) {
599  ss << i << ", ";
600  }
601  ss << "]";
602  str = ss.str();
603 
604  // Guard the invalid (negative) pads attribute.
605  for (const auto i : pads) {
606  if (i < 0) {
607  throw std::runtime_error(caffe2::MakeString(
608  "ONNX does not support negative pads in Pad, but get ", str));
609  }
610  }
611 
612  // first two dim is for batch and channel. Note that now all the values are
613  // non-negative
614  if (!(pads.size() == 8 &&
615  (pads.Get(0) + pads.Get(1) + pads.Get(4) + pads.Get(5) == 0))) {
616  throw std::runtime_error(caffe2::MakeString(
617  "Caffe2 only supports padding 2D Tensor, whereas padding is ", str));
618  }
619 
620  // rewrite the padding info
621  auto* attr = attributes.AddRewrittenAttibute(pad_name);
622  attr->add_ints(pads.Get(2));
623  attr->add_ints(pads.Get(3));
624  attr->add_ints(pads.Get(6));
625  attr->add_ints(pads.Get(7));
626 
627  return CommonOnnxNodeToCaffe2Ops(init_model, pred_model, onnx_node,
628  opset_version);
629 }

670 Caffe2Ops Caffe2Backend::CreateSlice(const ModelProto &init_model,
671  const ModelProto &pred_model,
672  OnnxNode *onnx_node,int opset_version) {
673  auto op_tmp = CommonOnnxNodeToCaffe2Ops(init_model, pred_model, onnx_node,
674  opset_version);
675  assert(op_tmp.ops.size() == 1);
676  auto* op = op_tmp.ops.Mutable(0);
677  std::unordered_map<std::string, caffe2::Argument*> args;
678  for (auto& arg: *op->mutable_arg()) {
679  args.emplace(arg.name(), &arg);
680  }
681 
682  caffe2::Argument starts_vals;
683  starts_vals.set_name("values");
684  auto pos = args.find("starts");
685  if (pos != args.end()) {
686  for (auto i: pos->second->ints()) {
687  starts_vals.add_ints(i);
688  }
689  args.erase(pos);
690  }
691 
692  caffe2::Argument ends_vals;
693  ends_vals.set_name("values");
694  pos = args.find("ends");
695  if (pos != args.end()) {
696  for (auto i: pos->second->ints()) {
697  ends_vals.add_ints(i < 0 ? i - 1 : i);
698  }
699  args.erase(pos);
700  }
701 
702  caffe2::Argument axes_vals;
703  axes_vals.set_name("values");
704  pos = args.find("axes");
705  if (pos != args.end()) {
706  for (auto i: pos->second->ints()) {
707  axes_vals.add_ints(i);
708  }
709  args.erase(pos);
710  } else {
711  auto ndim = starts_vals.ints_size();
712  for (int64_t i = 0; i < ndim; ++i) {
713  axes_vals.add_ints(i);
714  }
715  }
716 
717  assert(op->input_size() >= 1);
718  auto data = op->input(0);
719  auto shape_tensor = DummyName::NewDummyName();
720  Caffe2Ops ret;
721 
722  auto *c2_op = ret.ops.Add();
723  BuildOperator(c2_op, "Shape", {data}, {shape_tensor});
724 
725  auto axes_tensor = DummyName::NewDummyName();
726  c2_op = ret.ops.Add();
727  {
728  caffe2::Argument shape;
729  shape.set_name("shape");
730  shape.add_ints(axes_vals.ints_size());
731  BuildOperator(c2_op, "GivenTensorIntFill", {}, {axes_tensor},
732  {shape, axes_vals});
733  }
734 
735  auto starts_vals_tensor = DummyName::NewDummyName();
736  auto starts_tensor = DummyName::NewDummyName();
737  auto casted_starts_tensor = DummyName::NewDummyName();
738  c2_op = ret.ops.Add();
739  {
740  caffe2::Argument shape_starts;
741  shape_starts.set_name("shape");
742  shape_starts.add_ints(starts_vals.ints_size());
743  BuildOperator(c2_op, "GivenTensorInt64Fill", {}, {starts_vals_tensor},
744  {shape_starts, starts_vals});
745  }
746 
747  caffe2::Argument dtype;
748  dtype.set_name("dtype");
749  dtype.set_i(static_cast<int64_t>(caffe2::TensorProto::INT64));
750  caffe2::Argument constant;
751  constant.set_name("value");
752  constant.set_i(0);
753  c2_op = ret.ops.Add();
754  BuildOperator(c2_op, "ConstantFill", {shape_tensor}, {starts_tensor},
755  {dtype, constant});
756  c2_op = ret.ops.Add();
757  BuildOperator(c2_op, "ScatterAssign",
758  {starts_tensor, axes_tensor, starts_vals_tensor},
759  {starts_tensor});
760  // Slice only accepts starts as int
761  caffe2::Argument to;
762  to.set_name("to");
763  to.set_i(static_cast<int64_t>(caffe2::TensorProto::INT32));
764  c2_op = ret.ops.Add();
765  BuildOperator(c2_op, "Cast", {starts_tensor}, {casted_starts_tensor}, {to});
766 
767  auto ends_vals_tensor = DummyName::NewDummyName();
768  auto ends_tensor = DummyName::NewDummyName();
769  auto casted_ends_tensor = DummyName::NewDummyName();
770  c2_op = ret.ops.Add();
771  {
772  caffe2::Argument shape_ends;
773  shape_ends.set_name("shape");
774  shape_ends.add_ints(ends_vals.ints_size());
775  BuildOperator(c2_op, "GivenTensorInt64Fill", {}, {ends_vals_tensor},
776  {shape_ends, ends_vals});
777  }
778 
779  constant.set_i(-1);
780  c2_op = ret.ops.Add();
781  BuildOperator(c2_op, "ConstantFill", {shape_tensor}, {ends_tensor},
782  {dtype, constant});
783  c2_op = ret.ops.Add();
784  BuildOperator(c2_op, "ScatterAssign",
785  {ends_tensor, axes_tensor, ends_vals_tensor}, {ends_tensor});
786  // Slice only accepts ends as int
787  c2_op = ret.ops.Add();
788  BuildOperator(c2_op, "Cast", {ends_tensor}, {casted_ends_tensor}, {to});
789 
790  // attach the original op at the end
791  c2_op = ret.ops.Add();
792  c2_op->CopyFrom(*op);
793  c2_op->mutable_input()->Clear();
794  c2_op->add_input(data);
795  c2_op->add_input(casted_starts_tensor);
796  c2_op->add_input(casted_ends_tensor);
797  c2_op->mutable_arg()->Clear();
798  for (const auto& kv: args) {
799  c2_op->add_arg()->CopyFrom(*kv.second);
800  }
801 
802  return ret;
803 }

808 void Caffe2Backend::InplaceRewrite(GraphProto* graph) {
809  auto* nodes = graph->mutable_node();
810  auto renamed = InplaceRewrite(nodes);
811  for (int i = 0; i < graph->output_size(); ++i) {
812  graph->mutable_output(i)->set_name(LookUpWithDefault(
813  renamed, graph->output(i).name(), graph->output(i).name()));
814  }
815 }

817 // \brief currently we use this to translate ONNX-style
818 // consumed_input annotations to Caffe2-style in place
819 // updates (use same input and output names).
820 std::unordered_map<std::string, std::string> Caffe2Backend::InplaceRewrite(
821  ::google::protobuf::RepeatedPtrField<NodeProto> *nodes) {
822  std::unordered_map<std::string, std::string> renamed;
823  for (auto& node: *nodes) {
824  for (auto& input: *node.mutable_input()) {
825  input = LookUpWithDefault(renamed, input, input);
826  }
827 
828  // Get `consumed_inputs` attribute and remove it from node attributes
829  ::google::protobuf::RepeatedField<::google::protobuf::int64>
830  consumed_inputs;
831  int found_idx = -1;
832  for (int i = 0; i < node.attribute_size(); ++i) {
833  const auto& attr = node.attribute(i);
834  if (attr.name() == "consumed_inputs") {
835  consumed_inputs.CopyFrom(attr.ints());
836  found_idx = i;
837  break;
838  }
839  }
840  if (found_idx >= 0 && found_idx < node.attribute_size()) {
841  if (found_idx != node.attribute_size() - 1) {
842  node.mutable_attribute(found_idx)->CopyFrom(
843  node.attribute(node.attribute_size() - 1));
844  }
845  node.mutable_attribute()->RemoveLast();
846  }
847 
848  std::set<int> output_indexes;
849  for (int i = 0; i < node.output_size(); ++i) {
850  output_indexes.insert(i);
851  }
852  const auto& schema = GetOnnxSchema(node.op_type());
853 
854  int i = 0;
855  for (const auto& consumed: consumed_inputs) {
856  if (!consumed) {
857  ++i;
858  continue;
859  }
860 
861  auto output_idx = schema.consumed(i).second;
862  // consumed outputs are not always present
863  // for instance batch norm in test mode
864  // does not return the consumed inputs
865  if (output_idx < node.output_size()) {
866  output_indexes.erase(output_idx);
867  auto old_val = node.output(output_idx);
868  auto new_val = node.input(i);
869  node.set_output(output_idx, new_val);
870  renamed[old_val] = new_val;
871  }
872  ++i;
873  }
874 
875  for (auto idx: output_indexes) {
876  auto name = node.output(idx);
877  node.set_output(idx, LookUpWithDefault(renamed, name, name));
878  }
879  }
880 
881  return renamed;
882 }

906 // This translator performs the basic translation of ONNX nodes into
907 // Caffe2 operators. Besides doing a straightforward marshalling from
908 // one format to another, it also does these extra things:
909 //
910 // - Renames operators based on 'renamed_operators'
911 // - Renames attributes based on 'renamed_attrs' and
912 // 'get_per_op_renamed_attrs'
913 //
914 // If you're writing a custom translator, consider calling this first,
915 // and then fixing things up further.
916 Caffe2Ops Caffe2Backend::CommonOnnxNodeToCaffe2Ops(
917  const ModelProto &init_model, const ModelProto &pred_model,
918  OnnxNode *onnx_node, int opset_version) {
919  Caffe2Ops ret;
920  auto* c2_op = ret.ops.Add();
921 
922  const auto& node = onnx_node->node;
923  c2_op->mutable_input()->MergeFrom(node.input());
924  c2_op->mutable_output()->MergeFrom(node.output());
925  c2_op->set_name(node.name());
926 
927  const auto onnx_op_type = node.op_type();
928  auto broken_version = LookUpWithDefault(
929  get_broken_operators(), onnx_op_type, std::numeric_limits<int>::max());
930  if (broken_version <= opset_version) {
931  throw std::runtime_error(
932  caffe2::MakeString("Don't know how to translate op ", onnx_op_type,
933  " in ONNX operator set v", opset_version,
934  " (I only support prior to v", broken_version));
935  }
936  c2_op->set_type(
937  LookUpWithDefault(get_renamed_operators(), onnx_op_type, onnx_op_type));
938  if (!IsOperator(c2_op->type())) {
939  throw std::runtime_error(
940  caffe2::MakeString("Don't know how to translate op ", onnx_op_type));
941  }
942 
943  auto mapper = [&, this](const std::string& k) {
944  const auto it = get_per_op_renamed_attrs().find(onnx_op_type);
945  if (it != get_per_op_renamed_attrs().end()) {
946  const auto it_op = it->second.find(k);
947  if (it_op != it->second.end()) {
948  return it_op->second;
949  }
950  }
951  const auto it_global = get_renamed_attrs().find(k);
952  if (it_global != get_renamed_attrs().end()) {
953  return it_global->second;
954  }
955  return k;
956  };
957  c2_op->mutable_arg()->MergeFrom(
958  onnx_node->attributes.OnnxAttrToCaffe2Arg(mapper));
959 
960  return ret;
961 }

987 void Caffe2Backend::OnnxToCaffe2(
988  caffe2::NetDef *init_net, caffe2::NetDef *pred_net,
989  const ModelProto &onnx_model, const std::string &device,
990  int opset_version, bool include_initializers,
991  const std::vector<Caffe2Ops> &extras) {
992  auto device_option = GetDeviceOption(Device(device));
993 
994  ModelProto init_model = OptimizeOnnx(onnx_model, true);
995  InplaceRewrite(init_model.mutable_graph());
996 
997  ModelProto pred_model = OptimizeOnnx(onnx_model, false);
998  InplaceRewrite(pred_model.mutable_graph());
999 
1000  init_net->set_name(onnx_model.graph().name() + "_init");
1001  pred_net->set_name(onnx_model.graph().name() + "_predict");
1002 
1003  // Convert initializer if necessary
1004  if (include_initializers) {
1005  for (const auto& tp: onnx_model.graph().initializer()) {
1006  auto* c2_op = init_net->add_op();
1007  BuildTensorFillingOp(c2_op, tp);
1008  }
1009  }
1010 
1011  auto name_set =AllNamesInGraph(init_model.graph());
1012  auto name_set_pred = AllNamesInGraph(pred_model.graph());
1013  name_set.insert(name_set_pred.begin(), name_set_pred.end());
1014  DummyName::Reset(name_set);
1015 
1016  size_t idx_extra = 0;
1017  auto converter = [&](const ModelProto &model, caffe2::NetDef *net) mutable {
1018  net->mutable_device_option()->CopyFrom(device_option);
1019  for (const auto &node : model.graph().node()) {
1020  auto *init_net_tmp = include_initializers ? init_net : net;
1021  // For RNN operators, we rely on Python code to convert them for us, and
1022  // we simply deserilize the string. This is hack and eventually we want to
1023  // get rid of this to have one flow. Note that we need to update the dummy
1024  // name generator to avoid having duplicated names between Python and C++
1025  // generated dummies
1026  if (get_rnn_operators().count(node.op_type())) {
1027  if (idx_extra < extras.size()) {
1028  const auto &c2ops = extras[idx_extra++];
1029  for (const auto& op: c2ops.init_ops) {
1030  UpdateNames(op);
1031  }
1032  init_net_tmp->mutable_op()->MergeFrom(c2ops.init_ops);
1033  for (const auto& op: c2ops.ops) {
1034  UpdateNames(op);
1035  }
1036  net->mutable_op()->MergeFrom(c2ops.ops);
1037  for (const auto& input: c2ops.interface_blobs) {
1038  DummyName::AddName(input);
1039  }
1040  net->mutable_external_input()->MergeFrom(c2ops.interface_blobs);
1041  } else {
1042  throw std::runtime_error(
1043  caffe2::MakeString("Don't know how to convert ", node.op_type(),
1044  " without enough extra preconverted string"));
1045  }
1046  } else {
1047  auto onnx_node = OnnxNode(node);
1048  auto c2ops = OnnxNodeToCaffe2Ops(init_model, pred_model, &onnx_node,
1049  opset_version);
1050  init_net_tmp->mutable_op()->MergeFrom(c2ops.init_ops);
1051  net->mutable_op()->MergeFrom(c2ops.ops);
1052  net->mutable_external_input()->MergeFrom(c2ops.interface_blobs);
1053  }
1054  }
1055 
1056  for (const auto& value: model.graph().output()) {
1057  net->add_external_output(value.name());
1058  }
1059  for (const auto& value: model.graph().input()) {
1060  net->add_external_input(value.name());
1061  }
1062  };
1063 
1064  converter(init_model, init_net);
1065  converter(pred_model, pred_net);
1066 }

1068 Caffe2BackendRep*
1069 Caffe2Backend::Prepare(const std::string &onnx_model_str,
1070  const std::string &device,
1071  const std::vector<Caffe2Ops> &extras) {
1072  Caffe2BackendRep* rep = new Caffe2BackendRep();
1073  ModelProto onnx_model;
1074  ParseProtobufFromLargeString(onnx_model_str, &onnx_model);
1075  ONNX_NAMESPACE::checker::check_model(onnx_model);
1076 
1077  int opset_version = -1;
1078  for (const auto& imp: onnx_model.opset_import()) {
1079  if ((!imp.has_domain()) || imp.domain().empty()) {
1080  opset_version = imp.version();
1081  if (opset_version > kKnownOpsetVersion) {
1082  std::cout
1083  << "This version of onnx-caffe2 targets ONNX operator set version "
1084  << kKnownOpsetVersion
1085  << ", but the model we are trying to import uses version "
1086  << opset_version << ". We will try to import it anyway, "
1087  << "but if the model uses operators which had BC-breaking changes "
1088  "in the intervening versions, import will fail." << std::endl;
1089  }
1090  } else {
1091  std::cout << "Unrecognized operator set " << opset_version << std::endl;
1092  }
1093  }
1094  if (opset_version< 0) {
1095  if (onnx_model.ir_version() >= 0x00000003) {
1096  throw std::runtime_error(
1097  "Model with IR version >= 3 did not specify ONNX operator set "
1098  "version (onnx-caffe2 requires it)");
1099  } else {
1100  opset_version = 1;
1101  }
1102  }
1103 
1104  //TODO: avoid extra copy by directly feed initialiers to backend blobs
1105  OnnxToCaffe2(&rep->init_net(), &rep->pred_net(), onnx_model, device,
1106  opset_version, true, extras);
1107 
1108  // Get a list of uninitialized inputs to help with the inference setup
1109  auto& uninitialized_inputs = rep->uninitialized_inputs();
1110  std::unordered_set<std::string> initialized_inputs;
1111  for (const auto& tp: onnx_model.graph().initializer()) {
1112  initialized_inputs.emplace(tp.name());
1113  }
1114  for (const auto& input: onnx_model.graph().input()) {
1115  if (!initialized_inputs.count(input.name())) {
1116  uninitialized_inputs.emplace_back(input.name());
1117  }
1118  }
1119 
1120  return rep;
1121 }

1123 void Caffe2Backend::BuildTensorFillingOp(caffe2::OperatorDef *c2_op,
1124  const TensorProto &onnx_tensor,
1125  const std::string &name) {
1126  auto fill_name = name.empty() ? onnx_tensor.name() : name;
1127  CAFFE_ENFORCE(!fill_name.empty());
1128 
1129  if (onnx_tensor.has_segment()) {
1130  throw std::runtime_error("Currently not supporting loading segments.");
1131  }
1132 
1133  auto* c2_values = c2_op->add_arg();
1134  c2_values->set_name("values");
1135 
1136  if (onnx_tensor.data_type() == TensorProto::FLOAT) {
1137  c2_op->set_type("GivenTensorFill");
1138  auto* floats = c2_values->mutable_floats();
1139  if (!TryConvertingTensorRawValues<float>(onnx_tensor, floats)) {
1140  floats->CopyFrom(onnx_tensor.float_data());
1141  }
1142  } else if (onnx_tensor.data_type() == TensorProto::DOUBLE) {
1143  c2_op->set_type("GivenTensorDoubleFill");
1144  ::google::protobuf::RepeatedField<double> tmp;
1145  const ::google::protobuf::RepeatedField<double>* src = &tmp;
1146  if (!TryConvertingTensorRawValues<double>(onnx_tensor, &tmp)) {
1147  src = &onnx_tensor.double_data();
1148  } else {
1149  for (const auto i: *src) {
1150  c2_values->add_floats(i);
1151  }
1152  }
1153  } else if (onnx_tensor.data_type() == TensorProto::INT64) {
1154  c2_op->set_type("GivenTensorInt64Fill");
1155  auto *ints = c2_values->mutable_ints();
1156  if (!TryConvertingTensorRawValues<::google::protobuf::int64>(onnx_tensor,
1157  ints)) {
