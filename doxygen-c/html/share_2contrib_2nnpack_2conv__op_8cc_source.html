93 nnp_convolution_algorithm NNPACKConvOp::getConvolutionAlgorithm() const {
94  if (!OperatorBase::HasSingleArgumentOfType<std::string>("algo")) {
95  // No preference is stated. Heuristics for the best mobile device
96  // algorithm are different than NNPACK's version, as Winograd
97  // tends to be a lot faster. Use Winograd if the convolution
98  // is 3x3d1s1.
99  if (kernel_h() == 3 && kernel_w() == 3 && dilation_h() == 1 &&
100  dilation_w() == 1 && stride_h() == 1 && stride_w() == 1) {
101  // use Winograd
102  return nnp_convolution_algorithm_wt8x8;
103  }
104 
105  return nnp_convolution_algorithm_auto;
106  }
107 
108  // Otherwise, there is a preference.
109  auto algo = OperatorBase::GetSingleArgument<std::string>("algo", "AUTO");
110  if (algo == "AUTO") {
111  return nnp_convolution_algorithm_auto;
112  }
113  if (algo == "WINOGRAD") {
114  return nnp_convolution_algorithm_wt8x8;
115  }
116  if (algo == "WINOGRAD_FP16") {
117  return nnp_convolution_algorithm_wt8x8_fp16;
118  }
119  if (algo == "FT16") {
120  return nnp_convolution_algorithm_ft16x16;
121  }
122  if (algo == "FT8") {
123  return nnp_convolution_algorithm_ft8x8;
124  }
125  if (algo == "IMPLICIT_GEMM") {
126  return nnp_convolution_algorithm_implicit_gemm;
127  }
128  if (algo == "DIRECT") {
129  return nnp_convolution_algorithm_direct;
130  }
131  return nnp_convolution_algorithm_auto;
132 }

145 bool NNPACKConvOp::RunOnDeviceWithOrderNCHW() {
146  auto& X = Input(0);
147  auto& filter = Input(1);
148  auto* Y = Output(0);
149  CAFFE_ENFORCE(X.ndim() == 4, "Input dim should be 4");
150  const int N = X.dim32(0), C = X.dim32(1), H = X.dim32(2), W = X.dim32(3);
151  CAFFE_ENFORCE(filter.ndim() == 4, "");
152  const int M = filter.dim32(0);
153  CAFFE_ENFORCE(C % this->group_ == 0, "");
154  CAFFE_ENFORCE(M % this->group_ == 0, "");
155  CAFFE_ENFORCE(filter.dim32(1) == C / this->group_, "");
156  CAFFE_ENFORCE(filter.dim32(2) == kernel_h(), "");
157  CAFFE_ENFORCE(filter.dim32(3) == kernel_w(), "");
158  ConvPoolOpBase<CPUContext>::SetOutputSize(X, Y, filter.dim32(0));
159  const int oH = Y->dim32(2), oW = Y->dim32(3);
160 
161  const float* biasData = NULL;
162  if (InputSize() == 3) {
163  /* Convolution with bias */
164  auto& bias = Input(2);
165  CAFFE_ENFORCE(bias.ndim() == 1, "");
166  CAFFE_ENFORCE(bias.dim32(0) == M, "");
167  biasData = bias.template data<float>();
168  } else {
169  /* NNPACK interface requires bias. Use a dummy zero-filled vector. */
170  if (dummyBias_.size() != M) {
171  dummyBias_.resize(M);
172  }
173  biasData = dummyBias_.data();
174  }
175 
176  const size_t batch_size = X.dim32(0);
177  const size_t input_channels = X.dim32(1);
178  const size_t output_channels = Y->dim32(1);
179  const nnp_size input_size = {.width = static_cast<size_t>(X.dim32(3)),
180  .height = static_cast<size_t>(X.dim32(2))};
181  // filter is MCHW
182  const nnp_size kernel_size = {.width = static_cast<size_t>(filter.dim32(3)),
183  .height = static_cast<size_t>(filter.dim32(2))};
184  // pad is tblr
185  const nnp_padding padding = {.top = static_cast<size_t>(pad_t()),
186  .right = static_cast<size_t>(pad_r()),
187  .bottom = static_cast<size_t>(pad_b()),
188  .left = static_cast<size_t>(pad_l())};
189 
190  const nnp_size output_subsample = {.width = static_cast<size_t>(stride_w()),
191  .height = static_cast<size_t>(stride_h())};
192  initNNPACK();
193  pthreadpool pool(ws_->GetThreadPool());
194 
195  runWithSharedBuffer<CPUContext>(ws_, [&](Tensor<CPUContext>* buffer) {
196  if (transformStrategy_ == nnp_convolution_transform_strategy_precompute) {
197  transformedFilters_.resize(group_);
198 
199  size_t transformedFilterSize = 0;
200  nnp_status status = nnp_convolution_inference(
201  algorithm_,
202  nnp_convolution_transform_strategy_precompute,
203  C / group_,
204  M / group_,
205  input_size,
206  padding,
207  kernel_size,
208  output_subsample,
209  nullptr /* input */,
210  nullptr /* filters */,
211  nullptr /* bias */,
212  nullptr /* output */,
213  nullptr /* workspace buffer = transformed filter */,
214  &transformedFilterSize,
215  nnp_activation_identity,
216  nullptr /* activation parameter */,
217  &pool,
218  nullptr /* profile */);
219  if (status == nnp_status_success) {
220  /* For these convolution parameters filter transforms can be
221  * pre-computed */
222 
223  /* Division with rounding up, in case size is not multiple of
224  * sizeof(float) */
225  const size_t transformedFilterElements =
226  (transformedFilterSize + sizeof(float) - 1) / sizeof(float);
227 
228  for (auto g = 0; g < group_; g++) {
229  transformedFilters_[g] =
230  ws_->CreateBlob(
231  debug_def().name() + "_transformed_" + to_string(g))
232  ->GetMutable<TensorCPU>();
233  transformedFilters_[g]->Resize(transformedFilterElements);
234 
235  status = nnp_convolution_inference(
236  algorithm_,
237  nnp_convolution_transform_strategy_precompute,
238  C / group_,
239  M / group_,
240  input_size,
241  padding,
242  kernel_size,
243  output_subsample,
244  nullptr /* input */,
245  filter.template data<float>() + filter.size() / group_ * g,
246  nullptr /* bias */,
247  nullptr /* output */,
248  static_cast<void*>(
249  transformedFilters_[g]->template mutable_data<float>()),
250  &transformedFilterSize,
251  nnp_activation_identity,
252  nullptr /* activation parameter */,
253  &pool,
254  nullptr /* profile */);
255  CAFFE_ENFORCE(
256  nnp_status_success == status,
257  "NNPACK convolution filter pre-transformation return error");
258  }
259 
260  /*
261  * Now, we've precomputed all our filter transformations.
262  * Switch to reuse strategy to avoid doing transformation again on next
263  * iteration.
264  */
265  if (transformStrategy_ ==
266  nnp_convolution_transform_strategy_precompute) {
267  CAFFE_ENFORCE_EQ(transformedFilters_.size(), group_);
268  transformStrategy_ = nnp_convolution_transform_strategy_reuse;
269  }
270  } else {
271  LOG(WARNING)
272  << "Failed to query workspace size to precompute kernels, falling back to re-compute strategy";
273  transformStrategy_ = nnp_convolution_transform_strategy_compute;
274  }
275 
276  // Enforce when we leave this block that we have transitioned out of the
277  // precompute state.
278  CAFFE_ENFORCE(
279  transformStrategy_ != nnp_convolution_transform_strategy_precompute);
280  }
281 
282  CAFFE_ENFORCE(
283  transformStrategy_ == nnp_convolution_transform_strategy_reuse ||
284  transformStrategy_ == nnp_convolution_transform_strategy_compute);
285  const auto N = X.dim32(0);
286  for (auto n = 0; n < N; ++n) {
287  for (auto g = 0; g < group_; ++g) {
288  nnp_profile profile;
289  size_t workspaceSize = buffer->nbytes();
290  if (workspaceSize == 0) {
291  /* Allocate some memory to ensure buffer pointer is not NULL. This
292  * simplifies further logic. */
293  buffer->Resize(1);
294  workspaceSize = buffer->nbytes();
295  }
296  nnp_status status = nnp_convolution_inference(
297  algorithm_,
298  transformStrategy_,
299  C / group_,
300  M / group_,
301  input_size,
302  padding,
303  kernel_size,
304  output_subsample,
305  X.template data<float>() + n * C * H * W + g * H * W * (C / group_),
306  transformStrategy_ == nnp_convolution_transform_strategy_reuse
307  ? transformedFilters_[g]->template data<float>()
308  : filter.template data<float>() + filter.size() / group_ * g,
309  biasData + M / group_ * g,
310  Y->template mutable_data<float>() + n * oH * oW * M +
311  g * oH * oW * (M / group_),
312  static_cast<void*>(buffer->template mutable_data<float>()),
313  &workspaceSize,
314  nnp_activation_identity,
315  nullptr /* activation parameter */,
316  &pool,
317  FLAGS_caffe2_profile_nnpack ? &profile : nullptr);
318  if (status == nnp_status_insufficient_buffer) {
319  /* Query required workspace size, increase buffer, and try again */
320  status = nnp_convolution_inference(
321  algorithm_,
322  transformStrategy_,
323  C / group_,
324  M / group_,
325  input_size,
326  padding,
327  kernel_size,
328  output_subsample,
329  nullptr /* input */,
330  nullptr,
331  nullptr /* bias */,
332  nullptr /* output */,
333  nullptr /* workspace buffer */,
334  &workspaceSize,
335  nnp_activation_identity,
336  nullptr /* activation parameter */,
337  &pool,
338  nullptr /* profile */);
339  if (status == nnp_status_success) {
340  /* Division with rounding up, in case size is not multiple of
341  * sizeof(float) */
342  const size_t workspace_elements =
343  (workspaceSize + sizeof(float) - 1) / sizeof(float);
344  buffer->Resize(workspace_elements);
345 
346  /* Try convolution_inference again. If this time it fails, it is
347  * fatal. */
348  status = nnp_convolution_inference(
349  algorithm_,
350  transformStrategy_,
351  C / group_,
352  M / group_,
353  input_size,
354  padding,
355  kernel_size,
356  output_subsample,
357  X.template data<float>() + n * C * H * W +
358  g * H * W * (C / group_),
359  transformStrategy_ == nnp_convolution_transform_strategy_reuse
360  ? transformedFilters_[g]->template data<float>()
361  : filter.template data<float>() +
362  filter.size() / group_ * g,
363  biasData + M / group_ * g,
364  Y->template mutable_data<float>() + n * oH * oW * M +
365  g * oH * oW * (M / group_),
366  static_cast<void*>(buffer->template mutable_data<float>()),
367  &workspaceSize,
368  nnp_activation_identity,
369  nullptr /* activation parameter */,
370  &pool,
371  FLAGS_caffe2_profile_nnpack ? &profile : nullptr);
372  }
373  }
374 
375  VLOG(1) << "NNPACK buffer size: " << buffer->nbytes();
376  CAFFE_ENFORCE(
377  nnp_status_success == status,
378  "NNPACK convolution computation returned error");
379  if (FLAGS_caffe2_profile_nnpack) {
380  char buffer[1024];
381  const double gmacs =
382  double(
383  Y->dim32(2) * Y->dim32(3) * Y->dim32(1) * X.dim32(1) *
384  kernel_size.width * kernel_size.height / group_ / group_) /
385  1.0E9;
386  const double gflops = 2 * gmacs / profile.total;
387  auto ret = snprintf(
388  buffer,
389  sizeof(buffer),
390  "H: %3zu, W: %3zu, iC: %3zu, oC: %3zu, K: %1zu, S: %1zu, P: %1zu, GMACs: "
391  "%4.2f, totalT: %6.3f, inputT: %6.3f, "
392  "kernelT: %6.3f, blockT: %6.3f, outputT: %6.3f, GFLOPS: %6.3f",
393  size_t(X.dim(2)),
394  size_t(X.dim(3)),
395  size_t(X.dim(1)),
396  size_t(Y->dim(1)),
397  size_t(kernel_size.width),
398  size_t(output_subsample.width),
399  size_t(padding.top),
400  gmacs,
401  profile.total * 1E3,
402  profile.input_transform * 1E3,
403  profile.kernel_transform * 1E3,
404  profile.block_multiplication * 1E3,
405  profile.output_transform * 1E3,
406  gflops);
407  CAFFE_ENFORCE(ret > 0);
408  LOG(INFO) << buffer;
409  }
410  }
411  }
412  });
413  return true;
414 }
