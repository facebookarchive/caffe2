33 GLNet::GLNet(
34  const std::shared_ptr<const NetDef>& net_def,
35  Workspace* ws)
36  : NetBase(net_def, ws) {
37  ws_ = ws;
38  VLOG(1) << "Constructing GLNet " << net_def->name();
39  const bool net_def_has_device_option = net_def->has_device_option();
40  // Initialize the operators
41  for (int idx = 0; idx < net_def->op_size(); ++idx) {
42  const auto& operator_def = net_def->op(idx);
43  VLOG(1) << "Creating operator " << operator_def.name() << ": "
44  << operator_def.type();
45  output_blobs_.push_back(operator_def.output(0));
46  if (operator_def.has_device_option() && operator_def.device_option().device_type() == OPENGL) {
47  opengl_device_.push_back(true);
48  } else {
49  opengl_device_.push_back(false);
50  }
51 
52  std::unique_ptr<OperatorBase> op{nullptr};
53  if (!operator_def.has_device_option() && net_def_has_device_option) {
54  // In the case that the operator def does not specify a device option but
55  // the net def has a default option, we copy the device option over to the
56  // operator def.
57  OperatorDef temp_def(operator_def);
58  temp_def.mutable_device_option()->CopyFrom(net_def->device_option());
59  op = CreateOperator(temp_def, ws, idx);
60  } else {
61  op = CreateOperator(operator_def, ws, idx);
62  op->set_debug_def(
63  std::shared_ptr<const OperatorDef>{net_def, &(net_def->op(idx))});
64  }
65  operators_.emplace_back(std::move(op));
66  }
67 }

100 vector<float> GLNet::TEST_Benchmark(
101  const int warmup_runs,
102  const int main_runs,
103  const bool run_individual) {
104  LOG(INFO) << "Starting benchmark.";
105  LOG(INFO) << "Running warmup runs.";
106  CAFFE_ENFORCE(
107  warmup_runs >= 0,
108  "Number of warm up runs should be non negative, provided ",
109  warmup_runs,
110  ".");
111  for (int i = 0; i < warmup_runs; ++i) {
112  CAFFE_ENFORCE(Run(), "Warmup run ", i, " has failed.");
113  }
114 
115  auto last_blob = output_blobs_[output_blobs_.size() - 1];
116  Blob *gpu_out_blob = ws_->GetBlob(last_blob);
117  auto &g_ = gpu_out_blob->Get<GLTensor<half>>();
118  // Enforce gpu execution
119  g_.sync();
120 
121  LOG(INFO) << "Main runs.";
122  CAFFE_ENFORCE(
123  main_runs >= 0,
124  "Number of main runs should be non negative, provided ",
125  main_runs,
126  ".");
127  Timer timer;
128  for (int i = 0; i < main_runs; ++i) {
129  CAFFE_ENFORCE(Run(), "Main run ", i, " has failed.");
130  }
131  g_.sync();
132 
133  auto millis = timer.MilliSeconds();
134  LOG(INFO) << "[C2DEBUG] Main run finished. Milliseconds per iter: "
135  << millis / main_runs
136  << ". Iters per second: " << 1000.0 * main_runs / millis;
137 
138  vector<float> time_per_op(operators_.size(), 0);
139  vector<uint64_t> flops_per_op(operators_.size(), 0);
140  CaffeMap<string, float> time_per_op_type;
141  if (run_individual) {
142  for (int i = 0; i < main_runs; ++i) {
143  for (auto& op : operators_) {
144  op->ResetEvent();
145  }
146  int idx = 0;
147  for (auto& op : operators_) {
148  const string& op_type = op->debug_def().type();
149  if (i == 0) { // Gather flops on the first run.
150  auto* schema = OpSchemaRegistry::Schema(op_type);
151  if (schema && schema->HasCostInferenceFunction()) {
152  vector<TensorShape> shapes = op->InputTensorShapes();
153  flops_per_op[idx] =
154  schema->InferCost(op->debug_def(), shapes).flops;
155  }
156  }
157  timer.Start();
158  CAFFE_ENFORCE(
159  op->Run(),
160  "operator ",
161  op->debug_def().name(),
162  "(",
163  op_type,
164  ") has failed.");
165  if (opengl_device_[idx]) {
166  Blob *gpu_out_blob = ws_->GetBlob(output_blobs_[idx]);
167  auto &g_ = gpu_out_blob->Get<GLTensor<half>>();
168  g_.sync();
169  }
170  float spent = timer.MilliSeconds();
171  time_per_op[idx] += spent;
172  time_per_op_type[op_type] += spent;
173  ++idx;
174  }
175  }
176 
177  int idx = 0;
178  for (auto& op : operators_) {
179  const string& op_type = op->debug_def().type();
180  const string& print_name =
181  (op->debug_def().name().size()
182  ? op->debug_def().name()
183  : (op->debug_def().output_size() ? op->debug_def().output(0)
184  : "NO_OUTPUT"));
185  std::stringstream flops_str;
186  if (flops_per_op[idx]) {
187  flops_str << " ("
188  << to_string(1.0e-6 * flops_per_op[idx] / time_per_op[idx])
189  << " GFLOPS)";
190  }
191  LOG(INFO) << "[C2DEBUG] Operator #" << idx << " (" << print_name << ", " << op_type
192  << ") " << time_per_op[idx] / main_runs << " ms/iter"
193  << flops_str.str();
194  ++idx;
195  }
196  LOG(INFO) << "[C2DEBUG] Time per operator type:";
197  // sort by decreasing time spending.
198  std::vector<std::pair<string, float>> time_per_op_type_vec(
199  time_per_op_type.begin(), time_per_op_type.end());
200  std::sort(
201  time_per_op_type_vec.begin(),
202  time_per_op_type_vec.end(),
203  PairLargerThan<string, float>);
204  for (const auto& item : time_per_op_type_vec) {
205  LOG(INFO) << "[C2DEBUG] " << std::setw(15) << std::setfill(' ') << item.second / main_runs
206  << " " << item.first;
207  }
208  }
209  // We will reuse time_per_op to return the result of BenchmarkNet.
210  for (int i = 0; i < time_per_op.size(); ++i) {
211  time_per_op[i] /= main_runs;
212  }
213  time_per_op.insert(time_per_op.begin(), millis / main_runs);
214  return time_per_op;
215 }
