108 template <class Context>
109 bool SplitOp<Context>::RunOnDevice() {
110  auto& input = Input(0);
111  int canonical_axis = input.canonical_axis_index(axis_);
112  CAFFE_ENFORCE_LT(
113  canonical_axis, input.ndim(), "Axis not in input ndim range.");
114  const int input_channels = input.dim32(canonical_axis);
115  const int* axis_data;
116  vector<int> equal_split;
117  if (InputSize() == kSplitOpInputSize) {
118  // We obtain split from the input tensor.
119  CAFFE_ENFORCE_EQ(
120  split_.size(),
121  0,
122  "If you set split with an input blob, do not pass in "
123  "split in the argument.");
124  auto& split_tensor = OperatorBase::Input<TensorCPU>(1);
125  CAFFE_ENFORCE_EQ(split_tensor.size(), OutputSize());
126  axis_data = split_tensor.template data<int>();
127  } else if (split_.size() == 0) {
128  CAFFE_ENFORCE_EQ(
129  input_channels % OutputSize(),
130  0,
131  "If you did not specify split explicitly, the number of "
132  "input channels should be divisible by the output size.");
133  equal_split.resize(OutputSize(), input_channels / OutputSize());
134  axis_data = equal_split.data();
135  } else {
136  // We obtain split from the parameters.
137  CAFFE_ENFORCE_EQ(
138  split_.size(),
139  OutputSize(),
140  "The number of splits specified should be equal to the "
141  "number of outputs.");
142  axis_data = split_.data();
143  }
144 
145  CAFFE_ENFORCE_EQ(
146  add_axis_ ? OutputSize()
147  : std::accumulate(axis_data, axis_data + OutputSize(), 0),
148  input_channels,
149  "Sum of split dimensions do not match: should be ",
150  input_channels);
151  vector<TIndex> output_dims(input.dims());
152  int before = 1, after = 1;
153  for (int i = 0; i < canonical_axis; ++i) {
154  before *= input.dim32(i);
155  }
156  for (int i = canonical_axis + 1; i < input.ndim(); ++i) {
157  after *= input.dim32(i);
158  }
159  if (add_axis_) {
160  output_dims.erase(output_dims.begin() + canonical_axis);
161  }
162  size_t input_offset = 0;
163  for (int i = 0; i < OutputSize(); ++i) {
164  auto* output = Output(i);
165  auto axis_dim = add_axis_ ? 1 : axis_data[i];
166  if (!add_axis_) {
167  output_dims[canonical_axis] = axis_data[i];
168  }
169  output->Resize(output_dims);
170  math::CopyMatrix<Context>(
171  input.itemsize(),
172  before,
173  axis_dim * after,
174  static_cast<const char*>(input.raw_data()) + input_offset,
175  input.dim32(canonical_axis) * after,
176  output->raw_mutable_data(input.meta()),
177  axis_dim * after,
178  &context_,
179  input.meta().copy());
180  input_offset += axis_dim * after * input.itemsize();
181  }
182  return true;
183 }

185 template <class Context>
186 bool ConcatOp<Context>::RunOnDevice() {
187  auto* output = Output(0);
188  TensorCPU* split = OperatorBase::Output<TensorCPU>(1);
189  split->Resize(vector<TIndex>(1, InputSize()));
190  int* axis_data = split->template mutable_data<int>();
191  auto& input_zero = Input(0);
192  int adj_size = input_zero.ndim() + (add_axis_ ? 1 : 0);
193  int canonical_axis = canonical_axis_index_(axis_, adj_size);
194  CAFFE_ENFORCE_LT(canonical_axis, adj_size, "Axis not in input ndim range.");
195  for (int i = 1; i < InputSize(); ++i) {
196  CAFFE_ENFORCE(
197  Input(i).meta() == input_zero.meta(),
198  "All inputs must have the same type, expected: ",
199  input_zero.meta().name(),
200  " but got: ",
201  Input(i).meta().name(),
202  " for input: ",
203  i);
204  }
205 
206  int before = 1, after = 1;
207  vector<TIndex> output_dims(input_zero.dims());
208  for (int i = 0; i < input_zero.ndim(); ++i) {
209  if (i == canonical_axis && !add_axis_) {
210  continue;
211  }
212  int dim = input_zero.dim32(i);
213  if (i < canonical_axis) {
214  before *= dim;
215  } else { // i > canonical_axis || i == canonical_axis && add_axis_
216  after *= dim;
217  }
218  // check the input dims are compatible.
219  for (int j = 1; j < InputSize(); ++j) {
220  int dim_j = Input(j).dim32(i);
221  CAFFE_ENFORCE(
222  dim == dim_j,
223  "Expect dimension = ",
224  dim,
225  " got ",
226  dim_j,
227  " at axis = ",
228  i,
229  " for input: ",
230  j,
231  ". The input tensors can only have different dimensions "
232  "when arg 'add_axis' = 0 and along the axis = ",
233  canonical_axis,
234  " <",
235  Input(0).dims(),
236  "> vs <",
237  Input(j).dims(),
238  ">.");
239  }
240  }
241 
242  int output_channels = 0;
243  for (int i = 0; i < InputSize(); ++i) {
244  axis_data[i] = add_axis_ ? 1 : Input(i).dim32(canonical_axis);
245  output_channels += axis_data[i];
246  }
247  if (add_axis_) {
248  output_dims.insert(output_dims.begin() + canonical_axis, output_channels);
249  } else {
250  output_dims[canonical_axis] = output_channels;
251  }
252  output->Resize(output_dims);
253  size_t output_offset = 0;
254  for (int i = 0; i < InputSize(); ++i) {
255  auto& input = Input(i);
256  auto axis_dim = add_axis_ ? 1 : input.dim32(canonical_axis);
257  math::CopyMatrix<Context>(
258  input.itemsize(),
259  before,
260  axis_dim * after,
261  input.raw_data(),
262  axis_dim * after,
263  static_cast<char*>(output->raw_mutable_data(input_zero.meta())) +
264  output_offset,
265  output_channels * after,
266  &context_,
267  input_zero.meta().copy());
268  output_offset += axis_dim * after * input.itemsize();
269  }
270  return true;
271 }
