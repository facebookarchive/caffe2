187  template <class ContextForCopy>
188  void Extend(TIndex num, float growthPct, ContextForCopy* context) {
189  CAFFE_ENFORCE_GE_WITH_CALLER(dims_.size(), 1);
190  auto newDims = dims_;
191  newDims[0] += num;
192  if (!data_) {
193  Resize(newDims);
194  return;
195  }
196  auto newSize = std::accumulate(
197  newDims.begin(),
198  newDims.end(),
199  static_cast<TIndex>(1),
200  std::multiplies<TIndex>());
201  if (newSize * meta_.itemsize() <= capacity_) {
202  dims_ = newDims;
203  size_ = newSize;
204  return;
205  }
206  auto newCapacity = dims_;
207  newCapacity[0] = std::max<size_t>(
208  newDims[0], std::ceil(dims_[0] * (growthPct + 100) / 100));
209  Reserve(newCapacity, context);
210  dims_ = newDims;
211  size_ = newSize;
212  }

214  template <class T, class ContextForCopy>
215  void Reserve(const std::vector<T>& newCapacity, ContextForCopy* context) {
216  auto newSize = std::accumulate(
217  newCapacity.begin(),
218  newCapacity.end(),
219  static_cast<TIndex>(1),
220  std::multiplies<TIndex>());
221  if (newSize * meta_.itemsize() <= capacity_) {
222  return;
223  }
224  auto oldData = std::move(data_);
225  auto oldSize = size_;
226  auto oldDims = dims_;
227  Resize(newCapacity);
228  auto* newData = raw_mutable_data(meta_);
229  context->template CopyItems<ContextForCopy, ContextForCopy>(
230  meta_, oldSize, oldData.get(), newData);
231  dims_ = oldDims;
232  size_ = oldSize;
233  reserved_ = true;
234  }

268  template <typename... Ts>
269  void Resize(Ts... dim_source) {
270  bool size_changed = SetDims(dim_source...);
271  if (size_changed) {
272  // If needed, we will free the data. the next mutable_data() call
273  // will create the data storage.
274  int64_t new_size = size_ * meta_.itemsize();
275  bool reset_tensor = false;
276  if (reserved_) {
277  // If tensor is reserved then don't claim its memeory unless capacity_
278  // is smaller than new size
279  reset_tensor = capacity_ < new_size;
280  } else {
281  reset_tensor = capacity_ < new_size || !FLAGS_caffe2_keep_on_shrink ||
282  capacity_ - new_size > FLAGS_caffe2_max_keep_on_shrink_memory;
283  }
284 
285  if (reset_tensor) {
286  data_.reset();
287  capacity_ = 0;
288  // If reserved is true and we changed tensor memory then it is fine
289  // to switch it to false, if Resize is called from Reserve then
290  // reserved_
291  // will be set to true at end of Reserve()
292  reserved_ = false;
293  }
294  }
295  }

482  inline void* raw_mutable_data(const TypeMeta& meta) {
483  // For 0-size tensors it's fine to return any pointer (including nullptr)
484  if (meta_ == meta && (data_.get() || size_ == 0)) {
485  return data_.get();
486  } else {
487  meta_ = meta;
488  CAFFE_ENFORCE_WITH_CALLER(
489  size_ >= 0,
490  "Tensor is not initialized. You probably need to call Resize() "
491  "before calling mutable_data()");
492  if (size_ == 0) {
493  return data_.get();
494  }
495  if (meta.ctor()) {
496  // For types that need placement new, we will call it, as well as
497  // making sure that when the data is freed, it calls the right
498  // destruction procedure.
499  auto size = size_;
500  auto dtor = meta_.dtor();
501  data_.reset(
502  static_cast<void*>(Context::New(size_ * meta_.itemsize())),
503  [size, dtor](void* ptr) -> void {
504  dtor(ptr, size);
505  Context::Delete(ptr);
506  });
507  meta_.ctor()(data_.get(), size_);
508  } else {
509  // For fundamental type, new and delete is easier.
510  data_.reset(static_cast<void*>(Context::New(size_ * meta_.itemsize())),
511  Context::Delete);
512  }
513  capacity_ = size_ * meta_.itemsize();
514  return data_.get();
515  }
516  }

772 template <class T>
773 void TensorPrinter::Print(const Tensor<CPUContext>& tensor) {
774  std::stringstream values_stream;
775  // One most likely doesn't want to print int64-number of items for visual
776  // inspection, so we cast down to int here.
777  int total_count = std::min(tensor.size(), TIndex(limit_));
778  const T* tensor_data = tensor.template data<T>();
779  for (int i = 0; i < total_count - 1; ++i) {
780  values_stream << tensor_data[i] << ",";
781  }
782  // We do not add a comma after the last item.
783  values_stream << tensor_data[total_count - 1];
784  if (to_file_) {
785  (*log_file_) << MetaStr(tensor) << values_stream.str() << std::endl;
786  } else {
