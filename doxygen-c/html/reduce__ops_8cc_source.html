29 // For a Tensor X of n dimensions (dims[0], ..., dims[ndim-1]), given index
30 // is converted to corresponding n-dimensional index, e.g. for X.shape = (2,
31 // 3, 4) the linear index 12 maps to 3-dimensional index (1, 0, 0).
32 vector<TIndex> ConvertFromInputIndex(TIndex index, vector<TIndex>& dims) {
33  TIndex ndim = dims.size();
34  vector<TIndex> nd_idx(ndim);
35 
36  for (TIndex i = ndim - 1; i >= 0 && index > 0; i--) {
37  nd_idx[i] = index % dims[i];
38  index /= dims[i];
39  }
40  return nd_idx;
41 }

43 // For given n-dimensional index (nd_idx[0], ..., nd_idx[dims.size()-1]) and
44 // reduction axes, map the n-dimensional index to the corresponding linear
45 // index in the reduced tensor.
46 TIndex ConvertToOutputIndex(
47  const vector<int>& axes,
48  const vector<TIndex>& nd_idx,
49  vector<TIndex>& dims) {
50  TIndex index = 0;
51  TIndex multiplier = 1;
52  for (TIndex i = dims.size() - 1, j = axes.size() - 1; i >= 0; i--) {
53  if (j >= 0 && axes[j] == i) {
54  j--;
55  } else {
56  index += nd_idx[i] * multiplier;
57  multiplier *= dims[i];
58  }
59  }
60  return index;
61 }

68 template <typename T, class Context>
69 void ComputeOp(
70  const T* X_data,
71  const TIndex X_size,
72  vector<TIndex>& dims,
73  T* Y_data,
74  vector<int>& axes,
75  int keepdims,
76  T (*binary_op)(T, T)) {
77  for (TIndex x_idx = 0; x_idx < X_size; x_idx++) {
78  vector<TIndex> nd_idx = ConvertFromInputIndex(x_idx, dims);
79  TIndex y_idx = ConvertToOutputIndex(axes, nd_idx, dims);
80  Y_data[y_idx] = binary_op(Y_data[y_idx], X_data[x_idx]);
81  }
82 }

92 template <int DIMS>
93 DSizesType<DIMS> calcDSize(vector<TIndex>& dims) {
94  Eigen::DSizes<Eigen::DenseIndex, DIMS> dsizes_out;
95  size_t i = 0;
96  for (i = 0; i < DIMS; ++i) {
97  if (i < dims.size()) {
98  dsizes_out[i] = dims[i];
99  } else {
100  dsizes_out[i] = 1;
101  }
102  }
103  return dsizes_out;
104 }

108 template <typename T, class Context>
109 bool ReduceSumOp<T, Context>::Compute(
110  const T* X_data,
111  const TIndex X_size,
112  vector<TIndex>& dims,
113  T* Y_data,
114  const TIndex Y_size,
115  vector<int>& axes,
116  vector<TIndex>& Y_dims,
117  int keepdims) {
118  switch (dims.size()) {
119  case 1: {
120  std::array<int, 1> reduce_dims{{0}};
121  Eigen::DSizes<Eigen::DenseIndex, 1> dsizes_X = calcDSize<1>(dims);
122  Eigen::DSizes<Eigen::DenseIndex, 1> dsizes_Y = calcDSize<1>(Y_dims);
123  auto X_ten =
124  Eigen::TensorMap<ReductionTensor<const T, 1>>(X_data, dsizes_X);
125  auto Y_ten = Eigen::TensorMap<ReductionTensor<T, 1>>(Y_data, dsizes_Y);
126  Y_ten = X_ten.sum(reduce_dims);
127  } break;
128  case 2: {
129  Eigen::DSizes<Eigen::DenseIndex, 2> dsizes_X = calcDSize<2>(dims);
130  Eigen::DSizes<Eigen::DenseIndex, 2> dsizes_Y = calcDSize<2>(Y_dims);
131  auto X_ten =
132  Eigen::TensorMap<ReductionTensor<const T, 2>>(X_data, dsizes_X);
133  auto Y_ten = Eigen::TensorMap<ReductionTensor<T, 2>>(Y_data, dsizes_Y);
134  switch (axes.size()) {
135  case 1: {
136  std::array<int, 1> reduce_dims;
137  std::copy(axes.begin(), axes.end(), reduce_dims.begin());
138  Y_ten = X_ten.sum(reduce_dims);
139  } break;
140  case 2: {
141  std::array<int, 2> reduce_dims;
142  std::copy(axes.begin(), axes.end(), reduce_dims.begin());
143  Y_ten = X_ten.sum(reduce_dims);
144  } break;
145  }
146  } break;
147  case 3: {
148  Eigen::DSizes<Eigen::DenseIndex, 3> dsizes_X = calcDSize<3>(dims);
149  Eigen::DSizes<Eigen::DenseIndex, 3> dsizes_Y = calcDSize<3>(Y_dims);
150  auto X_ten =
151  Eigen::TensorMap<ReductionTensor<const T, 3>>(X_data, dsizes_X);
152  auto Y_ten = Eigen::TensorMap<ReductionTensor<T, 3>>(Y_data, dsizes_Y);
153  switch (axes.size()) {
154  case 1: {
155  std::array<int, 1> reduce_dims;
156  std::copy(axes.begin(), axes.end(), reduce_dims.begin());
157  Y_ten = X_ten.sum(reduce_dims);
158  } break;
159  case 2: {
160  std::array<int, 2> reduce_dims;
161  std::copy(axes.begin(), axes.end(), reduce_dims.begin());
162  Y_ten = X_ten.sum(reduce_dims);
163  } break;
164  case 3: {
165  std::array<int, 3> reduce_dims;
166  std::copy(axes.begin(), axes.end(), reduce_dims.begin());
167  Y_ten = X_ten.sum(reduce_dims);
168  }
169  }
170  } break;
171  default: {
172  math::Set<T, Context>(Y_size, 0.f, Y_data, &context_);
173  ComputeOp<T, Context>(X_data, X_size, dims, Y_data, axes, keepdims, Add);
174  }
175  }
176  return true;
177 }
